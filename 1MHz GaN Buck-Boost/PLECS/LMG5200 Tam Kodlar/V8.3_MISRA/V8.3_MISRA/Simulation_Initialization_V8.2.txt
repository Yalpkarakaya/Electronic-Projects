clear all; clc;

%% ═══════════════════════════════════════════════════════════════════════════════════════
%% SECTION 1: TARGET HARDWARE CONFIGURATION
%% ═══════════════════════════════════════════════════════════════════════════════════════
%% RATIONALE: Differentiates between STM32 hardware target and offline simulation.
%%            Hardware-specific parameters are loaded based on detected target.

target = plecs('get', './STM & Debug/Controller', 'CodeGenTarget');

if strcmp(target, 'STM32G4x') || strcmp(target, 'STM32G4')
    % Hardware deployment mode
    chip_type  = 'STM32G474RE';
    board_type = 'Nucleo-64';
    disp('✓ Target: STM32G474RE (Nucleo-64)');
    
elseif isempty(target) || strcmp(target, 'Generic')
    % Offline simulation mode
    chip_type  = 'Generic';
    board_type = 'Simulation';
    disp('→ Mode: Offline Simulation');
    
else
    error(['❌ Unsupported target: ', target]);
end


%% ═══════════════════════════════════════════════════════════════════════════════════════
%% SECTION 2: TIMING AND SWITCHING PARAMETERS
%% ═══════════════════════════════════════════════════════════════════════════════════════
%% RATIONALE: 1MHz switching frequency maximizes GaN efficiency while maintaining
%%            digital control stability. Simulation step 100x finer for accuracy.

fsw           = 1e6;                    % 1 MHz switching frequency [Hz]
Ts_controller = 1 / fsw;                % 1µs controller period [s]
Ts_sim        = Ts_controller / 40;     % 25ns simulation step [s] (V8.9: Memory optimization 100->40)


%% ═══════════════════════════════════════════════════════════════════════════════════════
%% SECTION 3: SYSTEM OPERATING SPECIFICATIONS
%% ═══════════════════════════════════════════════════════════════════════════════════════
%% RATIONALE: Military-grade 10-36V input range covers automotive, avionics, and
%%            portable military applications. 12V output standard for electronics.

% Input voltage range (MIL-STD-1275E compatible)
VIN_MIN     = 10.0;                     % Minimum input voltage [V]
VIN_MAX     = 36.0;                     % Maximum input voltage [V]
VIN_NOMINAL = 24.0;                     % Nominal input (for analysis) [V]

% Output specifications
VOUT_TARGET = 12.0;                     % Target output voltage [V]
IOUT_MAX    = 6.0;                      % Maximum output current [A]

% Input current limits

% Load resistance range

RLOAD_MAX   = 1000.0;                   % Maximum load (near no-load) [Ω]


%% ═══════════════════════════════════════════════════════════════════════════════════════
%% SECTION 4: PROTECTION THRESHOLDS
%% ═══════════════════════════════════════════════════════════════════════════════════════
%% RATIONALE: Multi-level protection prevents component damage while allowing
%%            short-term peak power capability (LMG5200 rated 40A peak/250µs).

% Inductor current limits (TITAN Section 2.2)
IIND_SOFT_LIMIT   = 10.0;               % TITAN: IL soft limit (225µs) [A]
IIND_HARD_LIMIT   = 35.0;               % TITAN: IL hard limit (<50ns) [A]
IIND_REDUCE_LIMIT = 30.0;               % Warning threshold [A]

% Peak and catastrophic limits (TITAN military spec)
IIND_PEAK_LIMIT   = 35.0;               % TITAN: IL hard limit [A]
IIND_PEAK_TIME    = 225e-6;             % TITAN: 225µs tolerance [s]
IIND_PEAK_WARNING = 30.0;               % Warning threshold [A]
IIND_CATASTROPHIC = 35.0;               % TITAN: = IL hard [A]

% Voltage protection
OVP_TRIP     = 14.5;                    % Output OVP trip [V]
OVP_ABSOLUTE = 15.5;                    % Absolute OVP (hardware) [V]
UVLO_MIN     = 9.0;                     % Under-voltage lockout [V]

% Duty cycle limits (V7.2 optimized for 12V target)
DUTY_MAX = 0.995;                       % Maximum duty (increased)
DUTY_MIN = 0.005;                       % Minimum duty (decreased)

% Integrator limits (V7.1 - sufficient for 8V→12V tracking)
INTEGRATOR_MAX =  0.35;                 % Positive integrator limit
INTEGRATOR_MIN = -0.30;                 % Negative integrator limit

% ─────────────────────────────────────────────────────────────────────────────────────────
% Freewheel/Brake Circuit (V8.2)
% RATIONALE: 5Ω brake resistor dissipates inductor energy during overvoltage/load dump
% ─────────────────────────────────────────────────────────────────────────────────────────
FREEWHEEL_BRAKE_RESISTANCE = 5.0;       % Brake resistor [Ω]
FREEWHEEL_BRAKE_POWER_MAX  = 30.0;      % Maximum dissipation [W]
FREEWHEEL_BRAKE_CURRENT_MAX = sqrt(FREEWHEEL_BRAKE_POWER_MAX / FREEWHEEL_BRAKE_RESISTANCE);  % ~2.45A
FREEWHEEL_BRAKE_VOLTAGE_MAX = 13.5;     % Maximum voltage across brake [V]


%% ═══════════════════════════════════════════════════════════════════════════════════════
%% SECTION 5: CIRCUIT COMPONENTS
%% ═══════════════════════════════════════════════════════════════════════════════════════
%% RATIONALE: Component values based on Coilcraft and TDK references.
%%            ESR values include frequency dependence at 1MHz.

% ─────────────────────────────────────────────────────────────────────────────────────────
% Inductor - 10µH (Coilcraft XAL1010 or equivalent)
% ─────────────────────────────────────────────────────────────────────────────────────────
L_inductance   = 10e-6;                 % Inductance [H]
L_dcr          = 0.010;                 % DC resistance [Ω]
L_saturation   = 12.0;                  % Saturation current [A]
L_parasitic_C  = 5e-12;                 % Parasitic capacitance [F]

% ─────────────────────────────────────────────────────────────────────────────────────────
% Input Capacitors (parallel ceramic + bulk)
% ─────────────────────────────────────────────────────────────────────────────────────────
Cin_ceramic     = 10e-9;                % Ceramic bypass [F]
Cin_bulk        = 47e-6;                % Polymer bulk [F]
Cin_esr_ceramic = 0.030;                % Ceramic ESR @ 1MHz [Ω]
Cin_esr_bulk    = 0.050;                % Polymer ESR [Ω]

% ─────────────────────────────────────────────────────────────────────────────────────────
% Output Capacitors (parallel ceramic + bulk)
% ─────────────────────────────────────────────────────────────────────────────────────────
Cout_ceramic     = 10e-9;               % Ceramic bypass [F]
Cout_bulk        = 220e-6;              % Polymer bulk [F]
Cout_esr_ceramic = 0.030;               % Ceramic ESR @ 1MHz [Ω]
Cout_esr_bulk    = 0.015;               % Polymer ESR [Ω]

% Calculated totals
Cout_total     = Cout_ceramic + Cout_bulk;
Cout_esr_total = (Cout_esr_ceramic * Cout_esr_bulk) / (Cout_esr_ceramic + Cout_esr_bulk);

% Ground plane parasitic
L_gnd_plane = 1e-9;                     % PCB parasitic inductance [H]


%% ═══════════════════════════════════════════════════════════════════════════════════════
%% SECTION 6: SENSING CIRCUITS
%% ═══════════════════════════════════════════════════════════════════════════════════════
%% RATIONALE: 12-bit ADC with proper scaling enables ±33A current sensing
%%            and 0-40V voltage sensing with adequate resolution.

ADC_resolution = 12;                    % 12-bit ADC
ADC_Vref       = 3.3;                   % Reference voltage [V]

% Voltage Sensing (resistor dividers)
% Vin: 10k/130k divider (R2/(R1+R2) = 10k/140k = 0.0714)
Vin_R1            = 130000;              % Upper resistor [Ω]
Vin_R2            = 10000;               % Lower resistor [Ω]
Vin_divider_ratio = Vin_R2 / (Vin_R1 + Vin_R2);
Vin_scale         = 1 / Vin_divider_ratio;   % = 14.0

% Vout: 10k/51k divider (R2/(R1+R2) = 10k/61k = 0.1639)
Vout_R1            = 51000;              % Upper resistor [Ω]
Vout_R2            = 10000;              % Lower resistor [Ω]
Vout_divider_ratio = Vout_R2 / (Vout_R1 + Vout_R2);
Vout_scale         = 1 / Vout_divider_ratio; % = 6.1

% Separate scale variables for Real Value subsystem visualization

VinReal_scale  = Vin_scale;              % 14.0x for Vin (10k/130k divider)
VoutReal_scale = Vout_scale;             % 6.1x for Vout (10k/51k divider)

% Current Sensing (INA240A2 + 1mΩ shunt)
% RATIONALE: INA240A2 provides 50x gain, enabling ±33A bidirectional sensing
Rs_shunt                = 1e-3;         % Shunt resistance [Ω]
INA240_gain             = 50;           % INA240A2 gain [V/V]
Im_sensitivity          = Rs_shunt * INA240_gain;   % 0.05 V/A
Im_offset               = ADC_Vref / 3.0;           % V8.11: Asymmetric offset (1.1V) - Neg: 0-1.1V, Pos: 1.1-3.3V
Im_max_unidirectional   = ADC_Vref / Im_sensitivity;  % 66A unidirectional
Im_max_negative         = Im_offset / Im_sensitivity;           % -22A (1.1V / 0.05)
Im_max_positive         = (ADC_Vref - Im_offset) / Im_sensitivity; % +44A (2.2V / 0.05)
Im_max_bidirectional    = Im_max_negative;            % V8.11: En düşük değeri kullan
Im_scale                = 1 / Im_sensitivity;

% Timing - 8ns dead-time

LMG_toff  = 3e-9;                       % Turn-off time [s]
