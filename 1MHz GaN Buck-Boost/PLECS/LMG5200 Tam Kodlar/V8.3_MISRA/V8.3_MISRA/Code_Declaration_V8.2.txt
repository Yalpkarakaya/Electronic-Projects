/**
 * @brief Clamps a float value between minimum and maximum bounds
 * @param x Input value
 * @param lo Lower bound
 * @param hi Upper bound
 * @return Clamped value
 */
static inline float Clamp_Float(float x, float lo, float hi)
{
    float result;
    if (x < lo)
    {
        result = lo;
    }
    else if (x > hi)
    {
        result = hi;
    }
    else
    {
        result = x;
    }
    return result;
}

/**
 * @brief Returns the minimum of two float values
 */
static inline float Min_Float(float a, float b)
{
    return (a < b) ? a : b;
}

/**
 * @brief Returns the maximum of two float values
 */
static inline float Max_Float(float a, float b)
{
    return (a > b) ? a : b;
}

/**
 * @brief Returns the absolute value of a float
 */
static inline float Abs_Float(float x)
{
    return (x < 0.0f) ? (-x) : x;
}

/**
 * @brief Returns the sign of a float (-1, 0, or 1)
 */
static inline int Sign_Float(float x)
{
    int result;
    if (x > 0.0f)
    {
        result = 1;
    }
    else if (x < 0.0f)
    {
        result = -1;
    }
    else
    {
        result = 0;
    }
    return result;
}

/**
 * @brief Linear interpolation between two values
 */
static inline float Lerp_Float(float a, float b, float t)
{
    return a + (t * (b - a));
}
