raw_iout = InputSignal(0, 4);

// V8.9: Hardware Fault Input (PLECS SR Flip-flop Q çıkışı)
// Bu sinyal analog koruma devresinden geliyor (IL > 35A tetikler)
// NOT: PLECS modelinde 6. giriş yoksa hw_fault = 0 (devre dışı)

float hw_fault = 0.0f;  // Devre dışı - PLECS'te 6. giriş eklenince yukarıdaki satırı aktifleştirin

// Sanity clamp (sensor range)
raw_vin  = Clamp_Float(raw_vin,  0.0f, 50.0f);
raw_vout = Clamp_Float(raw_vout, 0.0f, 20.0f);
raw_iin  = Clamp_Float(raw_iin,  -2.0f, 30.0f);
raw_iind = Clamp_Float(raw_iind, -5.0f, 50.0f);
raw_iout = Clamp_Float(raw_iout, -2.0f, 15.0f);

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// V7.10: 10ms STARTUP BLANKING - Vin okumasını yoksay, Vout'u koru
// Giriş bobininden kaynaklanan geçici yanlış okumaları önler
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
static float startup_blanking_timer = 0.0f;
static int startup_blanking_active = 0;  // V8.6: Blanking devre dışı (PLECS için) - Gerçek Vin okunsun

{
    if (startup_blanking_active) {
        startup_blanking_timer += TS;
        
        // 5ms boyunca Vin okuması yerine VIN_NOMINAL kullan (V8.4: 10ms→5ms hızlandırıldı)
        if (startup_blanking_timer < 0.005f) {
            // Gerçek Vin'i yoksay, sabit değer kullan
            raw_vin = VIN_NOMINAL;  // 15V sabit değer
            
            // Vout 12V'u geçmesin - duty'yi sınırla
            if (raw_vout > VOUT_TARGET) {  // Startup: pid_target_dynamic henüz yok
                // Overshoot algılandı - duty azalt sinyali
                // (emergency_duty_override bu aşamada henüz tanımlı değil,
                // ama filt_vin düşük tutarak PID doğal olarak duty'yi düşürecek)
            }
        } else {
            startup_blanking_active = 0;  // 5ms sonra normal moda geç
        }
    }
}

// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 1.5: INPUT FILTERING (CRITICAL MISSING BLOCK)                                ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝
// V8.6 Correction: Add missing filter calculations
{
    static int filter_init = 1;
    float alpha = 0.1f;  // Fast filter
    
    if (filter_init) {
        filt_vin = raw_vin;
        filt_vout = raw_vout;
        filt_iin = raw_iin;
        filt_iind = raw_iind;
        filt_iout = raw_iout;
        filter_init = 0;
    } else {
        filt_vin  += (raw_vin - filt_vin) * alpha;
        filt_vout += (raw_vout - filt_vout) * alpha;
        filt_iin  += (raw_iin - filt_iin) * alpha;
        filt_iind += (raw_iind - filt_iind) * alpha;
        filt_iout += (raw_iout - filt_iout) * alpha;
    }
    
    // V8.9: Seri direnç kompanzasyonu KALDIRILDI
    // VOUT_TARGET = 12.1V yapıldı, bu 0.1V düşümü telafi ediyor
    // Çift kompanzasyon hatası önlendi
}

// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                     MODULE 0.5: VIN SAFETY RANGE CHECK (V8.15)                                                     ║
// ║                     9V < Vin < 40V dışında sistem çalışmamalı - Freewheeling modu                                  ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝

static int vin_out_of_range = 0;
static int vin_range_ok_latched = 0;  // Hysteresis için

{
    // Hysteresis ile Vin aralık kontrolü
    if (!vin_range_ok_latched) {
        // Henüz aralığa girmedi - sıkı kontrol
        if (filt_vin >= (VIN_SAFE_MIN + VIN_HYSTERESIS) && 
            filt_vin <= (VIN_SAFE_MAX - VIN_HYSTERESIS)) {
            vin_range_ok_latched = 1;
            vin_out_of_range = 0;
        } else {
            vin_out_of_range = 1;
        }
    } else {
        // Aralıktaydı - hysteresis ile çık
        if (filt_vin < VIN_SAFE_MIN || filt_vin > VIN_SAFE_MAX) {
            vin_range_ok_latched = 0;
            vin_out_of_range = 1;
        } else {
            vin_out_of_range = 0;
        }
    }
    
    // Out-of-range durumunda FREEWHEELING zorla
    // Buck duty = 0 (Q1 OFF, Q2 ON)
    // Boost duty = 1 (Q3 ON, Q4 OFF)
    // Bu konfigürasyon induktörü kısa devre yapar, güvenli mod
}

static float vin_history[10] = {0};   // Son 10 Vin değeri
static int vin_history_idx = 0;       // Circular buffer index
static float vin_deriv = 0.0f;        // Vin türevi (V/µs)
static int vin_stable = 0;            // 1 = Vin stabil (türev ≈ 0)

// Charge Mode (Vin düştüğünde kapasitör şarjı)

// ═══════════════════════════════════════════════════════════════════════════════

static int freewheel_overcurrent_active = 0;

static float pid_target_dynamic = 13.0f;     // V8.10: Başlangıç: 13V (yüksüz)

static float debug_p_term = 0.0f;

// ═══════════════════════════════════════════════════════════════════════════════
// V8.4: GELİŞMİŞ VIN SLEW RATE KORUMA SİSTEMİ
// ═══════════════════════════════════════════════════════════════════════════════
static float vin_filtered = 24.0f;           // IIR filtreli Vin (başlangıç: nominal)
static int vin_slew_protection_active = 0;   // Histerezis: 1=koruma aktif
static float vin_slew_abs = 0.0f;            // |dVin/dt| mutlak değer

// ═══════════════════════════════════════════════════════════════════════════════
// V8.4: PREDİKTİF DUTY ÖN-YÜKLEME
// ═══════════════════════════════════════════════════════════════════════════════
static float duty_preload = 0.5f;            // Hesaplanan feedforward duty

// ═══════════════════════════════════════════════════════════════════════════════
// V8.6: FREEWHEEL STATE KAYIT SİSTEMİ
// ═══════════════════════════════════════════════════════════════════════════════
// RATIONALE: Freewheel girişinde duty değerleri kaydedilir, çıkışta restore edilir
// Bu sayede sistem freewheel sonrası önceki çalışma noktasına geri döner
static float saved_duty_buck_for_freewheel = 0.5f;     // Freewheel öncesi Buck duty
static float saved_duty_boost_for_freewheel = DUTY_MIN; // TITAN: Boost LO OFF
static int freewheel_prev_state = 0;                    // Edge detection için önceki durum

// ═══════════════════════════════════════════════════════════════════════════════
// TITAN Section 4.2: BOOTSTRAP REFRESH SYSTEM
// ═══════════════════════════════════════════════════════════════════════════════
// RATIONALE: High-side driver bootstrap kapasitör şarjını periyodik yenile.
//            1ms'de bir slave anahtarı 200ns kapatılır → bootstrap şarj edilir.
// KRITIK: Buck HO ve Boost HO için ayrı sayaçlar gerekli!
static int bootstrap_buck_counter = 0;                  // Buck HO refresh sayacı
static int bootstrap_boost_counter = 0;                 // Boost HO refresh sayacı
static int bootstrap_refresh_active = 0;                // Refresh pulse aktif mi?

{
    // ═══════════════════════════════════════════════════════════════════════════════
    // STAGE 0: V7.15 ADVANCED FREEWHEELING + SHUTDOWN PROTECTION
    // ═══════════════════════════════════════════════════════════════════════════════
    
    // Başlangıç sayacı - ilk birkaç cycle'da freewheeling'i atla
    static int init_cycles = 0;
    
    // İlk çalıştırma - TÜM değişkenleri açıkça başlat
    if (vin_prev_ff < 0.0f) {
        vin_prev_ff = raw_vin;
        /* [CRITICAL FIX C02]: Guard against divide-by-zero */
        if (raw_vin > 1.0f) {
            step_current_duty = pid_target_dynamic / raw_vin;
        } else {
            step_current_duty = 0.5f;  /* Safe default */
        }
        step_target_duty = step_current_duty;
        duty_before_cutoff = step_current_duty;
        init_cycles = 0;
        
        // KRITIK: Başlangıçta yük AYRIK olmalı!
        shutdown_active = 1;
        shutdown_latched = 0;
        startup_complete = 0;
        startup_timer = 0.0f;
        freewheeling_active = 0;
    }
    
    // İlk 10 cycle'ı say (başlangıç geçişi)
    if (init_cycles < 10) {
        init_cycles++;
        vin_prev_ff = raw_vin;  // Referansı güncelle ama freewheeling tetikleme
        shutdown_active = 1;     // Başlangıçta yük AYRIK kalmalı
    }
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // V8.6: FREEWHEEL EDGE DETECTION VE STATE MANAGEMENT
    // ═══════════════════════════════════════════════════════════════════════════════
    // Bu blok her cycle'da çalışır ve freewheel geçişlerini yönetir
    // RATIONALE: Freewheel girişinde duty kaydedilip 0'a çekilir,
    //            çıkışta eski duty değerine geri dönülür
    {
        // YÜKSELEN KENAR: Freewheel AKTİF OLUYOR (0 → 1)
        if (freewheeling_active && !freewheel_prev_state) {
            // ─────────────────────────────────────────────────────────────────────
            // AŞAMA 1: Mevcut duty değerlerini KAYDET (restore için)
            // ─────────────────────────────────────────────────────────────────────
            saved_duty_buck_for_freewheel = ss.duty_buck_current;
            saved_duty_boost_for_freewheel = ss.duty_boost_current;
            
            // ─────────────────────────────────────────────────────────────────────
            // AŞAMA 2: Freewheel duty değerlerini HEMEN uygula
            // ─────────────────────────────────────────────────────────────────────
            // duty = 0.0f → HO OFF, LO ON (PLECS modeli)
            ss.duty_buck_current = 0.0f;
            ss.duty_boost_current = 0.0f;
            
            // Emergency override'ı da set et (çift güvenlik)
            emergency_duty_override = -2.0f;
        }
        
        // DÜŞEN KENAR: Freewheel PASİF OLUYOR (1 → 0)
        else if (!freewheeling_active && freewheel_prev_state) {
            // ─────────────────────────────────────────────────────────────────────
            // AŞAMA 1: Kaydedilen duty değerlerini GERİ YÜKLE
            // ─────────────────────────────────────────────────────────────────────
            // NOT: Kullanıcı isteğine göre ESKİ değere dönülüyor!
            ss.duty_buck_current = saved_duty_buck_for_freewheel;
            ss.duty_boost_current = saved_duty_boost_for_freewheel;
            
            // ─────────────────────────────────────────────────────────────────────
            // AŞAMA 2: Emergency override'ı ve PID işaretçilerini güncelle
            // ─────────────────────────────────────────────────────────────────────
            emergency_duty_override = saved_duty_buck_for_freewheel;
            step_current_duty = saved_duty_buck_for_freewheel;
            step_target_duty = saved_duty_buck_for_freewheel;
        }
        
        // FREEWHEEL AKTİFKEN: Duty değerlerini sürekli 0'da tut
        // (PID veya başka kod duty'yi değiştirmiş olabilir)
        else if (freewheeling_active) {
            ss.duty_buck_current = 0.0f;
            ss.duty_boost_current = 0.0f;
        }
        
        // Önceki durumu güncelle (sonraki cycle için)
        freewheel_prev_state = freewheeling_active;
    }
    
    // ─────────────────────────────────────────────────────────────────────────────
    // BÖLÜM 1: VIN HISTORY BUFFER + DERIVATIVE HESABI
    // ─────────────────────────────────────────────────────────────────────────────
    // Son 10 Vin değerini sakla ve türev hesapla
    // NOT: vin_delta kaldırıldı - V8.4'te vin_deriv kullanılıyor
    vin_prev_ff = raw_vin;
    
    // Circular buffer güncelle
    vin_history[vin_history_idx] = raw_vin;
    vin_history_idx = (vin_history_idx + 1) % 10;
    
    // Türev hesabı: (en yeni - en eski) / 10µs
    // NOT: 10 sample * TS = 10µs (TS = 1µs)
    // vin_deriv [V/µs] = ΔV / 10µs (çünkü 10 sample'lık pencere)
    int oldest_idx = vin_history_idx;  // En eski değer
    float vin_oldest = vin_history[oldest_idx];
    float vin_newest = raw_vin;
    // ΔV / 10 = V per sample, ama biz V/µs istiyoruz
    // 10 sample = 10µs, yani dV / 10 * (1000000/10) = dV * 100000 V/s
    // veya basitçe: vin_deriv = dV / 10 [V/µs] (çünkü 10µs pencere)
    vin_deriv = (vin_newest - vin_oldest) / 10.0f;  // V/µs (yaklaşık)
    
    // Vin stabil mi? (|türev| < 0.3 V/µs = stabil)
    // 0.3 V/µs = 300,000 V/s = çok hızlı değişim
    vin_stable = (Abs_Float(vin_deriv) < 0.3f) ? 1 : 0;
    
    // ─────────────────────────────────────────────────────────────────────────────
    // BÖLÜM 2: V8.4 GELİŞMİŞ VIN SLEW RATE KORUMA SİSTEMİ
    // ─────────────────────────────────────────────────────────────────────────────
    // Özellikler:
    //   1. Histerezis eşikler (osilasyon önleme)
    //   2. IIR filtrelenmiş Vin (tracking için)
    //   3. Tek yönlü değil, mutlak değer tabanlı algılama
    //   4. Akıllı recovery (dinamik)
    
    // İlk 10µs'de FALSE TRİGGER önleme
    if (init_cycles >= 10) {
        
        // ═══════════════════════════════════════════════════════════════════════
        // V8.4: IIR FİLTRELİ VIN (Tracking için)
        // ═══════════════════════════════════════════════════════════════════════
        // Yavaş filtre: ani Vin değişimlerini yumuşatır
        #if VIN_TRACKING_ENABLE
        vin_filtered = vin_filtered * (1.0f - VIN_TRACKING_FILTER_ALPHA) + 
                       raw_vin * VIN_TRACKING_FILTER_ALPHA;
        #else
        vin_filtered = raw_vin;
        #endif
        
        // ═══════════════════════════════════════════════════════════════════════
        // V8.4: HİSTEREZİS TABANLI SLEW DETECTION
        // ═══════════════════════════════════════════════════════════════════════
        // Aktivasyon: |dV/dt| > VIN_SLEW_THRESHOLD_HIGH (2V/µs)
        // Deaktivasyon: |dV/dt| < VIN_SLEW_THRESHOLD_LOW (1V/µs)
        vin_slew_abs = Abs_Float(vin_deriv);
        
        if (!vin_slew_protection_active) {
            // Koruma pasif - aktivasyon kontrolü
            // V8.4 FIX: Yük BAĞLI DEĞİLKEN düşük akımda freewheel gereksiz
            // Yük BAĞLIYKEN Vin slew yeterli - hızlı koruma gerekli
            int load_connected = (load_state == LOAD_STATE_YUKE_BAGLI);
            int inductor_current_ok = load_connected ? 1 : (Abs_Float(raw_iind) > 5.0f);
            
            if (vin_slew_abs >= VIN_SLEW_THRESHOLD_HIGH && inductor_current_ok) {
                vin_slew_protection_active = 1;
                freewheel_timer = 0.0f;
                stable_timer = 0.0f;
                
                // Yön tespiti
                if (vin_deriv > 0) {
                    // VIN ARTIYOR → Freewheeling
                    freewheeling_active = 1;
                    charge_mode = 0;
                    vin_step_phase = 1;
                    emergency_duty_override = -2.0f;
                    // V8.6: Duty atamaları artık edge detection bloğunda yapılıyor
                } else {
                    // VIN AZALIYOR → Charge mode
                    freewheeling_active = 0;
                    charge_mode = 1;
                    vin_step_phase = -1;
                }
                emergency_active = 1;
            }
        } else {
            // Koruma aktif - deaktivasyon kontrolü (histerezis)
            if (vin_slew_abs < VIN_SLEW_THRESHOLD_LOW) {
                // Histerezis alt sınırına düştü - recovery kontrol et
                // (Recovery mantığı BÖLÜM 3'te)
            }
        }
        
        // ═══════════════════════════════════════════════════════════════════════
        // SENARYO C: BOBİN AKIMI ÇOK YÜKSEK (>25A) → ACİL FREEWHEELING
        // ═══════════════════════════════════════════════════════════════════════
        // Vin slew tespitinden bağımsız - 70A spike önleme
        if (raw_iind > 25.0f && !freewheeling_active) {
            // Aşırı bobin akımı - acil freewheel
            freewheeling_active = 1;
            charge_mode = 0;
            vin_slew_protection_active = 1;
            freewheel_timer = 0.0f;
            stable_timer = 0.0f;
            vin_step_phase = 1;
            
            emergency_duty_override = -2.0f;
            emergency_active = 1;
            // V8.6: Duty atamaları artık edge detection bloğunda yapılıyor
        }
    }
    
    // ─────────────────────────────────────────────────────────────────────────────
    // BÖLÜM 3: FREEWHEELING/CHARGE → NORMAL GEÇİŞ
    // ─────────────────────────────────────────────────────────────────────────────
    // Koşul: Vin stabil (türev ≈ 0) VE Vout >= 12.5V VE stabil süre yeterli
    if (freewheeling_active || charge_mode) {
        freewheel_timer += TS;
        
        // Stabil timer - Vin stabilse say
        if (vin_stable) {
            stable_timer += TS;
        } else {
            stable_timer = 0.0f;
        }
        
        // RECOVERY KOŞULLARI (V8.4: GEVŞETİLDİ):
        // 1. Vin stabil (son 10 değer türevi ≈ 0)
        // 2. Stabil süre >= 50µs (anlık spike değil, gerçek stabilite)
        // 3. Bobin akımı < 10A (güvenli seviye) - V8.4: 5A→10A gevşetildi
        // 4. Overvoltage yok (Vout < 40V)
        // NOT: Vout aralığı KALDIRILDI - düşük Vout'ta da recovery yapılabilir
        int inductor_safe = (Abs_Float(raw_iind) < 10.0f);
        int no_overvoltage = (raw_vout < 40.0f);
        int stable_enough = (stable_timer >= 0.000050f);  // 50µs minimum stabil
        
        if (vin_stable && stable_enough && inductor_safe && no_overvoltage) {
            // Hedef duty hesapla (vin_step_phase'e göre güvenlik marjı)
            step_target_duty = pid_target_dynamic / raw_vin;
            if (vin_step_phase > 0) {
                step_target_duty *= 0.90f;  // Vin yükseldi → duty daha düşük tut
            } else {
                step_target_duty *= 1.05f;  // Vin düştü → duty biraz yüksek tut
            }
            step_target_duty = Clamp_Float(step_target_duty, DUTY_MIN, DUTY_MAX);
            
            // RECOVERY - Freewheel'den çıkış
            // V8.6 REVİZYON: Duty restore işlemi edge detection bloğunda yapılıyor
            //               Burada sadece flag'ler sıfırlanıyor
            freewheeling_active = 0;
            charge_mode = 0;
            emergency_active = 0;
            vin_slew_protection_active = 0;  // V8.4: Histerezis state sıfırla
            emergency_timer = 0.0f;
            stable_timer = 0.0f;
            
            // V8.4: PID integratör sıfırla - temiz başlangıç
            vout_integrator = 0.0f;
            integrator = 0.0f;
        }
        
        // Max süre aşıldı → ZORLA NORMAL MODA GEÇ (güvenli veya değil)
        // V8.4: 50ms'de takılı kalmayı önle - PID integratör sıfırla ve yeniden başla
        if (freewheel_timer >= 0.050f) {  // 50ms max
            // Freewheel modunu kapat
            freewheeling_active = 0;
            charge_mode = 0;
            vin_slew_protection_active = 0;
            emergency_active = 0;  // V8.4: Emergency mode'u da kapat!
            stable_timer = 0.0f;
            freewheel_timer = 0.0f;
            
            // V8.4: PID integratör sıfırla ve duty'yi feedforward ile hesapla
            // Bu sayede sistem 5V'da takılı kalmaz, yeniden hedef voltaja ulaşır
            vout_integrator = 0.0f;  // Integratör sıfırla
            integrator = 0.0f;       // V8.4: Ana integratör de sıfırla
            step_target_duty = pid_target_dynamic / raw_vin;
            step_target_duty = Clamp_Float(step_target_duty, DUTY_MIN, DUTY_MAX);
            emergency_duty_override = step_target_duty;  // Duty'yi feedforward değerine set et
            step_current_duty = step_target_duty;
            
            // V8.4 KRİTİK: ss.duty_buck_current'ı da güncelle - bu final output'a gidiyor!
            ss.duty_buck_current = step_target_duty;
            ss.duty_boost_current = DUTY_MIN;  /* TITAN: Boost LO OFF (Buck mode) */
        }
    }
    
    // ─────────────────────────────────────────────────────────────────────────────
    // BÖLÜM 3.5: V8.2 ADAPTIVE dV/dt PROTECTION SYSTEM
    // ─────────────────────────────────────────────────────────────────────────────
    // DSP-based protection with:
    //   1. Cascaded MA + IIR filter (noise rejection)
    //   2. Adaptive threshold based on load and Vout margin
    //   3. Bidirectional detection (rising and falling Vin)
    //   4. Debounced trip with recovery logic
    
    // Initialize on first run
    if (!dvdt_prot_initialized) {
        // Initialize filter buffers with current Vin
        for (int i = 0; i < DVDT_MA_WINDOW_SIZE; i++) {
            dvdt_prot.ma_buffer[i] = raw_vin;
            dvdt_prot.slope_history[i] = raw_vin;
        }
        dvdt_prot.ma_index = 0;
        dvdt_prot.ma_sum = raw_vin * DVDT_MA_WINDOW_SIZE;
        dvdt_prot.iir_state = raw_vin;
        dvdt_prot.slope_index = 0;
        dvdt_prot.dvdt = 0.0f;
        dvdt_prot.tripped = 0;
        dvdt_prot.latch_count = 0;
        dvdt_prot.recovery_countdown = 0;
        dvdt_prot.protection_active = 0;
        dvdt_prot_initialized = 1;
    }
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // STAGE 1: DIGITAL FILTER (Cascaded MA + IIR)
    // ═══════════════════════════════════════════════════════════════════════════════
    #if DVDT_FILTER_ENABLED
    // Moving Average: Remove oldest, add newest
    dvdt_prot.ma_sum -= dvdt_prot.ma_buffer[dvdt_prot.ma_index];
    dvdt_prot.ma_buffer[dvdt_prot.ma_index] = raw_vin;
    dvdt_prot.ma_sum += raw_vin;
    dvdt_prot.ma_index = (dvdt_prot.ma_index + 1) % DVDT_MA_WINDOW_SIZE;
    
    float ma_output = dvdt_prot.ma_sum / (float)DVDT_MA_WINDOW_SIZE;
    
    // 1st Order IIR: y[n] = α * x[n] + (1-α) * y[n-1], α = 0.5
    dvdt_prot.iir_state = 0.5f * ma_output + 0.5f * dvdt_prot.iir_state;
    dvdt_prot.vin_filtered = dvdt_prot.iir_state;
    #else
    dvdt_prot.vin_filtered = raw_vin;
    #endif
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // STAGE 2: SLOPE CALCULATION (over k samples)
    // ═══════════════════════════════════════════════════════════════════════════════
    // Get oldest sample from history
    int oldest_slope_idx = dvdt_prot.slope_index;
    float vin_old_slope = dvdt_prot.slope_history[oldest_slope_idx];
    
    // Store new sample
    dvdt_prot.slope_history[dvdt_prot.slope_index] = dvdt_prot.vin_filtered;
    dvdt_prot.slope_index = (dvdt_prot.slope_index + 1) % DVDT_SLOPE_WINDOW;
    
    // dV/dt = (V_new - V_old) / (k × T_sample)
    // k=4, T_sample=1µs → divide by 4µs → multiply by 0.25
    /* [CRITICAL FIX C01]: Guard against divide-by-zero */
    float divisor = (float)DVDT_SLOPE_WINDOW * TS * 1e6f;
    if (divisor > 1e-10f) {
        dvdt_prot.dvdt = (dvdt_prot.vin_filtered - vin_old_slope) / divisor;
    } else {
        dvdt_prot.dvdt = 0.0f;  /* Safe default when divisor too small */
    }
    dvdt_prot.dvdt_abs = (dvdt_prot.dvdt < 0.0f) ? -dvdt_prot.dvdt : dvdt_prot.dvdt;
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // STAGE 3: ADAPTIVE THRESHOLD LOOKUP
    // ═══════════════════════════════════════════════════════════════════════════════
    // Determine load index based on inductor current (proxy for load current)
    float abs_iload = (raw_iind < 0.0f) ? -raw_iind : raw_iind;
    int load_idx;
    if (abs_iload < 1.0f) {
        load_idx = 0;
    } else if (abs_iload < 3.0f) {
        load_idx = 1;
    } else if (abs_iload < 5.0f) {
        load_idx = 2;
    } else {
        load_idx = 3;
    }
    
    // Determine margin index based on Vout deviation from target
    float vout_margin = raw_vout - pid_target_dynamic;
    float abs_margin = (vout_margin < 0.0f) ? -vout_margin : vout_margin;
    int margin_idx;
    if (abs_margin < 0.5f) {
        margin_idx = 0;
    } else if (abs_margin < 1.0f) {
        margin_idx = 1;
    } else if (abs_margin < 2.0f) {
        margin_idx = 2;
    } else {
        margin_idx = 3;
    }
    
    // Lookup threshold from LUT
    dvdt_prot.load_index = load_idx;
    dvdt_prot.margin_index = margin_idx;
    dvdt_prot.threshold_current = DVdt_Threshold_LUT[load_idx][margin_idx];
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // STAGE 4: THRESHOLD COMPARISON & TRIP DECISION
    // ═══════════════════════════════════════════════════════════════════════════════
    #if DVDT_BIDIRECTIONAL
    int threshold_exceeded = (dvdt_prot.dvdt_abs > dvdt_prot.threshold_current) ? 1 : 0;
    #else
    // Rising edge only
    int threshold_exceeded = (dvdt_prot.dvdt > dvdt_prot.threshold_current) ? 1 : 0;
    #endif
    
    if (threshold_exceeded) {
        dvdt_prot.latch_count++;
        
        // Debounce: Require consecutive trips to avoid noise-induced false trips
        if (dvdt_prot.latch_count >= DVDT_TRIP_DEBOUNCE_COUNT) {
            // TRIP ACTION
            if (!dvdt_prot.tripped) {
                // First trip - record diagnostic info
                dvdt_prot.slope_at_trip = dvdt_prot.dvdt;
                dvdt_prot.threshold_at_trip = dvdt_prot.threshold_current;
                dvdt_prot.total_trip_count++;
            }
            
            dvdt_prot.tripped = 1;
            dvdt_prot.recovery_countdown = DVDT_RECOVERY_CYCLES;
            dvdt_prot.protection_active = 1;
            vin_slew_protection_active = 1;  // V8.4: Histerezis state senkronize
            
            // Trigger freewheeling mode
            // V8.4 FIX: Yük bağlı değilken düşük akımda gereksiz freewheel önle
            int dvdt_load_connected = (load_state == LOAD_STATE_YUKE_BAGLI);
            int dvdt_current_ok = dvdt_load_connected ? 1 : (Abs_Float(raw_iind) > 5.0f);
            
            if (!freewheeling_active && dvdt_prot.dvdt > 0.0f && dvdt_current_ok) {
                // Rising Vin → Freewheeling (only if load connected OR current high)
                freewheeling_active = 1;
                emergency_duty_override = -2.0f;
                emergency_active = 1;
            } else if (!charge_mode && dvdt_prot.dvdt < 0.0f && dvdt_current_ok) {
                // Falling Vin → Charge mode (only if load connected OR current high)
                charge_mode = 1;
                emergency_active = 1;
            }
            
            // NOT: V8.3'te shutdown V8.3 state machine tarafından kontrol ediliyor
            // Eski shutdown_active = 1 kaldırıldı
        }
    } else {
        dvdt_prot.latch_count = 0;
        
        // Recovery logic
        if (dvdt_prot.tripped) {
            if (dvdt_prot.recovery_countdown > 0) {
                dvdt_prot.recovery_countdown--;
            } else {
                // Recovery complete
                dvdt_prot.tripped = 0;
                dvdt_prot.protection_active = 0;
                // NOT: vin_slew_protection_active BÖLÜM 3'te sıfırlanıyor
            }
        }
    }
    
    // ─────────────────────────────────────────────────────────────────────────────
    // BÖLÜM 4: V8.0 GELİŞMİŞ YÜK KORUMA SİSTEMİ
    // ─────────────────────────────────────────────────────────────────────────────
    // Strateji:
    //   1. Vout < 10.5V → HEMEN AYIR (yük koruma)
    //   2. Vout > 13.5V → HEMEN AYIR (yük koruma)
    //   3. Freewheeling/Charge mode aktif → AYRIK KAL
    //   4. Vout 12.3V-12.7V VE Vin stabil VE 100µs stabil → BAĞLA
    //   5. İlk bağlantı 12.5V'da, sonraki bağlantılar da aynı aralıkta
    #if SHUTDOWN_ENABLE
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // VOUT DURUM KONTROLÜ (Yeni threshold'lar)
    // ═══════════════════════════════════════════════════════════════════════════════
    int vout_disconnect_low = (raw_vout < VOUT_LOAD_DISCONNECT_LOW);    // < 10.5V
    int vout_connect_range = (raw_vout >= VOUT_LOAD_CONNECT_MIN) && 
                             (raw_vout <= VOUT_LOAD_CONNECT_MAX);       // 12.3V-12.7V
    // NOT: vout_disconnect_high (>13.5V) V8.3'te kullanılmıyor - 14V hedefinde cycling yapıyordu

    
    // ═══════════════════════════════════════════════════════════════════════════════
    // LATCH MEKANİZMASI - Osilasyon önleme
    // ═══════════════════════════════════════════════════════════════════════════════
    // Shutdown tetiklendiğinde 50µs minimum bekle (hızlı açma/kapama önleme)
    if (shutdown_latched) {
        shutdown_latch_timer += TS;
        if (shutdown_latch_timer >= SHUTDOWN_LATCH_TIME) {  // 50µs
            shutdown_latched = 0;
            shutdown_latch_timer = 0.0f;
        }
    }
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // STARTUP TRAKİNG - Sistem stabil olduğunda işaretle
    // ═══════════════════════════════════════════════════════════════════════════════
    if (!startup_complete) {
        if (vout_connect_range && !freewheeling_active && !charge_mode) {
            startup_timer += TS;
            if (startup_timer >= 0.001f) {  // 1ms stabil = startup tamamlandı
                startup_complete = 1;
            }
        } else {
            startup_timer = 0.0f;  // Stabil değilse timer sıfırla
        }
    }
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // V8.3: VOUT AVERAGING (100 Sample Rolling Average)
    // ═══════════════════════════════════════════════════════════════════════════════
    // Rapid cycling önlemek için Vout_avg kullanılır (state geçişlerinde)
    
    // Buffer'dan en eski değeri çıkar
    vout_avg_sum -= vout_avg_buffer[vout_avg_index];
    // Yeni değeri ekle
    vout_avg_buffer[vout_avg_index] = raw_vout;
    vout_avg_sum += raw_vout;
    // Index'i güncelle
    vout_avg_index = (vout_avg_index + 1) % VOUT_AVG_SAMPLE_COUNT;
    
    // İlk 100 sample dolunca flag set et
    if (!vout_avg_ready && vout_avg_index == 0) {
        vout_avg_ready = 1;
    }
    
    // Ortalamayı hesapla
    if (vout_avg_ready) {
        vout_avg = vout_avg_sum / (float)VOUT_AVG_SAMPLE_COUNT;
    } else {
        vout_avg = raw_vout;  // Buffer dolmadan anlık değer kullan
    }
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // V8.3: VOUT PEAK-TO-PEAK TRACKING (100µs periyod)
    // ═══════════════════════════════════════════════════════════════════════════════
    // Osilasyon kontrolü: %5'ten fazla osilasyon istenmiyor
    
    // Min/max güncelle
    if (raw_vout < vout_pp_min) vout_pp_min = raw_vout;
    if (raw_vout > vout_pp_max) vout_pp_max = raw_vout;
    vout_pp_sample_count++;
    
    // Her 100 sample'da (100µs) peak-to-peak hesapla ve sıfırla
    if (vout_pp_sample_count >= (int)PP_MEASUREMENT_PERIOD) {
        vout_pp_value = vout_pp_max - vout_pp_min;
        // Sonraki periyod için sıfırla
        vout_pp_min = raw_vout;
        vout_pp_max = raw_vout;
        vout_pp_sample_count = 0;
    }
    
    // Osilasyon %5'ten fazla mı? (hedef voltaja göre)
    float vout_pp_max_allowed = pid_target_dynamic * VOUT_PP_MAX_PERCENT;
    int vout_oscillation_ok = (vout_pp_value <= vout_pp_max_allowed);
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // V8.3: VIN STABILITY TRACKING (5 Ortalama Karşılaştırması + Peak-to-Peak)
    // ═══════════════════════════════════════════════════════════════════════════════
    // Koşullar:
    //   1. Vin 10V-36V aralığında
    //   2. Son 5 ortalama (500µs) arası değişim < 1V
    //   3. Peak-to-peak < 2V
    
    // Vin peak-to-peak tracking (aynı periyod vout ile)
    if (raw_vin < vin_pp_min) vin_pp_min = raw_vin;
    if (raw_vin > vin_pp_max) vin_pp_max = raw_vin;
    
    // Her 100 sample'da (100µs) peak-to-peak hesapla ve sıfırla
    if (vout_pp_sample_count == 0) {  // Vout ile senkron (yukarıda sıfırlandı)
        vin_pp_value = vin_pp_max - vin_pp_min;
        // Sonraki periyod için sıfırla
        vin_pp_min = raw_vin;
        vin_pp_max = raw_vin;
    }
    
    // Buffer'dan en eski değeri çıkar
    vin_avg_sum -= vin_avg_buffer[vin_avg_index];
    // Yeni değeri ekle
    vin_avg_buffer[vin_avg_index] = raw_vin;
    vin_avg_sum += raw_vin;
    // Index'i güncelle
    vin_avg_index = (vin_avg_index + 1) % VIN_STABILITY_AVG_COUNT;
    vin_avg_cycle_count++;
    
    // Her 100 sample'da bir ortalama kaydet
    if (vin_avg_cycle_count >= (int)VIN_STABILITY_AVG_COUNT) {
        vin_avg_cycle_count = 0;
        float current_vin_avg = vin_avg_sum / (float)VIN_STABILITY_AVG_COUNT;
        
        // History buffer'a ekle
        vin_stability_history[vin_stability_idx] = current_vin_avg;
        vin_stability_idx = (vin_stability_idx + 1) % VIN_STABILITY_HISTORY_COUNT;
    }
    
    // Vin stabilite kontrolü: Son 5 ortalama arasındaki max fark
    float vin_min = vin_stability_history[0];
    float vin_max = vin_stability_history[0];
    for (int i = 1; i < (int)VIN_STABILITY_HISTORY_COUNT; i++) {
        if (vin_stability_history[i] < vin_min) vin_min = vin_stability_history[i];
        if (vin_stability_history[i] > vin_max) vin_max = vin_stability_history[i];
    }
    float vin_avg_variation = (vin_max - vin_min);  // Ortalama değişimi
    float vin_avg_value = (vin_max + vin_min) / 2.0f;
    
    // V8.3: 3 koşul kontrolü
    int vin_in_absolute_range = (raw_vin >= VIN_ABSOLUTE_MIN && raw_vin <= VIN_ABSOLUTE_MAX);
    int vin_avg_change_ok = (vin_avg_variation <= VIN_AVG_CHANGE_MAX);  // < 1V
    int vin_pp_ok = (vin_pp_value <= VIN_PP_MAX_ABSOLUTE);              // < 2V
    int vin_is_stable_v83 = (vin_in_absolute_range && vin_avg_change_ok && vin_pp_ok);
    
    // Compiler warning suppression (vout_oscillation_ok gelecekte kullanılacak)
    (void)vout_oscillation_ok;
    (void)vin_avg_value;
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // V8.3: FREEWHEEL OVERCURRENT DETECTION (225µs Timer)
    // ═══════════════════════════════════════════════════════════════════════════════
    // Iind > 10A sürekli 225µs → Freewheel aktif + YUKSUZ state'e zorla
    
    if (Abs_Float(raw_iind) > FREEWHEEL_IIND_THRESHOLD_V83) {
        freewheel_overcurrent_timer += TS;
        
        if (freewheel_overcurrent_timer >= FREEWHEEL_ACTIVATION_TIME) {
            // 225µs aşıldı → Freewheel AKTİF
            freewheel_overcurrent_active = 1;
            freewheeling_active = 1;
            emergency_duty_override = -2.0f;  // TÜM PWM'ler OFF
            // V8.6: Duty atamaları artık edge detection bloğunda yapılıyor
            
            // State'i YUKSUZ'a zorla
            load_state = LOAD_STATE_YUKSUZ;
            load_state_timer = 0.0f;
        }
    } else {
        // Akım düştü - timer sıfırla ama freewheel henüz kapatma
        freewheel_overcurrent_timer = 0.0f;
    }
    
    // Freewheel deaktivasyon kontrolü
    if (freewheel_overcurrent_active) {
        // Deaktivasyon koşulları: Iind < 10A VE Vin stabil
        if (Abs_Float(raw_iind) < FREEWHEEL_IIND_THRESHOLD_V83 && vin_is_stable_v83) {
            freewheel_overcurrent_active = 0;
            freewheeling_active = 0;
            emergency_duty_override = 0.0f;
            freewheel_overcurrent_timer = 0.0f;
        }
    }
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // V8.3: 3-STATE LOAD MACHINE
    // ═══════════════════════════════════════════════════════════════════════════════
    // YUKSUZ (0): Başlangıç, Shutdown=LOW, Target=14V
    // YUKE_HAZIR (1): Yüke hazır, Shutdown=HIGH, Target=14V  
    // YUKE_BAGLI (2): Yük bağlı, Shutdown=HIGH, Target=12V (ramp)
    // ═══════════════════════════════════════════════════════════════════════════════
    
    // State timer güncelle
    load_state_timer += TS;
    
    switch (load_state) {
        
        // ─────────────────────────────────────────────────────────────────────────
        case LOAD_STATE_YUKSUZ:
        // ─────────────────────────────────────────────────────────────────────────
        {
            // Shutdown = LOW (yük ayrık)
            shutdown_active = 1;
            
            // PID hedef = 14V
            pid_target_dynamic = PID_TARGET_UNLOADED;
            
            // Undervoltage timer - çok düşük voltajda kalma süresi
            if (vout_disconnect_low) {
                undervolt_timer += TS;
                // Uzun süre undervoltage → latch mekanizması aktifle
                if (undervolt_timer >= UNDERVOLT_TIMEOUT) {
                    shutdown_latched = 1;
                    shutdown_latch_timer = 0.0f;
                }
            } else {
                undervolt_timer = 0.0f;
            }
            
            // Geçiş kontrolü: Vout_avg ∈ [13.5V, 14.5V] → YUKE_HAZIR
            // shutdown_stable_timer ile stabilite kontrolü
            // first_connect_done: İlk bağlantı için daha uzun stabilite süresi
            float required_stable = first_connect_done ? 
                                   SHUTDOWN_STABLE_TIME : 
                                   (SHUTDOWN_STABLE_TIME * 2.0f);
            
            if (vout_avg >= VOUT_READY_MIN && vout_avg <= VOUT_READY_MAX) {
                shutdown_stable_timer += TS;
                // Freewheel aktif değilse ve stabil ise geçiş yap
                if (!freewheel_overcurrent_active && !freewheeling_active && 
                    shutdown_stable_timer >= required_stable && !shutdown_latched) {
                    load_state = LOAD_STATE_YUKE_HAZIR;
                    load_state_timer = 0.0f;
                    shutdown_stable_timer = 0.0f;
                }
            } else {
                shutdown_stable_timer = 0.0f;
            }
        }
        break;
        
        // ─────────────────────────────────────────────────────────────────────────
        case LOAD_STATE_YUKE_HAZIR:
        // ─────────────────────────────────────────────────────────────────────────
        {
            // Shutdown = HIGH (yük bağlanabilir)
            shutdown_active = 0;
            
            // PID hedef = 14V (henüz yük yok)
            pid_target_dynamic = PID_TARGET_UNLOADED;
            
            // ═══════════════════════════════════════════════════════════════════
            // V8.4: PREDİKTİF DUTY HESAPLAMA
            // ═══════════════════════════════════════════════════════════════════
            // Yük bağlanmadan ÖNCE gereken duty'yi hesapla
            // Bu değer integratöre ön-yüklenecek (PID sıfırdan başlamayacak)
            #if DUTY_PRELOAD_ENABLE
            if (vin_filtered > 1.0f) {  // Sıfıra bölme koruması
                duty_preload = PID_TARGET_LOADED / vin_filtered;
                duty_preload *= DUTY_PRELOAD_MARGIN;  // +15% güvenlik marjı
                duty_preload = Clamp_Float(duty_preload, DUTY_MIN, DUTY_MAX);
            }
            #endif
            
            // Vout aralık kontrolü - 14V hedef için 13.8V-14.2V aralığında kalmalı
            int vout_in_ready_range = (raw_vout >= VOUT_READY_MIN && raw_vout <= VOUT_READY_MAX);
            
            // NOT: vout_disconnect_high (>13.5V) kullanılmıyor çünkü 14V hedefinde
            // her zaman true olur ve cycling'e sebep olur!
            // Bunun yerine VOUT_READY_MAX (14.2V) üst sınır olarak kullanılıyor.
            
            // Yük algılama: Iout >= 0.1A → YUKE_BAGLI
            if (raw_iout >= IOUT_LOAD_DETECT_THRESHOLD) {
                load_state = LOAD_STATE_YUKE_BAGLI;
                load_state_timer = 0.0f;
                vout_out_of_range_timer = 0.0f;  // KRITIK: Timer sıfırla!
                first_connect_done = 1;  // İlk yük bağlantısı yapıldı
                
                // ═══════════════════════════════════════════════════════════════
                // V8.4: İNTEGRATÖR ÖN-YÜKLEME
                // ═══════════════════════════════════════════════════════════════
                // PID sıfırdan başlamasın - hesaplanan duty ile başlasın
                #if DUTY_PRELOAD_ENABLE
                vout_integrator = duty_preload;
                #endif
            }
            // Vout aralık dışına çıktı → YUKSUZ (hem alt hem üst sınır)
            else if (!vout_in_ready_range) {
                load_state = LOAD_STATE_YUKSUZ;
                load_state_timer = 0.0f;
            }
            // Freewheel aktif → YUKSUZ
            else if (freewheel_overcurrent_active || freewheeling_active) {
                load_state = LOAD_STATE_YUKSUZ;
                load_state_timer = 0.0f;
            }
        }
        break;
        
        // ─────────────────────────────────────────────────────────────────────────
        case LOAD_STATE_YUKE_BAGLI:
        // ─────────────────────────────────────────────────────────────────────────
        {
            // Shutdown = HIGH (yük bağlı)
            shutdown_active = 0;
            
            // PID hedef = 12V (ramp ile geçiş)
            if (pid_target_dynamic > PID_TARGET_LOADED) {
                // 250µs'de 14V → 12V = 8000 V/s ramp
                pid_target_dynamic -= PID_RAMP_RATE * TS;
                if (pid_target_dynamic < PID_TARGET_LOADED) {
                    pid_target_dynamic = PID_TARGET_LOADED;
                }
            }
            
            // ═══════════════════════════════════════════════════════════════════
            // V8.3: 3-FAZLI TOLERANS SİSTEMİ
            // ═══════════════════════════════════════════════════════════════════
            // Faz 1: Ramp (0-250µs) → 8V-14.25V (geniş tolerans)
            // Faz 2: Geçiş (250-750µs) → 10V-14V (orta tolerans)
            // Faz 3: Steady (750µs+) → 11V-13V (sıkı tolerans)
            
            int vout_in_range = 0;
            float vout_min_limit = 0.0f;
            float vout_max_limit = 0.0f;
            
            if (load_state_timer < LOADED_PHASE1_TIME) {
                // Faz 1: Ramp sırasında
                vout_min_limit = LOADED_PHASE1_MIN;
                vout_max_limit = LOADED_PHASE1_MAX;
            } else if (load_state_timer < LOADED_PHASE2_TIME) {
                // Faz 2: Geçiş (hedef 12V'ye ulaştı)
                vout_min_limit = LOADED_PHASE2_MIN;
                vout_max_limit = LOADED_PHASE2_MAX;
            } else {
                // Faz 3: Steady-state
                vout_min_limit = LOADED_PHASE3_MIN;
                vout_max_limit = LOADED_PHASE3_MAX;
            }
            
            vout_in_range = (raw_vout >= vout_min_limit && raw_vout <= vout_max_limit);
            
            // ═══════════════════════════════════════════════════════════════════
            // V8.3: 100µs DEBOUNCE - Sınır dışında kalma süresi
            // ═══════════════════════════════════════════════════════════════════
            // Anlık spike'lar state değişikliği tetiklememeli
            
            if (!vout_in_range) {
                vout_out_of_range_timer += TS;
            } else {
                vout_out_of_range_timer = 0.0f;  // Aralığa geri döndü
            }
            
            int should_disconnect = (vout_out_of_range_timer >= VOUT_DEBOUNCE_TIME);
            
            // Vin stabilitesi de kontrol edilmeli (steady-state'te)
            // NOT: vin_is_stable_v83 yerine mevcut vin_stable kullanıldı (KRITIK FIX)
            int vin_stable_for_operation = (load_state_timer < LOADED_PHASE2_TIME) || vin_stable;
            
            // Yük kaybı: Iout < 0.1A → YUKE_HAZIR
            if (raw_iout < IOUT_LOAD_DETECT_THRESHOLD) {
                load_state = LOAD_STATE_YUKE_HAZIR;
                load_state_timer = 0.0f;
                vout_out_of_range_timer = 0.0f;
                pid_target_dynamic = PID_TARGET_UNLOADED;  // 14V'ye geri dön
            }
            // Vout sınır dışı 100µs+ → YUKSUZ
            else if (should_disconnect) {
                load_state = LOAD_STATE_YUKSUZ;
                load_state_timer = 0.0f;
                vout_out_of_range_timer = 0.0f;
                pid_target_dynamic = PID_TARGET_UNLOADED;
            }
            // Vin instabil (steady-state'te) → YUKSUZ
            else if (!vin_stable_for_operation) {
                load_state = LOAD_STATE_YUKSUZ;
                load_state_timer = 0.0f;
                vout_out_of_range_timer = 0.0f;
                pid_target_dynamic = PID_TARGET_UNLOADED;
            }
            // Freewheel aktif → YUKSUZ
            else if (freewheel_overcurrent_active || freewheeling_active) {
                load_state = LOAD_STATE_YUKSUZ;
                load_state_timer = 0.0f;
                vout_out_of_range_timer = 0.0f;
                pid_target_dynamic = PID_TARGET_UNLOADED;
            }
        }
        break;
        
        default:
            // Güvenlik: Bilinmeyen state → YUKSUZ
            load_state = LOAD_STATE_YUKSUZ;
            load_state_timer = 0.0f;
            shutdown_active = 1;
            pid_target_dynamic = PID_TARGET_UNLOADED;
            break;
    }
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // V8.3: DYNAMIC DISCHARGE LOGIC (PID Hedef Tabanlı)
    // ═══════════════════════════════════════════════════════════════════════════════
    // Aktivasyon: Vout > (pid_target_dynamic + 1.5V)
    // Deaktivasyon: Vout < (pid_target_dynamic - 0.5V)
    // MUTLAK MİNİMUM: Vout < 12.5V → DAIMA KAPALI (ne olursa olsun)
    
    // vout_connect_range kullanımı: Yüklü state'teki recovery kontrolü için
    (void)vout_connect_range;  // YUKE_HAZIR state'inde kullanılabilir (gelecekte)
    // V8.12: SABİT DISCHARGE THRESHOLD'LARI (pid_target_dynamic'den bağımsız)
    // Aktivasyon: Vout > 13.5V (her koşulda)
    // Deaktivasyon: Vout <= 13.0V (13V'a kadar çalışsın)
    #define DISCHARGE_ACTIVATE_FIXED   (13.5f)
    #define DISCHARGE_DEACTIVATE_FIXED (13.0f)
    
    if (raw_vout < DISCHARGE_DEACTIVATE_FIXED) {
        // 13V altında → discharge DAIMA kapalı
        if (discharge_timer >= DISCHARGE_MIN_ON_TIME) {
            discharge_active = 0;
            discharge_timer = 0.0f;
        }
    }
    else if (!discharge_active && raw_vout > DISCHARGE_ACTIVATE_FIXED) {
        // Aktivasyon: Vout > 13.5V
        discharge_active = 1;
        discharge_timer = 0.0f;
    }
    else if (discharge_active && raw_vout <= DISCHARGE_DEACTIVATE_FIXED) {
        // Deaktivasyon: Vout <= 13.0V
        if (discharge_timer >= DISCHARGE_MIN_ON_TIME) {
            discharge_active = 0;
            discharge_timer = 0.0f;
        }
    }
    
    // Discharge timer güncelle
    if (discharge_active) {
        discharge_timer += TS;
    }
    
    #endif
    
    // ─────────────────────────────────────────────────────────────────────────────
    // BÖLÜM 4: DUTY HESAPLAMA - SAF VOUT TABANLI PID KONTROLCÜ
    // ─────────────────────────────────────────────────────────────────────────────
    // KRİTİK: Duty SADECE Vout'a göre hesaplanır!
    // Vin SADECE koruma için okunur (freewheeling, ani değişim tespiti)
    // V8.3: Hedef state machine tarafından belirlenir: YUKSUZ/HAZIR→14V, BAGLI→12V
    
    // Önceki hata (D-term için)
    static float vout_error_prev = 0.0f;
    
    // Freewheeling dışında PID çalışır (charge_mode'da DA çalışır!)
    if (!freewheeling_active) {
        // ═══════════════════════════════════════════════════════════════════════════
        // V8.3: DİNAMİK HEDEF - State machine tarafından belirlenir
        // ═══════════════════════════════════════════════════════════════════════════
        // YUKSUZ/YUKE_HAZIR → 14V hedef
        // YUKE_BAGLI → 12V hedef (250µs ramp ile geçiş)
        // Ramp mantığı state machine içinde uygulanıyor
        // NOT: pid_target_dynamic static değişken olarak tanımlı, state machine güncelliyor
        
        // ═══════════════════════════════════════════════════════════════════════════
        // V8.3: AGRESİF PID MODU (YUKE_BAGLI ilk 1ms)
        // ═══════════════════════════════════════════════════════════════════════════
        // Yük bağlandığında hızlı toparlanma için PID kazançları 2x
        int aggressive_pid_active = (load_state == LOAD_STATE_YUKE_BAGLI) && 
                                   (load_state_timer < AGGRESSIVE_PID_TIME);
        float pid_multiplier = aggressive_pid_active ? AGGRESSIVE_PID_MULTIPLIER : 1.0f;
        
        // ═══════════════════════════════════════════════════════════════════════════
        // V8.4: DÜŞÜK VIN EKSTRA BOOST
        // ═══════════════════════════════════════════════════════════════════════════
        // Vin < 18V ise PID daha agresif olmalı (duty headroom az)
        // Vin < 15V ise 3x, Vin < 18V ise 2x ekstra
        if (raw_vin < 15.0f) {
            pid_multiplier *= 3.0f;  // Vin < 15V → toplam 15x (5*3)
        } else if (raw_vin < 18.0f) {
            pid_multiplier *= 2.0f;  // Vin < 18V → toplam 10x (5*2)
        }
        
        // ═══════════════════════════════════════════════════════════════════════════
        // VOUT HATASI + V8.4 DC OFFSET AGGRESSIVE TRIM
        // ═══════════════════════════════════════════════════════════════════════════
        // Agresif integral trim: Vout ortalamasını TAM 12.0V'a hızla yakınsar
        static float dc_offset_trim = 0.0f;
        static float vout_avg_slow = 13.0f;  // V8.10: 12→13 (yüksüz başlangıç)
        
        // Hızlı ortalama güncelle (tau ≈ 1ms)
        vout_avg_slow = vout_avg_slow * 0.99f + raw_vout * 0.01f;
        
        // Stabil çalışmada (freewheeling yok, yük bağlı) trim güncelle
        if (!freewheeling_active && !charge_mode && load_state == LOAD_STATE_YUKE_BAGLI) {
            // DC hata = gerçek ortalama - hedef (12.0V)
            float dc_error = vout_avg_slow - 12.0f;
            
            // Agresif integral (tau ≈ 10ms) - hızlı DC offset düzeltme
            dc_offset_trim += dc_error * 0.001f;
            
            // Trim sınırla (max ±0.5V)
            dc_offset_trim = Clamp_Float(dc_offset_trim, -0.5f, 0.5f);
        }
        
        // Hedefi trim ile ayarla
        float adjusted_target = pid_target_dynamic - dc_offset_trim;
        float vout_error = adjusted_target - raw_vout;
        float error_magnitude = Abs_Float(vout_error);
        
        // ═══════════════════════════════════════════════════════════════════════════
        // V8.4: HATA TABANLI P-BOOST
        // ═══════════════════════════════════════════════════════════════════════════
        // Büyük hatalarda daha agresif P kazancı
        float Kp_boost = 1.0f;
        if (error_magnitude > KP_ERROR_BOOST_THRESHOLD_2) {
            Kp_boost = KP_BOOST_LEVEL_2;  // Error > 2V → 3x P
        } else if (error_magnitude > KP_ERROR_BOOST_THRESHOLD_1) {
            Kp_boost = KP_BOOST_LEVEL_1;  // Error > 1V → 2x P
        }
        
        // ═══════════════════════════════════════════════════════════════════════════
        // P-TERİM: Oransal düzeltme (anlık tepki)
        // ═══════════════════════════════════════════════════════════════════════════
        // Vout < 12V → error > 0 → duty artar
        // Vout > 12V → error < 0 → duty azalır
        float Kp = 0.10f * pid_multiplier * Kp_boost;  // V8.11: 2x artırıldı (0.05→0.10)
        float p_term = vout_error * Kp;
        
        // ═══════════════════════════════════════════════════════════════════════════
        // I-TERİM: İntegral düzeltme (steady-state error eliminasyonu)
        // ═══════════════════════════════════════════════════════════════════════════
        // Ki artırıldı - daha hızlı ve hassas 12V ortalama
        float Ki = 15000.0f * pid_multiplier;  // V8.11: %50 artırıldı (10000→15000) faz 9 için
        
        // Anti-windup: Sadece hata makul aralıkta iken integratörü güncelle
        // V8.11: Eşik 2V→10V artırıldı (büyük hatalarda da tepki)
        if (Abs_Float(vout_error) < 10.0f) {
            vout_integrator += vout_error * Ki * TS;
            
            // İntegratör sınırla - NEGATİF değerlere de izin ver!
            // V8.11: Sınır genişletildi - büyük duty değişimlerine izin ver
            vout_integrator = Clamp_Float(vout_integrator, -0.5f, 1.5f);
        }
        
        float i_term = vout_integrator;
        
        // ═══════════════════════════════════════════════════════════════════════════
        // D-TERİM: Türev düzeltme (osilasyon sönümleme)
        // ═══════════════════════════════════════════════════════════════════════════
        // V8.10: D-term için LOW-PASS FILTER eklendi (üçgen osilasyon önleme)
        // Ham türev çok gürültülü ve osilasyona neden olur, filtre yumuşatır
        // V8.10: D-term düşük agresiflikle (osilasyon önleme + overshoot azaltma)
        // Önceki: 0.00001f osilasyona neden oldu. Şimdi 10x daha düşük.
        float Kd_multiplier = aggressive_pid_active ? KD_AGGRESSIVE_MULTIPLIER : 1.0f;
        float Kd = 0.00000015f * Kd_multiplier;  // V8.11: %50 artırıldı (0.0000001→0.00000015)
        
        // Türev hesapla
        float error_derivative_raw = (vout_error - vout_error_prev) / TS;
        
        // Low-pass filter (tau ≈ 10µs, alpha = 0.1)
        // Yüksek frekanslı gürültüyü ve osilasyonu söndürür
        static float error_derivative_filtered = 0.0f;
        error_derivative_filtered = error_derivative_filtered * 0.9f + error_derivative_raw * 0.1f;
        
        float d_term = Kd * error_derivative_filtered;  // Kd=0 → d_term=0
        
        // Önceki hatayı kaydet
        vout_error_prev = vout_error;
        
        // ═══════════════════════════════════════════════════════════════════════════
        // TOPLAM DUTY: PID çıkışı (SAF VOUT TABANLI!)
        // ═══════════════════════════════════════════════════════════════════════════
        // NOT: Feedforward YOK! Duty tamamen Vout hatasına göre hesaplanır.
        // İntegratör zamanla doğru duty değerine oturur.
        float duty_pid = p_term + i_term + d_term;
        
        // V8.10: Debug kopyaları (OutputSignal için)
        debug_p_term = p_term;
        debug_i_term = i_term;
        debug_d_term = d_term;

        
        // ═══════════════════════════════════════════════════════════════════════════
        // V8.3: DUTY BOOST (YUKE_BAGLI ilk 500µs)
        // ═══════════════════════════════════════════════════════════════════════════
        // Yük bağlandığında hızlı toparlanma için duty'ye +5% ekleniyor
        if ((load_state == LOAD_STATE_YUKE_BAGLI) && (load_state_timer < DUTY_BOOST_TIME)) {
            duty_pid += DUTY_BOOST_AMOUNT;
        }
        
        // ═══════════════════════════════════════════════════════════════════════════
        // AKIM KORUMA
        // ═══════════════════════════════════════════════════════════════════════════
        if (raw_iind >= I_PWM_CUTOFF) {
            duty_pid = 0.0f;
            pwm_cutoff_active = 1;
            vout_integrator *= 0.5f;  // İntegratörü azalt
        } else if (raw_iind >= I_HARD_BRAKE) {
            duty_pid *= 0.5f;
            vout_integrator *= 0.9f;
        } else if (raw_iind >= I_CONTINUOUS_LIMIT) {
            duty_pid *= 0.85f;
        }
        
        // ═══════════════════════════════════════════════════════════════════════════
        // MIL-SPEC EMERGENCY RECOVERY MODU
        // ═══════════════════════════════════════════════════════════════════════════
        // Freewheeling'den çıkışta yumuşak geçiş için
        if (emergency_active) {
            emergency_timer += TS;
            
            // 5ms boyunca recovery modunda kal
            if (emergency_timer < 0.005f) {
                // İntegratör yavaşça geri yükle
                vout_integrator *= 0.999f;  // Çok yavaş decay
                
                // PID çıkışını sınırla (agresif değişim önleme)
                duty_pid = Clamp_Float(duty_pid, step_current_duty - 0.1f, step_current_duty + 0.1f);
            } else {
                // Recovery tamamlandı
                emergency_active = 0;
                emergency_timer = 0.0f;
            }
        }
        
        // Duty sınırla
        float duty_final = Clamp_Float(duty_pid, DUTY_MIN, DUTY_MAX);
        
        // step_current_duty'yi güncelle
        step_current_duty = duty_final;
        step_target_duty = duty_final;
        emergency_duty_override = duty_final;
    }
    
    // ═══════════════════════════════════════════════════════════════════════════════
    // STAGE 1: LMG5200 AKIM KORUMA (KRİTİK - EN ÖNCE ÇALIŞIR)
    // ═══════════════════════════════════════════════════════════════════════════════
    
    // ─────────────────────────────────────────────────────────────────────────────
    // 35A+ → PWM ANINDA KES (tüm anahtarlar kapalı)
    // ─────────────────────────────────────────────────────────────────────────────
    if (raw_iind >= I_PWM_CUTOFF) {
        pwm_cutoff_active = 1;
        pwm_cutoff_timer = 0.0f;
        emergency_duty_override = 0.0f;  // DUTY = 0
        duty_before_cutoff = step_current_duty;
        
        // CROWBAR aktifse: paralel direnç aç
        #if CROWBAR_ENABLE
        crowbar_output = 1;  // Crowbar MOSFET aç
        #endif
    }
    
    // ─────────────────────────────────────────────────────────────────────────────
    // 30A-35A → SERT FREN (%20/cycle duty azalt)
    // ─────────────────────────────────────────────────────────────────────────────
    else if (raw_iind >= I_HARD_BRAKE) {
        hard_brake_timer += TS;
        
        // 50µs'den uzun sürerse daha da agresif ol
        if (hard_brake_timer > T_HARD_BRAKE_MAX) {
            step_current_duty *= 0.5f;  // %50 azalt (çok agresif)
        } else {
            step_current_duty *= DUTY_REDUCE_HARD;  // %20 azalt
        }
        
        step_current_duty = Max_Float(step_current_duty, DUTY_MIN);
        emergency_duty_override = step_current_duty;
        emergency_active = 1;
    }
    
    // ─────────────────────────────────────────────────────────────────────────────
    // 20A-30A → ORTA FREN (%5/cycle duty azalt)
    // ─────────────────────────────────────────────────────────────────────────────
    else if (raw_iind >= 20.0f) {
        step_current_duty *= DUTY_REDUCE_MEDIUM;
        step_current_duty = Max_Float(step_current_duty, DUTY_MIN);
        emergency_duty_override = step_current_duty;
        emergency_active = 1;
        hard_brake_timer = 0.0f;
    }
    
    // ─────────────────────────────────────────────────────────────────────────────
    // 10A-20A → YUMUŞAK FREN (%2/cycle duty azalt, 240µs limit)
    // ─────────────────────────────────────────────────────────────────────────────
    else if (raw_iind > I_CONTINUOUS_LIMIT) {
        overcurrent_timer += TS;
        
        step_current_duty *= DUTY_REDUCE_SOFT;
        step_current_duty = Max_Float(step_current_duty, DUTY_MIN);
        emergency_duty_override = step_current_duty;
        emergency_active = 1;
        hard_brake_timer = 0.0f;
        
        // 240µs AŞİLDI → SERT MÜDAHALE
        if (overcurrent_timer > T_OVERCURRENT_MAX) {
            step_current_duty *= 0.7f;  // %30 ekstra azalt
            emergency_duty_override = step_current_duty;
        }
    }
    
    // ─────────────────────────────────────────────────────────────────────────────
    // <10A → NORMAL ÇALIŞMA
    // ─────────────────────────────────────────────────────────────────────────────
    else {
        overcurrent_timer = 0.0f;  // Timer sıfırla
        hard_brake_timer = 0.0f;
        
        // PWM kesme modundan çık (CROWBAR: 8A'ya düşünce)
        if (pwm_cutoff_active) {
            #if CROWBAR_ENABLE
            if (raw_iind < CROWBAR_RELEASE_CURRENT) {
                pwm_cutoff_active = 0;
                crowbar_output = 0;  // Crowbar kapat
                emergency_duty_override = duty_before_cutoff;
                step_current_duty = duty_before_cutoff;
            }
            #else
            pwm_cutoff_timer += TS;
            if (pwm_cutoff_timer > T_PWM_CUTOFF_HOLD) {
                pwm_cutoff_active = 0;
                emergency_duty_override = duty_before_cutoff * 0.5f;  // %50 ile başla
                step_current_duty = emergency_duty_override;
            }
            #endif
        }
        
        // Emergency modundan çık
        if (emergency_active && !pwm_cutoff_active) {
            // Hedef duty'ye yavaşça yaklaş
            float ideal_duty = pid_target_dynamic / raw_vin;
            ideal_duty = Clamp_Float(ideal_duty, DUTY_MIN, DUTY_MAX);
            float duty_error = ideal_duty - step_current_duty;
            step_current_duty += duty_error * 0.02f;  // %2 yaklaşma
            emergency_duty_override = step_current_duty;
            
            // Hedefe yaklaşıldıysa çık
            if (Abs_Float(duty_error) < 0.02f) {
                emergency_active = 0;
                emergency_duty_override = -1.0f;  // Normal moda dön
            }
        }
    }
    
    // V7.13: STAGE 2 ve STAGE 3 artık V7.13 instant feedforward tarafından yönetiliyor
    // Eski kod kaldırıldı - vin_delta_total artık hesaplanmıyor
}

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 1.2: ADAPTIVE FILTERING
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

float alpha_v, alpha_i;

// Faza gore filtre hizi ayarla
if (sys_state == STATE_INIT) {
    alpha_v = FILTER_ALPHA_FAST;
    alpha_i = FILTER_ALPHA_FAST;
} else if (sys_state == STATE_SOFTSTART) {
    if (ss.phase <= SS_PHASE_CHARACTERIZE) {
        // Olcum fazinda hizli filtre
        alpha_v = FILTER_ALPHA_FAST;
        alpha_i = FILTER_ALPHA_FAST;
    } else if (ss.phase <= SS_PHASE_RAMP_SETTLE) {
        // Ramp fazinda orta hiz
        alpha_v = FILTER_ALPHA_NORMAL;
        alpha_i = FILTER_ALPHA_FAST;
    } else {
        // Blend/stabilize'da yavas
        alpha_v = FILTER_ALPHA_NORMAL;
        alpha_i = FILTER_ALPHA_NORMAL;
    }
} else {
    // Normal calisma
    if (behavior.oscillation_detected) {
        // Osilasyonda daha yavas filtre
        alpha_v = FILTER_ALPHA_SLOW;
        alpha_i = FILTER_ALPHA_NORMAL;
    } else {
        alpha_v = FILTER_ALPHA_NORMAL;
        alpha_i = FILTER_ALPHA_NORMAL;
    }
}

// EMA filtre uygula
filt_vin  = filt_vin  + alpha_v * (raw_vin  - filt_vin);
filt_vout = filt_vout + alpha_v * (raw_vout - filt_vout);
filt_iin  = filt_iin  + alpha_i * (raw_iin  - filt_iin);
filt_iind = filt_iind + alpha_i * (raw_iind - filt_iind);
filt_iout = filt_iout + alpha_i * (raw_iout - filt_iout);

// ═══════════════════════════════════════════════════════════════════════════════════════════
// V7.5: BOOTSTRAP DUTY KOMPANZASYONU
// Bootstrap kapasitör şarj süresi Vin'e bağlı - düşük Vin'de daha uzun süre gerekli
// Formula: duty_min = BOOT_CHARGE_CONSTANT / (Vin - BOOT_VOLTAGE)
// ═══════════════════════════════════════════════════════════════════════════════════════════
float duty_min_dynamic = DUTY_MIN;
float duty_max_dynamic = DUTY_BUCK_MAX;
float boot_compensation = 0.0f;

if (filt_vin > BOOT_VOLTAGE + 1.0f) {  // Minimum 1V marj
    // Dinamik minimum duty hesapla
    duty_min_dynamic = BOOT_CHARGE_CONSTANT / (filt_vin - BOOT_VOLTAGE);
    duty_min_dynamic = Clamp_Float(duty_min_dynamic, DUTY_MIN_ABSOLUTE, DUTY_MIN_STARTUP);
    
    // Bootstrap kompanzasyonu (duty'ye eklenir)
    // Düşük Vin'de daha yüksek kompanzasyon
    boot_compensation = duty_min_dynamic * 0.5f;  // %50 marj
    
    // Dinamik maksimum duty (yüksek Vin'de daha düşük max)
    duty_max_dynamic = DUTY_BUCK_MAX - duty_min_dynamic;
}

// Output clamp (safety)
filt_iout = Clamp_Float(filt_iout, 0.0f, 15.0f);
filt_iin  = Clamp_Float(filt_iin, 0.0f, 30.0f);

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 1.2B: INSTANT VIN RESPONSE SYSTEM (V7.3)
// Vin step degisimlerini ANLIK tespit edip AGRESIF duty duzeltmesi uygular
// 15V→24V = 60% artis → duty 0.8→0.5 = 37.5% azalma HEMEN uygulanmali
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

static float vin_prev = 0.0f;
static float vin_rate = 0.0f;
static float instant_duty_correction = 0.0f;

// Vin degisim hizini hesapla (RAW deger kullan - filtreli degil, daha hizli tepki)
// NOT: vin_delta zaten V7.13 instant FF'de tanımlı, burada farklı isim kullan
float vin_delta_inst = raw_vin - vin_prev;
vin_rate = vin_delta_inst / TS;  // V/s cinsinde

// Ilk cevrim kontrolu
if (vin_prev < 1.0f) {
    vin_prev = raw_vin;
    vin_rate = 0.0f;
}

// Buyuk Vin degisimi tespit (1V/ms = 1000V/s ustu kritik)
float vin_change_threshold = 500000.0f;  // 0.5V/us = 500kV/s

if (Abs_Float(vin_rate) > vin_change_threshold && raw_vin > VIN_MIN) {
    // KRITIK VIN DEGISIMI - ANLIK TEPKI
    
    // Ideal duty hesapla (yeni Vin icin)
    float ideal_duty_new = pid_target_dynamic / raw_vin;
    float ideal_duty_old = pid_target_dynamic / vin_prev;
    
    // Duty degisimi hesapla
    float duty_delta = ideal_duty_new - ideal_duty_old;
    
    // AGRESIF duzeltme (hemen uygula, %100 oraninda)
    instant_duty_correction = duty_delta;
    
    // Guvenlik limiti
    instant_duty_correction = Clamp_Float(instant_duty_correction, -0.4f, 0.4f);
    
} else if (Abs_Float(vin_rate) > 100000.0f) {
    // ORTA HIZDA VIN DEGISIMI - yavas gecis (0.1V/us = 100kV/s)
    float ideal_duty_new = pid_target_dynamic / raw_vin;
    float ideal_duty_old = pid_target_dynamic / vin_prev;
    float duty_delta = ideal_duty_new - ideal_duty_old;
    
    // %50 oraninda anlik, gerisi PID'e birak
    instant_duty_correction = duty_delta * 0.5f;
    instant_duty_correction = Clamp_Float(instant_duty_correction, -0.2f, 0.2f);
    
} else {
    // Normal calisma - yavas decay
    instant_duty_correction *= 0.95f;
    if (Abs_Float(instant_duty_correction) < 0.001f) {
        instant_duty_correction = 0.0f;
    }
}

// Vin gecmisini guncelle
vin_prev = raw_vin;


// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 1.2C: SMART 12V LOCK MODE (V7.4)
// Kapsamlı 12V kilitleme sistemi:
// 1. 20ms'de 12V'a oturma
// 2. Son 1ms ortalama kontrolü
// 3. Osilasyon ölçümü ve adaptif PID
// 4. Vin step koruması (akım sınırlı)
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

// ═══════════════════════════════════════════════════════════════════════════════════════════
// STATIC VARIABLES
// ═══════════════════════════════════════════════════════════════════════════════════════════
static int lock_mode_active = 0;
static int lock_mode_phase = 0;        // 0=settling, 1=locked, 2=oscillation_tuning
static float lock_mode_timer = 0.0f;
static float lock_duty_trim = 0.0f;
static float lock_error_integral = 0.0f;

// 1ms Buffer (1000 örnek @ 1MHz = 1ms) - V7.4
// NOT: 0.22ms periyotlu osilasyonu ölçmek için ~4.5 periyot gerekli
// 1ms / 0.22ms = 4.5 periyot - yeterli ortalama için
#define LOCK_BUFFER_SIZE 1000
static float vout_buffer[LOCK_BUFFER_SIZE];
static int vout_buffer_idx = 0;
static int vout_buffer_filled = 0;

// Osilasyon ölçümü
static float osc_vout_min = 100.0f;
static float osc_vout_max = 0.0f;
static float osc_amplitude = 0.0f;

// Adaptive PID
static float adaptive_kp = 0.03f;
static float adaptive_ki = 0.5f;

// ═══════════════════════════════════════════════════════════════════════════════════════════
// V7.5: 3 KATMANLI VIN STEP KORUMA SİSTEMİ
// Katman 1: Anlık Vin Step Detection (0 gecikme, RAW değer)
// Katman 2: Duty Atlama (Slew Bypass - anında yeni duty)
// Katman 3: Acil Akım Freni (I>25A → duty azalt)
// ═══════════════════════════════════════════════════════════════════════════════════════════

static float vin_step_protection_duty = 0.0f;
static int vin_step_active = 0;
static int vin_step_direction = 0;      // +1: artiş, -1: düşüş
static float vin_step_timer = 0.0f;
static float vin_step_target_duty = 0.5f;
static int slew_bypass_active = 0;      // Slew limiter bypass flag
static int vin_step_debounce = 0;       // V8.9: Debounce counter

{
    // V7.8: KATMAN 1 - ANLIK VIN STEP DETECTION
    // NOT: Emergency aktifken çalışma (Sorun 3 fix)
    // V8.9: 3-cycle debounce eklendi - tek seferlik spike'ları yoksay
    float vin_step_magnitude = Abs_Float(vin_delta_inst);
    
    // V8.9: Eşik 0.5 → 1.0V/µs (hassasiyet azaltıldı)
    // Emergency aktif DEĞİLSE çalış (çakışmayı önle)
    if (!emergency_active && vin_step_magnitude > 1.0f && raw_vin > VIN_MIN) {
        vin_step_debounce++;
        
        // V8.9: 3 ardışık cycle boyunca devam ederse tetikle
        if (vin_step_debounce >= 3) {
            vin_step_active = 1;
            vin_step_timer = 0.0f;
            slew_bypass_active = 1;  // Slew limiter'ı devre dışı bırak
            
            if (vin_delta_inst > 0) {
                // VIN ARTIŞ (15V→24V) - KRITIK DURUM
                vin_step_direction = 1;
                
                // KATMAN 2: ANLIK DUTY ATLAMA
                // Yeni duty = Vout_target / yeni_vin (anında hesapla)
                vin_step_target_duty = pid_target_dynamic / raw_vin;
                vin_step_target_duty = Clamp_Float(vin_step_target_duty, DUTY_MIN, DUTY_BUCK_MAX);
                // V8.9: 0.95 çarpanı KALDIRILDI - Vout düşüşüne neden oluyordu
                
            } else {
                // VIN DÜŞÜŞ (24V→15V) - daha yumuşak tepki
                vin_step_direction = -1;
                
                // Yeni duty hesapla ama yavaş artır
                vin_step_target_duty = pid_target_dynamic / raw_vin;
                vin_step_target_duty = Clamp_Float(vin_step_target_duty, DUTY_MIN, DUTY_BUCK_MAX);
            }
        }
    } else {
        // Koşul sağlanmadı - debounce sıfırla
        vin_step_debounce = 0;
    }
    
    // Vin step aktifken
    if (vin_step_active) {
        vin_step_timer += TS;
        
        // V7.8: KATMAN 3 - ACIL AKIM FRENİ (raw_iind kullan - 0 gecikme)
        if (raw_iind > 30.0f) {
            // 30A üzeri → duty SIFIR (acil durdur)
            vin_step_target_duty = DUTY_MIN;
            slew_bypass_active = 1;
        } else if (raw_iind > 20.0f) {
            // 20-30A arası → duty %50 azalt
            vin_step_target_duty *= 0.5f;
            slew_bypass_active = 1;
        } else if (raw_iind > 15.0f) {
            // 15-20A arası → duty %20 azalt
            vin_step_target_duty *= 0.8f;
        }
        
        // Step koruma süresi (5ms sonra normal moda dön)
        if (vin_step_timer > 0.005f) {
            vin_step_active = 0;
            slew_bypass_active = 0;
            vin_step_direction = 0;
        }
    }
    
    // vin_step_protection_duty hesapla (diğer modüller için)
    if (vin_step_active && vin_step_direction > 0) {
        // Vin artışında: mevcut duty ile hedef arası fark
        vin_step_protection_duty = vin_step_target_duty - feedforward_buck;
    } else {
        // Normal veya Vin düşüşünde: yavaş decay
        vin_step_protection_duty *= 0.95f;
    }
}

// ═══════════════════════════════════════════════════════════════════════════════════════════
// 12V LOCK MODE MAIN LOGIC
// ═══════════════════════════════════════════════════════════════════════════════════════════

// 12V Lock Mode tetikleme (11V üzeri)
if (filt_vout > 11.0f && filt_vout < 13.0f && filt_vin > VIN_MIN) {
    
    if (!lock_mode_active) {
        // İlk kez aktif - reset
        lock_mode_active = 1;
        lock_mode_phase = 0;  // Settling phase
        lock_mode_timer = 0.0f;
        lock_error_integral = 0.0f;
        vout_buffer_idx = 0;
        vout_buffer_filled = 0;
        osc_vout_min = 100.0f;
        osc_vout_max = 0.0f;
        // Adaptive PID - başlangıç değerleri (agresif)
        adaptive_kp = 0.03f;
        adaptive_ki = 0.5f;
    }
    
    lock_mode_timer += TS;
    
    // Vout buffer'a ekle (1ms ortalama için)
    vout_buffer[vout_buffer_idx] = filt_vout;
    vout_buffer_idx = (vout_buffer_idx + 1) % LOCK_BUFFER_SIZE;
    if (vout_buffer_idx == 0) vout_buffer_filled = 1;
    
    // 12V'dan sapma
    float lock_error = pid_target_dynamic - filt_vout;
    
    // ───────────────────────────────────────────────────────────────────────────
    // PHASE 0: SETTLING (ilk 20ms)
    // ───────────────────────────────────────────────────────────────────────────
    if (lock_mode_phase == 0) {
        // Agresif PID
        lock_error_integral += adaptive_ki * lock_error * TS;
        lock_error_integral = Clamp_Float(lock_error_integral, -0.3f, 0.3f);
        
        float lock_pid_out = adaptive_kp * lock_error + lock_error_integral;
        
        // Auto-calibration (5ms sonra)
        if (lock_mode_timer > 0.005f && Abs_Float(lock_error) > 0.1f) {
            lock_duty_trim += lock_error * 0.0003f;
            lock_duty_trim = Clamp_Float(lock_duty_trim, -0.15f, 0.15f);
        }
        
        // 20ms doldu → 1ms ortalama kontrolü
        if (lock_mode_timer > 0.02f && vout_buffer_filled) {
            // Son 1ms ortalaması hesapla
            float vout_avg = 0.0f;
            for (int i = 0; i < LOCK_BUFFER_SIZE; i++) {
                vout_avg += vout_buffer[i];
            }
            vout_avg /= LOCK_BUFFER_SIZE;
            
            // Ortalama 12V değil mi?
            float avg_error = pid_target_dynamic - vout_avg;
            if (Abs_Float(avg_error) > 0.03f) {
                // 12V'dan sapma var → trim düzelt
                lock_duty_trim += avg_error * 0.1f;  // Büyük düzeltme
                lock_duty_trim = Clamp_Float(lock_duty_trim, -0.2f, 0.2f);
            } else {
                // 12V'a ulaşıldı → PHASE 1'e geç
                lock_mode_phase = 1;
                lock_mode_timer = 0.0f;
                osc_vout_min = 100.0f;
                osc_vout_max = 0.0f;
            }
        }
        
        // Duty correction
        instant_duty_correction = lock_pid_out + lock_duty_trim + vin_step_protection_duty;
    }
    
    // ───────────────────────────────────────────────────────────────────────────
    // PHASE 1: LOCKED - Osilasyon ölçümü ve PID adaptasyonu
    // ───────────────────────────────────────────────────────────────────────────
    else if (lock_mode_phase == 1) {
        // Daha yumuşak PID (osilasyonu azaltmak için)
        float phase1_kp = adaptive_kp * 0.6f;  // %60
        float phase1_ki = adaptive_ki * 0.4f;  // %40
        
        lock_error_integral += phase1_ki * lock_error * TS;
        lock_error_integral = Clamp_Float(lock_error_integral, -0.3f, 0.3f);
        
        float lock_pid_out = phase1_kp * lock_error + lock_error_integral;
        
        // Osilasyon min/max ölç (5ms boyunca)
        if (filt_vout < osc_vout_min) osc_vout_min = filt_vout;
        if (filt_vout > osc_vout_max) osc_vout_max = filt_vout;
        
        lock_mode_timer += TS;
        
        if (lock_mode_timer > 0.005f) {
            // 5ms ölçüm tamamlandı
            osc_amplitude = osc_vout_max - osc_vout_min;
            
            // Osilasyon çok fazla mı? (>100mV p-p)
            if (osc_amplitude > 0.1f) {
                // PID'i yumuşat
                adaptive_kp *= 0.9f;
                adaptive_ki *= 0.85f;
                
                // Min değerler
                if (adaptive_kp < 0.005f) adaptive_kp = 0.005f;
                if (adaptive_ki < 0.05f) adaptive_ki = 0.05f;
            }
            // Osilasyon kabul edilebilir (50-100mV)
            else if (osc_amplitude > 0.05f) {
                // Hafif yumuşat
                adaptive_kp *= 0.95f;
                adaptive_ki *= 0.92f;
            }
            // Osilasyon çok düşük (<50mV) - İYİ!
            else {
                // PID'i koru veya hafif artır (tepki hızı için)
                adaptive_kp = Clamp_Float(adaptive_kp * 1.02f, 0.005f, 0.04f);
                adaptive_ki = Clamp_Float(adaptive_ki * 1.02f, 0.05f, 0.6f);
            }
            
            // 1ms ortalama kontrolü (devamlı)
            if (vout_buffer_filled) {
                float vout_avg = 0.0f;
                for (int i = 0; i < LOCK_BUFFER_SIZE; i++) {
                    vout_avg += vout_buffer[i];
                }
                vout_avg /= LOCK_BUFFER_SIZE;
                
                // Ortalama 12V'dan sapma → düzelt
                float avg_error = pid_target_dynamic - vout_avg;
                if (Abs_Float(avg_error) > 0.02f) {
                    lock_duty_trim += avg_error * 0.05f;
                    lock_duty_trim = Clamp_Float(lock_duty_trim, -0.2f, 0.2f);
                }
            }
            
            // Reset ölçüm
            lock_mode_timer = 0.0f;
            osc_vout_min = 100.0f;
            osc_vout_max = 0.0f;
        }
        
        // Duty correction
        instant_duty_correction = lock_pid_out + lock_duty_trim + vin_step_protection_duty;
    }
    
    // Clamp
    instant_duty_correction = Clamp_Float(instant_duty_correction, -0.4f, 0.4f);
    
} else if (filt_vout < 10.5f || filt_vout > 13.5f) {
    // Lock mode dışı - reset (ama trim'i koru)
    lock_mode_active = 0;
    lock_mode_phase = 0;
    lock_mode_timer = 0.0f;
    // Vin step protection devam eder
    instant_duty_correction = vin_step_protection_duty;
}


// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 1.3: HISTORY BUFFER UPDATE (Circular Buffer)
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

{
    // Eski degeri al (delta hesabi icin)
    int old_index = (history.head + HISTORY_DEPTH - 1) % HISTORY_DEPTH;
    float vout_old = history.vout[old_index];
    float iind_old = history.iind[old_index];
    
    // Yeni deger yaz
    history.vin[history.head]  = filt_vin;
    history.vout[history.head] = filt_vout;
    history.iin[history.head]  = filt_iin;
    history.iind[history.head] = filt_iind;
    history.iout[history.head] = filt_iout;
    
    // Head ilerlet
    history.head = (history.head + 1) % HISTORY_DEPTH;
    if (history.count < HISTORY_DEPTH) {
        history.count++;
    }
    
    // Delta hesapla (trend icin)
    if (history.count > 1) {
        history.vout_delta_sum = history.vout_delta_sum * 0.95f + (filt_vout - vout_old);
        history.iind_delta_sum = history.iind_delta_sum * 0.95f + (filt_iind - iind_old);
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.4: ONLINE STATISTICS (every 16 cycles to save CPU)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    if ((history.head % 16) == 0 && history.count >= 16) {
        float sum_vin = 0.0f, sum_vout = 0.0f, sum_iind = 0.0f;
        float min_vin = 100.0f, max_vin = 0.0f;
        float min_vout = 100.0f, max_vout = 0.0f;
        float min_iind = 100.0f, max_iind = -100.0f;
        
        for (int i = 0; i < HISTORY_DEPTH; i++) {
            sum_vin  += history.vin[i];
            sum_vout += history.vout[i];
            sum_iind += history.iind[i];
            
            if (history.vin[i] < min_vin) min_vin = history.vin[i];
            if (history.vin[i] > max_vin) max_vin = history.vin[i];
            if (history.vout[i] < min_vout) min_vout = history.vout[i];
            if (history.vout[i] > max_vout) max_vout = history.vout[i];
            if (history.iind[i] < min_iind) min_iind = history.iind[i];
            if (history.iind[i] > max_iind) max_iind = history.iind[i];
        }
        
        history.vin_mean  = sum_vin / HISTORY_DEPTH;
        history.vout_mean = sum_vout / HISTORY_DEPTH;
        history.iind_mean = sum_iind / HISTORY_DEPTH;
        
        history.vin_min = min_vin;
        history.vin_max = max_vin;
        history.vout_min = min_vout;
        history.vout_max = max_vout;
        history.iind_min = min_iind;
        history.iind_max = max_iind;
    }
}

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 1.5: TIMER UPDATES
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

state_timer += TS;
ss.total_timer += TS;
ss.phase_timer += TS;

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// V8.16: LOAD CONNECT BOOST MODE - 250µs boyunca 10x hızlı slew rate
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

// Transition algılama: load_state 0 veya 1'den 2'ye geçtiyse
if (load_state == LOAD_STATE_YUKE_BAGLI && prev_load_state != LOAD_STATE_YUKE_BAGLI) {
    // Yüke yeni bağlandı - boost mode başlat
    load_connect_boost_active = 1;
    load_connect_boost_timer = 0.0f;
}

// Boost mode timer güncelleme
if (load_connect_boost_active) {
    load_connect_boost_timer += TS;
    
    // 2ms sonra boost mode kapat (V8.16: 250µs → 2ms test)
    if (load_connect_boost_timer >= 0.002f) {
        load_connect_boost_active = 0;
    }
}

// Önceki state'i kaydet
prev_load_state = load_state;

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 1.5B: DYNAMIC BOOTSTRAP DUTY CALCULATION (Update)
// Bootstrap kapasitör şarjı için Vin'e bağlı dinamik duty_min hesabı
// NOT: duty_min_dynamic, duty_max_dynamic, boot_compensation yukarıda tanımlı (satır 90-92)
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

// Vin güvenlik kontrolü (sıfıra bölme önleme)
if (filt_vin > (BOOT_VOLTAGE + 1.0f)) {
    // Dinamik hesap
    duty_min_dynamic = BOOT_CHARGE_CONSTANT / (filt_vin - BOOT_VOLTAGE);
    
    // Clamp (güvenlik sınırları)
    duty_min_dynamic = Clamp_Float(duty_min_dynamic, DUTY_MIN_ABSOLUTE, DUTY_MIN_STARTUP);
    
    // duty_max = 1 - duty_min (simetrik)
    duty_max_dynamic = 1.0f - duty_min_dynamic;
    duty_max_dynamic = Clamp_Float(duty_max_dynamic, 0.95f, DUTY_MAX);
    
    // PWM kompanzasyon değeri (feedforward'a eklenecek)
    boot_compensation = duty_min_dynamic;
} else {
    // Düşük Vin veya startup - güvenli değerler
    duty_min_dynamic = DUTY_MIN_STARTUP;  // 0.02
    duty_max_dynamic = 0.98f;
    boot_compensation = DUTY_MIN_STARTUP;
}

// Suppress unused warnings - values kept for debugging/future use
(void)boot_compensation;
(void)slew_bypass_active;

// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                      MODULE 1.6: ADVANCED PREDICTION ENGINE (V7.1)                                              ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝
// Akim, Vin ve Vout icin turev tabanli gelecek tahmini ve akilli mudahale karari

{
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.6.1: PREDICTION BUFFER UPDATE
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    pred.i_buffer[pred.buffer_index] = filt_iind;
    pred.vin_buffer[pred.buffer_index] = filt_vin;
    pred.vout_buffer[pred.buffer_index] = filt_vout;
    
    pred.buffer_index = (pred.buffer_index + 1) % PRED_BUFFER_SIZE;
    if (pred.buffer_index == 0) pred.buffer_filled = 1;
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.6.2: DERIVATIVE CALCULATION (5-sample window)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    if (pred.buffer_filled || pred.buffer_index >= PRED_DERIV_WINDOW + 5) {
        // Indexler
        int idx_now = (pred.buffer_index + PRED_BUFFER_SIZE - 1) % PRED_BUFFER_SIZE;
        int idx_5   = (pred.buffer_index + PRED_BUFFER_SIZE - 6) % PRED_BUFFER_SIZE;
        int idx_10  = (pred.buffer_index + PRED_BUFFER_SIZE - 11) % PRED_BUFFER_SIZE;
        
        // 1. Turev (hiz): (value_now - value_5ago) / (5 * TS)
        float dt_5 = 5.0f * TS;
        pred.di_dt = (pred.i_buffer[idx_now] - pred.i_buffer[idx_5]) / dt_5;
        pred.dvin_dt = (pred.vin_buffer[idx_now] - pred.vin_buffer[idx_5]) / dt_5;
        pred.dvout_dt = (pred.vout_buffer[idx_now] - pred.vout_buffer[idx_5]) / dt_5;
        
        // 2. Turev (ivme): (di_dt_now - di_dt_5ago) / (5 * TS)
        float di_dt_old = (pred.i_buffer[idx_5] - pred.i_buffer[idx_10]) / dt_5;
        pred.d2i_dt2 = (pred.di_dt - di_dt_old) / dt_5;
        
        float dvin_dt_old = (pred.vin_buffer[idx_5] - pred.vin_buffer[idx_10]) / dt_5;
        pred.d2vin_dt2 = (pred.dvin_dt - dvin_dt_old) / dt_5;
        
        // Filtreleme (EMA)
        pred.di_dt_filtered = pred.di_dt_filtered * 0.8f + pred.di_dt * 0.2f;
        pred.dvin_dt_filtered = pred.dvin_dt_filtered * 0.8f + pred.dvin_dt * 0.2f;
        pred.dvout_dt_filtered = pred.dvout_dt_filtered * 0.8f + pred.dvout_dt * 0.2f;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.6.3: FUTURE PREDICTION (2nd order polynomial: X(t) = X0 + v*t + 0.5*a*t^2)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    {
        float I0 = filt_iind;
        float v = pred.di_dt_filtered;
        float a = pred.d2i_dt2;
        
        // Akim tahminleri
        float t10 = 10.0e-6f;
        float t30 = 30.0e-6f;
        float t50 = 50.0e-6f;
        
        pred.i_predicted_10us = I0 + v * t10 + 0.5f * a * t10 * t10;
        pred.i_predicted_30us = I0 + v * t30 + 0.5f * a * t30 * t30;
        pred.i_predicted_50us = I0 + v * t50 + 0.5f * a * t50 * t50;
        
        // Vin tahmini (1. derece yeterli)
        pred.vin_predicted_20us = filt_vin + pred.dvin_dt_filtered * 20.0e-6f;
        
        // Vout tahmini
        pred.vout_predicted_10us = filt_vout + pred.dvout_dt_filtered * 10.0e-6f;
        
        // Tepe noktasi tahmini (ivme < 0 ve hiz > 0 ise tepeye yaklasiyoruz)
        if (v > 0 && a < 0) {
            pred.time_to_peak = -v / a;  // t = -v/a
            pred.projected_peak_value = I0 + v * pred.time_to_peak + 0.5f * a * pred.time_to_peak * pred.time_to_peak;
        } else {
            pred.time_to_peak = 0.0f;
            pred.projected_peak_value = I0;
        }
        
        pred.predictions_made++;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.6.4: TIME TO SAFE ZONE CALCULATION
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    if (filt_iind > I_CONTINUOUS_MAX) {
        if (pred.di_dt_filtered < -100.0f) {
            // Akim dusuyor, 10A'ya ne zaman iner?
            // Basitleştirilmiş lineer tahmin: t = (I_now - 10A) / |dI/dt|
            pred.time_to_safe_zone = (filt_iind - I_CONTINUOUS_MAX) / Abs_Float(pred.di_dt_filtered);
            
            // 2. derece duzeltme (ivme varsa)
            if (pred.d2i_dt2 < 0) {
                // Ivme de negatif, daha hizli dusecek
                pred.time_to_safe_zone *= 0.9f;
            } else if (pred.d2i_dt2 > 0) {
                // Ivme pozitif, düşüş yavaslayacak
                pred.time_to_safe_zone *= 1.2f;
            }
        } else if (pred.di_dt_filtered > 100.0f) {
            // Akim yükseliyor
            pred.time_to_safe_zone = 999.0f;  // Belirsiz/cok uzun
        } else {
            // Akim sabit (|dI/dt| < 100 A/s)
            pred.time_to_safe_zone = 999.0f;
        }
        
        // Toplam overcurrent suresi tahmini
        pred.total_overcurrent_time_est = thermal_limiter.overcurrent_timer + pred.time_to_safe_zone;
    } else {
        pred.time_to_safe_zone = 0.0f;
        pred.total_overcurrent_time_est = 0.0f;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.6.5: SMART INTERVENTION DECISION
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // GUVENLIK: Buffer dolmadan tahmin yapma - varsayilan "mudahale etme" modu
    if (!pred.buffer_filled) {
        pred.natural_return_ok = 1;
        pred.intervention_needed = 0;
        pred.intervention_strength = 0.0f;
        pred.intervention_type = 0;
    }
    else if (filt_iind > I_CONTINUOUS_MAX) {
        float limit_ratio = pred.total_overcurrent_time_est / TIME_OVERCURRENT_LIMIT;
        
        if (limit_ratio < NATURAL_RETURN_SAFE_MARGIN) {
            // Dogal donus yeterli! (%70 marj ile)
            pred.natural_return_ok = 1;
            pred.intervention_needed = 0;
            pred.intervention_strength = 0.0f;
            pred.intervention_type = 0;
            pred.interventions_avoided++;
        } else if (limit_ratio < INTERVENTION_MARGIN) {
            // Belirsiz bolge, hafif mudahale
            pred.natural_return_ok = 0;
            pred.intervention_needed = 1;
            pred.intervention_strength = (limit_ratio - NATURAL_RETURN_SAFE_MARGIN) / (INTERVENTION_MARGIN - NATURAL_RETURN_SAFE_MARGIN);
            pred.intervention_type = 1;  // Yavas
        } else if (limit_ratio < INTERVENTION_URGENT) {
            // Mudahale gerekli
            pred.natural_return_ok = 0;
            pred.intervention_needed = 1;
            pred.intervention_strength = 0.5f + 0.5f * (limit_ratio - INTERVENTION_MARGIN) / (INTERVENTION_URGENT - INTERVENTION_MARGIN);
            pred.intervention_type = 2;  // Hizli
        } else {
            // Acil mudahale
            pred.natural_return_ok = 0;
            pred.intervention_needed = 1;
            pred.intervention_strength = 1.0f;
            pred.intervention_type = 3;  // Acil
        }
    } else {
        pred.natural_return_ok = 1;
        pred.intervention_needed = 0;
        pred.intervention_strength = 0.0f;
        pred.intervention_type = 0;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.6.6: VIN SLEW RATE CLASSIFICATION
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    {
        float abs_dvin = Abs_Float(pred.dvin_dt_filtered);
        
        if (abs_dvin < VIN_SLEW_SLOW) {
            pred.vin_slew_class = 0;  // Yavas
        } else if (abs_dvin < VIN_SLEW_NORMAL) {
            pred.vin_slew_class = 1;  // Normal
        } else if (abs_dvin < VIN_SLEW_FAST) {
            pred.vin_slew_class = 2;  // Hizli
        } else {
            pred.vin_slew_class = 3;  // Kritik
        }
        
        // Ani degisim tespiti
        if (abs_dvin > VIN_SLEW_CRITICAL) {
            pred.vin_step_detected = 1;
            pred.vin_step_magnitude = pred.dvin_dt_filtered * 10.0e-6f;  // 10us'deki degisim
            pred.feedforward_boost_active = 1;
        } else {
            pred.vin_step_detected = 0;
            pred.feedforward_boost_active = 0;
        }
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.6.7: VOUT OSCILLATION ANALYSIS (her 100 cycle'da bir)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    static int osc_analysis_counter = 0;
    if (++osc_analysis_counter >= OSC_ANALYSIS_PERIOD && pred.buffer_filled) {
        osc_analysis_counter = 0;
        
        // Min/Max/Mean hesapla
        float vout_min = 999.0f, vout_max = -999.0f, vout_sum = 0.0f;
        int sign_changes = 0;
        int prev_sign = 0;
        
        for (int i = 0; i < PRED_BUFFER_SIZE; i++) {
            float v = pred.vout_buffer[i];
            if (v < vout_min) vout_min = v;
            if (v > vout_max) vout_max = v;
            vout_sum += v;
        }
        
        pred.vout_dc_offset = vout_sum / PRED_BUFFER_SIZE;
        pred.vout_amplitude = (vout_max - vout_min) / 2.0f;
        
        // Isaret degisimi say (frekans tahmini icin)
        for (int i = 1; i < PRED_BUFFER_SIZE; i++) {
            float dev = pred.vout_buffer[i] - pred.vout_dc_offset;
            int curr_sign = (dev > 0) ? 1 : -1;
            if (prev_sign != 0 && curr_sign != prev_sign) {
                sign_changes++;
            }
            prev_sign = curr_sign;
        }
        pred.zero_crossings = sign_changes;
        
        // Frekans tahmini: f = sign_changes / (2 * PRED_BUFFER_SIZE * TS)
        pred.vout_frequency_est = (float)sign_changes / (2.0f * PRED_BUFFER_SIZE * TS);
        
        // Osilasyon tipi belirle
        if (pred.vout_amplitude < OSC_AMPLITUDE_NONE) {
            pred.oscillation_type = OSC_TYPE_NONE;
        } else if (pred.vout_amplitude < OSC_AMPLITUDE_ACCEPTABLE) {
            pred.oscillation_type = OSC_TYPE_DAMPED;
        } else if (pred.vout_amplitude < OSC_AMPLITUDE_WARNING) {
            pred.oscillation_type = OSC_TYPE_SUSTAINED;
        } else {
            pred.oscillation_type = OSC_TYPE_GROWING;
        }
        
        // Adaptif PID onerisi
        if (pred.oscillation_type >= OSC_TYPE_SUSTAINED) {
            // Osilasyon var, PID'i yumusat
            pred.recommended_kp = KP_BASE * PID_ADAPT_OSC_KP;
            pred.recommended_ki = KI_BASE * PID_ADAPT_OSC_KI;
            pred.recommended_kd = KD_BASE * PID_ADAPT_OSC_KD;
            pred.pid_update_pending = 1;
        } else if (pred.oscillation_type == OSC_TYPE_NONE && behavior.behavior_score > 80) {
            // Cok stabil, biraz agresiflet
            pred.recommended_kp = KP_BASE * PID_ADAPT_STABLE_KP;
            pred.recommended_ki = KI_BASE;
            pred.recommended_kd = KD_BASE * 0.9f;
            pred.pid_update_pending = 1;
        } else {
            pred.pid_update_pending = 0;
        }
    }
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                      MODULE 1.7: MODEL-BASED PREDICTIVE OBSERVER (MBPO)                                         ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝
// Devre elemanlarinin (L, C, Rload) fiziksel denklemlerini kullanarak beklenen davranisi
// hesaplar ve gercek olcumlerle karsilastirarak anomali tespiti ve adaptif duzeltme yapar.
// "Bu duty degerinde, bu zaman araliginda, cikis voltaji su kadar olmali" diye tahmin yapar.

{
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.7.1: LOAD ESTIMATION (YUK TAHMINI)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // Rload = Vout / Iout (Iout > 0.1A iken gecerli)
    
    if (filt_iout > RLOAD_EST_MIN_CURRENT && filt_vout > 1.0f) {
        model_obs.rload_instant = filt_vout / filt_iout;
        
        // Makul aralikta mi?
        if (model_obs.rload_instant > 0.5f && model_obs.rload_instant < 100.0f) {
            // EMA filtrele
            model_obs.rload_estimated = model_obs.rload_estimated * (1.0f - RLOAD_EST_ALPHA) + 
                                        model_obs.rload_instant * RLOAD_EST_ALPHA;
            model_obs.rload_valid = 1;
            model_obs.rload_last_update = state_timer;
            
            // Varyans hesapla
            float diff = model_obs.rload_instant - model_obs.rload_estimated;
            model_obs.rload_variance = model_obs.rload_variance * 0.95f + diff * diff * 0.05f;
            
            // Guvenilirlik: dusuk varyans = yuksek guven
            model_obs.rload_confidence = 1.0f / (1.0f + model_obs.rload_variance);
        }
    } else {
        model_obs.rload_valid = 0;
    }
    
    // Zaman sabitlerini guncelle (yuk degisirse)
    if (model_obs.rload_valid && model_obs.rload_confidence > 0.5f) {
        model_obs.tau_RC = model_obs.rload_estimated * model_obs.Cout_effective;
        model_obs.settling_time_est = 5.0f * model_obs.tau_RC;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.7.2: PHYSICS-BASED PREDICTION (FIZIK MODELI)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // Buck: dI_L/dt = (Vin * D - Vout) / L
    // dVout/dt = (I_L - Iout) / Cout
    
    {
        float D = timer_a_duty;  // Mevcut duty
        float Vin = filt_vin;
        float Vout = filt_vout;
        float I_L = filt_iind;
        float Iout = filt_iout;
        
        // Etkin L ve C kullan (kalibrasyon sonrasi)
        float L_eff = model_obs.L_effective;
        float C_eff = model_obs.Cout_effective;
        
        // Ideal duty hesapla
        if (Vin > 1.0f) {
            model_obs.duty_ideal = pid_target_dynamic / Vin;
            model_obs.duty_ideal = Clamp_Float(model_obs.duty_ideal, DUTY_MIN, DUTY_MAX);
        }
        
        // dI_L/dt model tahmini (ortalama - hem ON hem OFF durumu)
        // ON: dI/dt = (Vin - Vout) / L
        // OFF: dI/dt = -Vout / L
        // Ortalama: dI/dt = (Vin * D - Vout) / L
        model_obs.di_dt_model = (Vin * D - Vout) / L_eff;
        
        // dVout/dt model tahmini
        // dVout/dt = (I_L - Vout/Rload) / Cout
        float Iout_model = (model_obs.rload_valid) ? (Vout / model_obs.rload_estimated) : Iout;
        model_obs.dvout_dt_model = (I_L - Iout_model) / C_eff;
        
        // Eskiyi kaydet
        model_obs.iL_prediction_old = model_obs.iL_predicted;
        model_obs.vout_prediction_old = model_obs.vout_predicted;
        
        // Gelecek tahmini (10us sonra)
        float dt = MODEL_PREDICTION_STEP * TS;
        model_obs.iL_predicted = I_L + model_obs.di_dt_model * dt;
        model_obs.vout_predicted = Vout + model_obs.dvout_dt_model * dt;
        
        // Clamp (fiziksel sinirlar)
        model_obs.iL_predicted = Clamp_Float(model_obs.iL_predicted, -5.0f, 50.0f);
        model_obs.vout_predicted = Clamp_Float(model_obs.vout_predicted, 0.0f, 30.0f);
        
        model_obs.prediction_timestamp = state_timer;
        model_obs.predictions_total++;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.7.3: BEHAVIOR VALIDATION (DOGRULAMA)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // Onceki tahmini simdi gercekle karsilastir
    
    static int validation_counter = 0;
    if (++validation_counter >= MODEL_PREDICTION_STEP) {
        validation_counter = 0;
        
        // Hata hesapla
        model_obs.vout_error = Abs_Float(filt_vout - model_obs.vout_prediction_old);
        model_obs.iL_error = Abs_Float(filt_iind - model_obs.iL_prediction_old);
        
        // Yuzdesel hata (referansa gore)
        if (model_obs.vout_prediction_old > 0.5f) {
            model_obs.vout_error_percent = model_obs.vout_error / model_obs.vout_prediction_old;
        } else {
            model_obs.vout_error_percent = model_obs.vout_error;
        }
        
        if (Abs_Float(model_obs.iL_prediction_old) > 0.1f) {
            model_obs.iL_error_percent = model_obs.iL_error / Abs_Float(model_obs.iL_prediction_old);
        } else {
            model_obs.iL_error_percent = model_obs.iL_error;
        }
        
        // Hata integrali (drift tespiti)
        model_obs.error_integral += model_obs.vout_error_percent * TS * MODEL_PREDICTION_STEP;
        model_obs.error_integral *= 0.999f;  // Yavas sifirlama
        
        // Dogrulama sonucu
        model_obs.validations_done++;
        if (model_obs.vout_error_percent < MODEL_ERROR_MEDIUM) {
            model_obs.validations_passed++;
            model_obs.predictions_accurate++;
        }
        
        // Basari orani
        if (model_obs.validations_done > 10) {
            model_obs.validation_success_rate = (float)model_obs.validations_passed / model_obs.validations_done;
            model_obs.model_accuracy = model_obs.validation_success_rate;
        }
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.7.4: ANOMALY DETECTION (ANOMALI TESPITI)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 2ms boyunca model tutmazsa veya beklenen davranisi gostermezse anomali
    
    int prev_anomaly = model_obs.anomaly_type;
    model_obs.anomaly_type = ANOMALY_TYPE_NONE;
    model_obs.slow_response_detected = 0;
    model_obs.fast_response_detected = 0;
    model_obs.overshoot_predicted = 0;
    
    if (sys_state >= STATE_SOFTSTART && sys_state <= STATE_RUN_BUCKBOOST) {
        // SLOW RESPONSE: Vout beklenenden yavas artiyor
        if (model_obs.dvout_dt_model > 100.0f && pred.dvout_dt_filtered < model_obs.dvout_dt_model * 0.3f) {
            model_obs.slow_response_detected = 1;
            model_obs.anomaly_type = ANOMALY_TYPE_SLOW_RESPONSE;
        }
        
        // FAST RESPONSE: Vout beklenenden hizli artiyor
        if (model_obs.dvout_dt_model > 0 && pred.dvout_dt_filtered > model_obs.dvout_dt_model * 2.0f) {
            model_obs.fast_response_detected = 1;
            model_obs.anomaly_type = ANOMALY_TYPE_FAST_RESPONSE;
        }
        
        // OVERSHOOT PREDICTED: Vout hedefe yaklasirken hizli artis
        if (filt_vout > pid_target_dynamic * 0.9f && pred.dvout_dt_filtered > 50000.0f) {
            model_obs.overshoot_predicted = 1;
            model_obs.anomaly_type = ANOMALY_TYPE_OVERSHOOT;
        }
        
        // LOAD MISMATCH: Yuk tahmini surekli yanlis
        if (model_obs.rload_valid && model_obs.rload_variance > 2.0f) {
            model_obs.anomaly_type = ANOMALY_TYPE_LOAD_MISMATCH;
        }
        
        // COMPONENT FAULT: Model surekli tutmuyor
        if (model_obs.validations_done > 100 && model_obs.validation_success_rate < 0.5f) {
            model_obs.component_fault_suspected = 1;
            model_obs.anomaly_type = ANOMALY_TYPE_COMPONENT_FAULT;
        }
        
        // SENSOR ERROR: Olcum fiziksel olarak imkansiz
        if (filt_vout > filt_vin * 1.5f || filt_iind < -10.0f || filt_iind > 60.0f) {
            model_obs.sensor_error_suspected = 1;
            model_obs.anomaly_type = ANOMALY_TYPE_SENSOR_ERROR;
        }
    }
    
    // Anomali suresi takibi
    if (model_obs.anomaly_type != ANOMALY_TYPE_NONE) {
        if (model_obs.anomaly_type == prev_anomaly) {
            model_obs.anomaly_duration += TS;
            model_obs.anomaly_count++;
        } else {
            model_obs.anomaly_duration = 0.0f;
            model_obs.anomaly_count = 1;
        }
        
        // Ciddiyet hesapla
        model_obs.anomaly_severity = model_obs.anomaly_duration / ANOMALY_DURATION_THRESHOLD;
        model_obs.anomaly_severity = Clamp_Float(model_obs.anomaly_severity, 0.0f, 1.0f);
        
        // Anomali kesinlesti mi?
        if (model_obs.anomaly_duration >= ANOMALY_DURATION_THRESHOLD || 
            model_obs.anomaly_count >= ANOMALY_COUNT_THRESHOLD) {
            model_obs.anomaly_confirmed = 1;
            model_obs.anomalies_detected++;
        }
    } else {
        model_obs.anomaly_duration = 0.0f;
        model_obs.anomaly_count = 0;
        model_obs.anomaly_confirmed = 0;
        model_obs.anomaly_severity = 0.0f;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.7.5: MODEL CALIBRATION (KALIBRASYON)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // Model surekli yanlis tahmin ediyorsa, L veya C degerini duzelt
    
    if (model_obs.anomaly_type == ANOMALY_TYPE_COMPONENT_FAULT && !model_obs.calibration_complete) {
        model_obs.calibration_active = 1;
        
        // Yavas kalibrasyon - model hatasina gore L veya C duzelt
        if (model_obs.vout_error_percent > MODEL_ERROR_LARGE) {
            // Vout beklenenden farkli - Cout'u duzelt
            float correction = 1.0f + (model_obs.vout_error_percent - MODEL_ERROR_MEDIUM) * MODEL_CALIBRATION_ALPHA;
            correction = Clamp_Float(correction, 0.8f, 1.2f);
            
            if (filt_vout > model_obs.vout_prediction_old) {
                // Gercek daha yuksek - Cout daha kucuk olmali
                model_obs.Cout_correction *= correction;
            } else {
                // Gercek daha dusuk - Cout daha buyuk olmali
                model_obs.Cout_correction /= correction;
            }
            model_obs.Cout_correction = Clamp_Float(model_obs.Cout_correction, 0.5f, 2.0f);
        }
        
        if (model_obs.iL_error_percent > MODEL_ERROR_LARGE) {
            // I_L beklenenden farkli - L'yi duzelt
            float correction = 1.0f + (model_obs.iL_error_percent - MODEL_ERROR_MEDIUM) * MODEL_CALIBRATION_ALPHA;
            correction = Clamp_Float(correction, 0.8f, 1.2f);
            
            if (filt_iind > model_obs.iL_prediction_old) {
                // Gercek akim daha yuksek - L daha kucuk olmali
                model_obs.L_correction /= correction;
            } else {
                // Gercek akim daha dusuk - L daha buyuk olmali
                model_obs.L_correction *= correction;
            }
            model_obs.L_correction = Clamp_Float(model_obs.L_correction, 0.5f, 2.0f);
        }
        
        // Etkin degerleri guncelle
        model_obs.L_effective = MODEL_L_NOMINAL * model_obs.L_correction;
        model_obs.Cout_effective = MODEL_COUT_NOMINAL * model_obs.Cout_correction;
        
        // Rezonans frekansini guncelle (sqrt yerine yaklasik hesap)
        // tau_LC = sqrt(L * C) ≈ tau_LC_nominal * sqrt(L_corr * C_corr)
        // Newton-Raphson: sqrt(x) ≈ (x/guess + guess) / 2 (tek iterasyon)
        float lc_product = model_obs.L_correction * model_obs.Cout_correction;
        float sqrt_approx = (lc_product / 1.0f + 1.0f) * 0.5f;  // guess = 1.0
        sqrt_approx = (lc_product / sqrt_approx + sqrt_approx) * 0.5f;  // 2. iterasyon
        model_obs.tau_LC = 1.817e-6f * sqrt_approx;  // Base tau_LC * correction
        model_obs.f_resonance = 1.0f / (6.2832f * model_obs.tau_LC);
        
        // Kalibrasyon tamamlandi mi?
        if (model_obs.validation_success_rate > 0.8f) {
            model_obs.calibration_complete = 1;
            model_obs.calibration_active = 0;
        }
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 1.7.6: ADAPTIVE DUTY CORRECTION (ADAPTIF DUZELTME)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // Anomaliye gore duty duzeltme oner (guvenlik sinirlarini asma!)
    
    model_obs.correction_active = 0;
    model_obs.correction_direction = 0;
    model_obs.duty_correction = 0.0f;
    
    // GUVENLIK: Termal koruma aktifse duzeltme yapma!
    if (!thermal_limiter.thermal_clamp_active && !thermal_limiter.peak_absolute && 
        sys_state >= STATE_RUN_BUCK && model_obs.anomaly_confirmed) {
        
        switch (model_obs.anomaly_type) {
            case ANOMALY_TYPE_SLOW_RESPONSE:
                // Vout yavas artiyor - duty artir
                model_obs.correction_direction = +1;
                model_obs.duty_correction = DUTY_CORRECTION_GAIN * model_obs.anomaly_severity;
                model_obs.correction_active = 1;
                break;
                
            case ANOMALY_TYPE_FAST_RESPONSE:
            case ANOMALY_TYPE_OVERSHOOT:
                // Vout cok hizli artiyor - duty azalt
                model_obs.correction_direction = -1;
                model_obs.duty_correction = -DUTY_CORRECTION_GAIN * model_obs.anomaly_severity;
                model_obs.correction_active = 1;
                break;
                
            case ANOMALY_TYPE_LOAD_MISMATCH:
                // Yuk tahmini tutmuyor - duty'yi ideal'e yaklas
                model_obs.correction_direction = (timer_a_duty < model_obs.duty_ideal) ? +1 : -1;
                model_obs.duty_correction = (model_obs.duty_ideal - timer_a_duty) * 0.01f;
                model_obs.correction_active = 1;
                break;
                
            case ANOMALY_TYPE_COMPONENT_FAULT:
            case ANOMALY_TYPE_SENSOR_ERROR:
                // Ciddi sorun - duzeltme yapma, sadece uyar
                model_obs.correction_active = 0;
                model_obs.faults_prevented++;
                break;
                
            default:
                break;
        }
        
        // Duzeltme limitleri
        model_obs.duty_correction = Clamp_Float(model_obs.duty_correction, -DUTY_CORRECTION_MAX, DUTY_CORRECTION_MAX);
        
        // Kumulatif duzeltme
        if (model_obs.correction_active) {
            model_obs.duty_correction_integral += model_obs.duty_correction;
            model_obs.duty_correction_integral = Clamp_Float(model_obs.duty_correction_integral, -0.1f, 0.1f);
            model_obs.corrections_applied++;
        }
    }
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 2: BEHAVIORAL ANALYSIS ENGINE                                                ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝

{
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 2.1: TREND ANALYSIS
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Vout trend
    behavior.vout_trend_rate = history.vout_delta_sum * 1000.0f;  // V/ms
    
    if (behavior.vout_trend_rate > TREND_THRESHOLD) {
        behavior.vout_trend = TREND_RISING;
    } else if (behavior.vout_trend_rate < -TREND_THRESHOLD) {
        behavior.vout_trend = TREND_FALLING;
    } else {
        behavior.vout_trend = TREND_STABLE;
    }
    
    // I_L trend
    behavior.iind_trend_rate = history.iind_delta_sum * 1000.0f;  // A/ms
    
    if (behavior.iind_trend_rate > TREND_THRESHOLD * 10.0f) {
        behavior.iind_trend = TREND_RISING;
    } else if (behavior.iind_trend_rate < -TREND_THRESHOLD * 10.0f) {
        behavior.iind_trend = TREND_FALLING;
    } else {
        behavior.iind_trend = TREND_STABLE;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 2.2: OSCILLATION DETECTION
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Derivative hesapla
    behavior.vout_derivative = (filt_vout - history.vout[(history.head + HISTORY_DEPTH - 2) % HISTORY_DEPTH]) / TS;
    behavior.vout_derivative_filtered = behavior.vout_derivative_filtered * 0.9f + behavior.vout_derivative * 0.1f;
    
    // Clamp
    behavior.vout_derivative_filtered = Clamp_Float(behavior.vout_derivative_filtered, -10000.0f, 10000.0f);
    
    // Isaret degisimi takibi
    int current_sign = Sign_Float(behavior.vout_derivative_filtered);
    
    if (current_sign != 0 && current_sign != behavior.prev_sign && behavior.prev_sign != 0) {
        // Isaret degisti - bu anlamlı bir degisim mi?
        float amplitude = Abs_Float(behavior.vout_derivative_filtered) / 1000.0f;  // V/ms
        
        if (amplitude > OSC_DETECT_THRESHOLD) {
            behavior.sign_changes++;
            
            // Amplitut takibi
            float peak_to_peak = history.vout_max - history.vout_min;
            behavior.oscillation_amplitude = behavior.oscillation_amplitude * 0.8f + peak_to_peak * 0.2f;
        }
    }
    behavior.prev_sign = current_sign;
    
    // Pencere zamanlayici
    behavior.osc_timer += TS;
    
    if (behavior.osc_timer >= OSC_DETECTION_WINDOW) {
        // Pencere doldu - degerlendir
        if (behavior.sign_changes >= OSC_SIGN_CHANGE_LIMIT) {
            if (!behavior.oscillation_detected) {
                // Yeni osilasyon tespit edildi
                behavior.oscillation_detected = 1;
                behavior.damping_factor = OSC_DAMPING_FACTOR;
                behavior.recovery_timer = 0.0f;
                diag.oscillation_events++;
                
                // Frekans tahmini
                behavior.oscillation_frequency = (float)behavior.sign_changes / (2.0f * OSC_DETECTION_WINDOW);
            }
        }
        
        // Reset pencere
        behavior.sign_changes = 0;
        behavior.osc_timer = 0.0f;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 2.3: OSCILLATION RECOVERY (Yavas damping artisi)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    if (behavior.oscillation_detected) {
        behavior.recovery_timer += TS;
        
        // Amplitude azaldiysa recovery hizlandir
        float current_amplitude = history.vout_max - history.vout_min;
        
        if (behavior.recovery_timer > OSC_RECOVERY_TIME) {
            if (current_amplitude < VOUT_TOLERANCE) {
                // Osilasyon azaldi, yavasca normale don
                behavior.damping_factor += 0.00005f;
                
                if (behavior.damping_factor >= 1.0f) {
                    behavior.damping_factor = 1.0f;
                    behavior.oscillation_detected = 0;
                    behavior.oscillation_amplitude = 0.0f;
                }
            }
        }
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 2.4: SIMPLIFIED FREQUENCY ANALYSIS (8-bin DFT)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Her 64 cycle'da bir frekans analizi yap
    if ((history.head % HISTORY_DEPTH) == 0 && history.count >= HISTORY_DEPTH) {
        // Basit Goertzel-like frekans tespiti
        // Hedef frekanslar: ~1kHz, 2kHz, 5kHz, 10kHz, 20kHz, 50kHz, 100kHz, 200kHz
        
        float dc_sum = 0.0f;
        for (int i = 0; i < HISTORY_DEPTH; i++) {
            dc_sum += history.vout[i];
        }
        float dc_mean = dc_sum / HISTORY_DEPTH;
        
        // AC component
        float ac_power = 0.0f;
        for (int i = 0; i < HISTORY_DEPTH; i++) {
            float ac = history.vout[i] - dc_mean;
            ac_power += ac * ac;
        }
        ac_power /= HISTORY_DEPTH;
        
        // Dominant frequency estimate (zero-crossing based)
        int zero_crossings = 0;
        for (int i = 1; i < HISTORY_DEPTH; i++) {
            float prev_ac = history.vout[i-1] - dc_mean;
            float curr_ac = history.vout[i] - dc_mean;
            if ((prev_ac >= 0 && curr_ac < 0) || (prev_ac < 0 && curr_ac >= 0)) {
                zero_crossings++;
            }
        }
        
        // Frekans tahmin
        float estimated_freq = (float)zero_crossings / (2.0f * HISTORY_DEPTH * TS);
        
        // Bin'e yerlestir
        // sqrt approximation (Newton-Raphson 2 iteration)
        float sqrt_approx = ac_power;
        if (sqrt_approx > 0.0f) {
            sqrt_approx = 0.5f * (sqrt_approx + ac_power / sqrt_approx);
            sqrt_approx = 0.5f * (sqrt_approx + ac_power / sqrt_approx);
        }
        behavior.dominant_magnitude = sqrt_approx;
        
        if (estimated_freq < 2000.0f) behavior.dominant_bin = 0;
        else if (estimated_freq < 5000.0f) behavior.dominant_bin = 1;
        else if (estimated_freq < 10000.0f) behavior.dominant_bin = 2;
        else if (estimated_freq < 20000.0f) behavior.dominant_bin = 3;
        else if (estimated_freq < 50000.0f) behavior.dominant_bin = 4;
        else if (estimated_freq < 100000.0f) behavior.dominant_bin = 5;
        else if (estimated_freq < 200000.0f) behavior.dominant_bin = 6;
        else behavior.dominant_bin = 7;
        
        behavior.freq_bins[behavior.dominant_bin] = behavior.dominant_magnitude;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 2.5: BEHAVIOR SCORING
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    int score = BEHAVIOR_EXCELLENT;
    
    // Vout error penalty
    float vout_error = Abs_Float(filt_vout - pid_target_dynamic);
    if (vout_error > 1.0f) score -= 40;
    else if (vout_error > 0.5f) score -= 20;
    else if (vout_error > 0.2f) score -= 10;
    
    // Oscillation penalty
    if (behavior.oscillation_detected) score -= 30;
    else if (behavior.oscillation_amplitude > 0.5f) score -= 15;
    
    // Current limit penalty (thermal limiter)
    if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) score -= 20;
    else if (thermal_limiter.peak_warning || thermal_limiter.modulation_active) score -= 10;
    
    // DCM risk penalty
    if (dcm_prev.dcm_detected) score -= 25;
    else if (dcm_prev.dcm_imminent) score -= 10;
    
    // Clamp
    score = Clamp_Float(score, 0, 100);
    behavior.behavior_score = score;
    
    // History guncelle
    behavior.score_history[behavior.score_index] = score;
    behavior.score_index = (behavior.score_index + 1) % 8;
    
    // Ortalama hesapla
    float sum_score = 0.0f;
    for (int i = 0; i < 8; i++) {
        sum_score += (float)behavior.score_history[i];
    }
    behavior.score_average = sum_score / 8.0f;
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 3: PREDICTIVE CONTROLLER                                                     ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝

{
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 3.1: MODEL UPDATE (Online system identification)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Load estimate from impedance observer
    if (impedance_obs.is_valid && impedance_obs.confidence > 0.5f) {
        predictor.model_R = predictor.model_R * 0.999f + impedance_obs.r_load_filtered * 0.001f;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 3.2: STATE PREDICTION (8-step horizon)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Basit discrete-time model:
    // i_L(k+1) = i_L(k) + (Ts/L) * (Vin*D - Vout*(1-D))  [simplified]
    // v_out(k+1) = v_out(k) + (Ts/C) * (i_L(k) - Vout/R)
    
    float pred_iL = filt_iind;
    float pred_Vout = filt_vout;
    float current_duty = timer_a_duty;
    
    // DCM margin tracking
    float min_iL_predicted = pred_iL;
    
    for (int k = 0; k < PREDICTION_HORIZON; k++) {
        // Sonraki duty tahmini (basit: sabit veya trend-based)
        float next_duty = current_duty;
        
        // Inductor current prediction
        float diL = (TS / predictor.model_L) * (filt_vin * next_duty - pred_Vout);
        pred_iL = pred_iL + diL;
        
        // DCM check
        if (pred_iL < min_iL_predicted) min_iL_predicted = pred_iL;
        
        // Output voltage prediction
        float dVout = (TS / predictor.model_C) * (pred_iL - pred_Vout / predictor.model_R);
        pred_Vout = pred_Vout + dVout;
        
        // Store predictions
        predictor.pred_iind[k] = pred_iL;
        predictor.pred_vout[k] = pred_Vout;
        predictor.pred_duty[k] = next_duty;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 3.3: DCM RISK ASSESSMENT
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    predictor.dcm_margin = min_iL_predicted - IIND_DCM_THRESHOLD;
    
    if (predictor.dcm_margin < 0.0f) {
        predictor.dcm_risk = 1.0f;
    } else if (predictor.dcm_margin < IIND_DCM_WARNING) {
        predictor.dcm_risk = 1.0f - (predictor.dcm_margin / IIND_DCM_WARNING);
    } else {
        predictor.dcm_risk = 0.0f;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 3.4: OPTIMAL DUTY CALCULATION (Cost function minimization)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Hedef: Vout = 12V, minimize duty changes, avoid DCM
    float vout_error_pred = predictor.pred_vout[0] - pid_target_dynamic;
    float duty_correction = 0.0f;
    
    // P kontrolu - tahmine dayali
    duty_correction = -vout_error_pred * KP_BASE * predictor.model_confidence;
    
    // DCM riski varsa duty'yi artir
    if (predictor.dcm_risk > 0.5f) {
        duty_correction += predictor.dcm_risk * 0.01f;
    }
    
    predictor.optimal_duty = current_duty + duty_correction;
    predictor.optimal_duty = Clamp_Float(predictor.optimal_duty, DUTY_MIN, DUTY_MAX);
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 3.5: PREDICTION ACCURACY TRACKING
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Onceki tahmini mevcut gercekle karsilastir
    float pred_error = Abs_Float(predictor.pred_vout[0] - filt_vout);
    predictor.pred_error_sum = predictor.pred_error_sum * 0.99f + pred_error;
    predictor.pred_error_count = predictor.pred_error_count * 0.99f + 1.0f;
    
    float avg_error = predictor.pred_error_sum / Max_Float(predictor.pred_error_count, 1.0f);
    predictor.pred_accuracy = 1.0f - Clamp_Float(avg_error / 1.0f, 0.0f, 1.0f);
    
    // Model confidence update
    predictor.model_confidence = predictor.model_confidence * 0.999f + predictor.pred_accuracy * 0.001f;
    predictor.model_confidence = Clamp_Float(predictor.model_confidence, PRED_CONFIDENCE_MIN, PRED_CONFIDENCE_MAX);
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 4: ADAPTIVE SCENARIO ENGINE                                                  ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝

{
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 4.1: SCENARIO DETECTION
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // High Vin detection (>20V)
    scenario.high_vin_detected = (filt_vin > HIGH_VIN_THRESHOLD) ? 1 : 0;
    
    // Low Vin detection (<11V)
    scenario.low_vin_detected = (filt_vin < VIN_BOOST_THRESHOLD) ? 1 : 0;
    
    // Heavy load detection (>4A)
    scenario.heavy_load_detected = (filt_iout > 4.0f) ? 1 : 0;
    
    // Light load detection (<1A)
    scenario.light_load_detected = (filt_iout < 1.0f) ? 1 : 0;
    
    // Oscillation detection
    scenario.osc_detected = behavior.oscillation_detected;
    
    // DCM risk detection
    scenario.dcm_risk_detected = (predictor.dcm_risk > 0.5f) ? 1 : 0;
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 4.2: SCENARIO SELECTION (Priority-based)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    int new_scenario = SCENARIO_NORMAL;
    
    // En yuksek oncelikli senaryoyu sec
    if (scenario.osc_detected) {
        new_scenario = SCENARIO_OSCILLATING;
    } else if (scenario.dcm_risk_detected) {
        new_scenario = SCENARIO_DCM_RISK;
    } else if (scenario.high_vin_detected) {
        new_scenario = SCENARIO_HIGH_VIN;
    } else if (scenario.low_vin_detected) {
        new_scenario = SCENARIO_LOW_VIN;
    } else if (scenario.heavy_load_detected) {
        new_scenario = SCENARIO_HEAVY_LOAD;
    } else if (scenario.light_load_detected) {
        new_scenario = SCENARIO_LIGHT_LOAD;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 4.3: SMOOTH SCENARIO TRANSITION
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    if (new_scenario != scenario.current_scenario) {
        scenario.previous_scenario = scenario.current_scenario;
        scenario.current_scenario = new_scenario;
        scenario.scenario_blend = 0.0f;
        scenario.blend_timer = 0.0f;
        scenario.strategy_changes++;
        diag.scenario_changes++;
    }
    
    // Blend ilerlet (sigmoid benzeri)
    if (scenario.scenario_blend < 1.0f) {
        scenario.blend_timer += TS;
        float t = scenario.blend_timer / SCENARIO_BLEND_TIME;
        
        // S-curve blending: smoother than linear
        if (t < 1.0f) {
            // Cubic smoothstep: 3t^2 - 2t^3
            scenario.scenario_blend = t * t * (3.0f - 2.0f * t);
        } else {
            scenario.scenario_blend = 1.0f;
        }
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 4.4: PARAMETER INTERPOLATION
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Hedef parametreleri senaryoya gore belirle
    float target_kp, target_ki, target_kd, target_slew, target_ilimit;
    
    switch (scenario.current_scenario) {
        case SCENARIO_HIGH_VIN:
            // 24V+ icin cok yumusak
            target_kp = KP_BUCK * HIGH_VIN_KP_FACTOR;
            target_ki = KI_BUCK * HIGH_VIN_KI_FACTOR;
            target_kd = KD_BUCK * 0.5f;
            target_slew = SLEW_NORMAL * 0.5f;
            target_ilimit = IIND_MAX_STEADY;
            break;
            
        case SCENARIO_LOW_VIN:
            // Boost mode
            target_kp = KP_BOOST;
            target_ki = KI_BOOST;
            target_kd = KD_BOOST;
            target_slew = SLEW_NORMAL;
            target_ilimit = IIND_MAX_STEADY * 1.2f;  // Boost'ta biraz daha fazla izin
            break;
            
        case SCENARIO_OSCILLATING:
            // Osilasyon damping
            target_kp = KP_BASE * behavior.damping_factor;
            target_ki = KI_BASE * behavior.damping_factor * 0.5f;  // Ki'yi daha da dusur
            target_kd = KD_BASE * 2.0f;  // Derivative damping artir
            target_slew = SLEW_DAMPED;
            target_ilimit = IIND_MAX_STEADY;
            break;
            
        case SCENARIO_DCM_RISK:
            // DCM engelleme
            target_kp = KP_BASE;
            target_ki = KI_BASE * 0.7f;
            target_kd = KD_BASE;
            target_slew = SLEW_NORMAL;
            target_ilimit = IIND_MAX_STEADY;
            break;
            
        case SCENARIO_HEAVY_LOAD:
            // Agir yuk
            target_kp = KP_BASE * 1.2f;
            target_ki = KI_BASE * 1.1f;
            target_kd = KD_BASE;
            target_slew = SLEW_NORMAL;
            target_ilimit = IIND_MAX_STEADY;
            break;
            
        case SCENARIO_LIGHT_LOAD:
            // Hafif yuk - DCM riski
            target_kp = KP_BASE * 0.8f;
            target_ki = KI_BASE * 0.6f;
            target_kd = KD_BASE;
            target_slew = SLEW_NORMAL * 0.7f;
            target_ilimit = IIND_MAX_STEADY * 0.5f;
            break;
            
        default:  // SCENARIO_NORMAL
            target_kp = KP_BASE;
            target_ki = KI_BASE;
            target_kd = KD_BASE;
            target_slew = SLEW_NORMAL;
            target_ilimit = IIND_MAX_STEADY;
            break;
    }
    
    // Interpolate - V8.13: 10x hızlandırıldı (0.01 → 0.1) - P-term=0 sorunu önleme
    float b = scenario.scenario_blend;
    scenario.active_kp = LERP(scenario.active_kp, target_kp, b * 0.1f);
    scenario.active_ki = LERP(scenario.active_ki, target_ki, b * 0.1f);
    scenario.active_kd = LERP(scenario.active_kd, target_kd, b * 0.1f);
    scenario.active_slew = LERP(scenario.active_slew, target_slew, b * 0.1f);
    scenario.active_iind_limit = LERP(scenario.active_iind_limit, target_ilimit, b * 0.1f);
    
    // Strategy timer
    scenario.strategy_timer += TS;
    
    // Confidence update
    if (behavior.behavior_score >= BEHAVIOR_GOOD) {
        scenario.strategy_confidence = Min_Float(scenario.strategy_confidence + 0.0001f, 1.0f);
    } else {
        scenario.strategy_confidence = Max_Float(scenario.strategy_confidence - 0.001f, 0.3f);
    }
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 5: DCM PREVENTION SYSTEM                                                     ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝

// V8.9: Soft-start sırasında DCM prevention devre dışı (akım doğal olarak düşük olabilir)
if (sys_state != STATE_SOFTSTART) {
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 5.1: MINIMUM CURRENT TRACKING
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // History'deki minimum akimi bul
    if (history.count >= HISTORY_DEPTH) {
        dcm_prev.iind_minimum = history.iind_min;
    } else {
        dcm_prev.iind_minimum = filt_iind;
    }
    
    // Filtrele
    dcm_prev.iind_minimum_filtered = dcm_prev.iind_minimum_filtered * 0.99f + 
                                      dcm_prev.iind_minimum * 0.01f;
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 5.2: DCM DETECTION
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    dcm_prev.dcm_margin = dcm_prev.iind_minimum_filtered - IIND_DCM_THRESHOLD;
    
    if (dcm_prev.iind_minimum_filtered < IIND_DCM_THRESHOLD) {
        dcm_prev.dcm_detected = 1;
        dcm_prev.dcm_events++;
        dcm_prev.dcm_total_time += TS;
    } else {
        dcm_prev.dcm_detected = 0;
    }
    
    if (dcm_prev.iind_minimum_filtered < IIND_DCM_WARNING) {
        dcm_prev.dcm_imminent = 1;
    } else {
        dcm_prev.dcm_imminent = 0;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 5.3: DCM PREVENTION ACTION
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    #if DCM_PREVENTION_ACTIVE
    
    if (dcm_prev.dcm_detected || dcm_prev.dcm_imminent) {
        dcm_prev.prevention_active = 1;
        dcm_prev.prevention_cycles++;
        
        // Boost duty'yi azaltarak (daha fazla enerji aktararak) minimum akimi artir
        // veya Buck duty'yi azaltarak akisi dusur
        
        if (dcm_prev.dcm_detected) {
            // Gercek DCM - acil mudahale
            dcm_prev.duty_boost_request = -0.005f;  // Boost duty azalt (daha fazla boost)
        } else {
            // DCM yaklasiyor
            float margin_ratio = dcm_prev.dcm_margin / IIND_DCM_WARNING;
            dcm_prev.duty_boost_request = -0.002f * (1.0f - margin_ratio);
        }
    } else {
        dcm_prev.prevention_active = 0;
        dcm_prev.duty_boost_request = 0.0f;
        dcm_prev.prevention_cycles = 0;
    }
    
    #else
    dcm_prev.prevention_active = 0;
    dcm_prev.duty_boost_request = 0.0f;
    #endif
    
} else {
    // Soft-start sırasında - DCM prevention pasif
    dcm_prev.prevention_active = 0;
    dcm_prev.duty_boost_request = 0.0f;
    dcm_prev.dcm_detected = 0;
    dcm_prev.dcm_imminent = 0;
}
// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 6: PREDICTIVE THERMAL LIMITER (LMG5200 BASED)                                 ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝
// NOT: Yapay yumusak limitler kaldirildi. Sistem artik LMG5200'un tam kapasitesini kullaniyor:
//      - 10A surekli akim (termal limit)
//      - 35A mutlak peak (aninda koruma)
//      - 235us termal zamanlayici

{
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 6.1: ABSOLUTE PEAK PROTECTION (RULE 1) - EN YUKSEK ONCELIK
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // I_L >= 35A: ANINDA duty = 0.02, hic gecikme yok!
    
    if (filt_iind >= I_PEAK_ABSOLUTE) {
        // KRITIK: 35A mutlak peak - ANINDA KAPAT
        thermal_limiter.peak_absolute = 1;
        thermal_limiter.thermal_clamp_active = 1;
        thermal_limiter.duty_reduction_request = 1.0f;  // Maksimum azaltma
        thermal_limiter.peak_events++;
        thermal_limiter.peak_value = filt_iind;
        diag.ocp_events++;
        
        // Duty'yi aninda discharge moduna al
        // (Bu deger main control loop'ta uygulanacak)
    }
    
    // 30A peak uyarisi - duty azaltmaya basla
    else if (filt_iind >= I_PEAK_SAFETY) {
        thermal_limiter.peak_warning = 1;
        thermal_limiter.peak_absolute = 0;
        
        // Lineer duty azaltma: 30A'da yavas, 35A'ya yaklastikca hizlanir
        float peak_ratio = (filt_iind - I_PEAK_SAFETY) / (I_PEAK_ABSOLUTE - I_PEAK_SAFETY);
        thermal_limiter.duty_reduction_request = DUTY_REDUCTION_RATE_NORMAL + 
                                                  peak_ratio * (DUTY_REDUCTION_RATE_FAST - DUTY_REDUCTION_RATE_NORMAL);
    }
    else {
        thermal_limiter.peak_warning = 0;
        thermal_limiter.peak_absolute = 0;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 6.2: THERMAL TIMER LOGIC (RULE 2 & 3)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // I_L > 10A: Timer baslat/artir
    // I_L <= 10A: Timer sifirla ANINDA
    // Timer > 235us: Duty = 0.02 kilitle
    
// V8.16: THERMAL TIMER AKTİF
#if 1
    if (filt_iind > I_CONTINUOUS_MAX) {
        // Akim 10A uzerinde - timer calistir
        thermal_limiter.timer_active = 1;
        thermal_limiter.overcurrent_timer += TS;
        thermal_limiter.total_overcurrent_time += TS;
        
        // Timer doldu mu? (235us)
        // V8.16: Boost mode sırasında clamp'ı ATLAT
        if (thermal_limiter.overcurrent_timer >= TIME_OVERCURRENT_LIMIT && !load_connect_boost_active) {
            // TERMAL LIMIT ASILDI - duty'yi 0.02'ye kilitle
            thermal_limiter.thermal_clamp_active = 1;
            thermal_limiter.overcurrent_events++;
            diag.current_limit_events++;
        }
    }
    else {
        // Akim 10A veya altinda - timer ANINDA sifirla
        thermal_limiter.timer_active = 0;
        thermal_limiter.overcurrent_timer = 0.0f;
        
        // Soft recovery baslat (eger klamp aktifse)
        if (thermal_limiter.thermal_clamp_active && !thermal_limiter.soft_recovery_active) {
            thermal_limiter.soft_recovery_active = 1;
            thermal_limiter.recovery_timer = 0.0f;
            thermal_limiter.recovery_duty = DUTY_DISCHARGE;  // 0.02'den basla
        }
    }
#else
    // Thermal timer devre dışı - timer durumu sıfırla
    thermal_limiter.timer_active = 0;
    thermal_limiter.overcurrent_timer = 0.0f;
#endif
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 6.3: dI/dt SLOPE CALCULATION (50 orneklik pencere)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Buffer'a yeni ornek ekle
    thermal_limiter.current_buffer[thermal_limiter.buffer_index] = filt_iind;
    thermal_limiter.buffer_index = (thermal_limiter.buffer_index + 1) % SLOPE_BUFFER_SIZE;
    
    if (!thermal_limiter.buffer_filled && thermal_limiter.buffer_index == 0) {
        thermal_limiter.buffer_filled = 1;
    }
    
    // dI/dt hesapla (buffer doluysa)
    if (thermal_limiter.buffer_filled) {
        // Basit lineer regresyon yerine: ilk ve son 10 ornegin ortalamasini kullan
        float old_sum = 0.0f, new_sum = 0.0f;
        int old_start = thermal_limiter.buffer_index;  // En eski ornek
        int new_start = (thermal_limiter.buffer_index + SLOPE_BUFFER_SIZE - 10) % SLOPE_BUFFER_SIZE;  // En yeni 10
        
        for (int i = 0; i < 10; i++) {
            old_sum += thermal_limiter.current_buffer[(old_start + i) % SLOPE_BUFFER_SIZE];
            new_sum += thermal_limiter.current_buffer[(new_start + i) % SLOPE_BUFFER_SIZE];
        }
        
        float old_avg = old_sum / 10.0f;
        float new_avg = new_sum / 10.0f;
        float time_diff = (SLOPE_BUFFER_SIZE - 10) * TS;  // 40us
        
        thermal_limiter.di_dt_raw = (new_avg - old_avg) / time_diff;  // A/s
        
        // Filtrele
        thermal_limiter.di_dt_filtered = thermal_limiter.di_dt_filtered * (1.0f - SLOPE_FILTER_ALPHA) + 
                                          thermal_limiter.di_dt_raw * SLOPE_FILTER_ALPHA;
    }
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 6.4: PREDICTIVE MODULATION (RULE 4) - AKILLI DUTY KONTROLU
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // Bolge: 10A < I_L < 35A VE Timer < 235us
    // - dI/dt negatifse (akim dusuyor): 10A'ya timer dolmadan inecek mi hesapla
    // - Eger inecekse: duty azaltmayi DURDUR, dogal dususe izin ver
    // - Inmeyecekse: duty azaltmaya devam et
    
    thermal_limiter.modulation_active = 0;
    thermal_limiter.hold_reduction = 0;
    thermal_limiter.prediction_valid = 0;
    
// V8.16: OCP AKTİF
#if 1
    if (filt_iind > I_CONTINUOUS_MAX && 
        filt_iind < I_PEAK_ABSOLUTE && 
        thermal_limiter.overcurrent_timer < TIME_OVERCURRENT_LIMIT &&
        !thermal_limiter.thermal_clamp_active) {
        
        thermal_limiter.modulation_active = 1;
        
        // ════════════════════════════════════════════════════════════════════════════
        // V8.15: LİNEER AKIM × ZAMAN KORUMA SİSTEMİ
        // ════════════════════════════════════════════════════════════════════════════
        // Akım Katsayısı: 9A = %1, 35A = %100 (lineer)
        // Zaman Katsayısı: 0µs = %0, 225µs = %100 (lineer)
        // Toplam Kuvvet = Baz_Kuvvet × Akım_Katsayısı × Zaman_Katsayısı
        // ════════════════════════════════════════════════════════════════════════════
        
        const float I_LOW = 9.0f;    // Alt sınır (bu ve altı = %1)
        const float I_HIGH = 35.0f;  // Üst sınır (bu = %100)
        const float T_MAX = 0.000225f;  // 225µs = %100
        
        float timer = thermal_limiter.overcurrent_timer;
        
        // ─────────────────────────────────────────────────────────────────────────────
        // AKIM KATSAYISI: 9A→%1, 35A→%100
        // ─────────────────────────────────────────────────────────────────────────────
        float current_mult;
        if (filt_iind <= I_LOW) {
            current_mult = 0.01f;  // %1 minimum
        } else if (filt_iind >= I_HIGH) {
            current_mult = 1.0f;   // %100 maximum
        } else {
            // Lineer interpolasyon: (I - 9) / (35 - 9) → %1 ile %100 arasında
            current_mult = 0.01f + (filt_iind - I_LOW) / (I_HIGH - I_LOW) * 0.99f;
        }
        
        // ─────────────────────────────────────────────────────────────────────────────
        // ZAMAN KATSAYISI: 0µs→%0, 225µs→%100
        // ─────────────────────────────────────────────────────────────────────────────
        float time_mult = timer / T_MAX;
        time_mult = Clamp_Float(time_mult, 0.0f, 1.0f);
        
        // ─────────────────────────────────────────────────────────────────────────────
        // TOPLAM KUVVET = BAZ × AKIM × ZAMAN
        // ─────────────────────────────────────────────────────────────────────────────
        // Örnek: 12A ve 20µs'de:
        //   current_mult = 0.01 + (12-9)/(35-9)*0.99 = 0.01 + 0.115*0.99 ≈ 0.124 (%12.4)
        //   time_mult = 20µs / 225µs ≈ 0.089 (%8.9)
        //   Toplam = 0.124 × 0.089 ≈ 0.011 (%1.1)
        // ─────────────────────────────────────────────────────────────────────────────
        float combined_mult = current_mult * time_mult;
        
        // V8.16: Load connect boost mode - OCP kuvvetini %10'a düşür
        float boost_multiplier = load_connect_boost_active ? 0.1f : 1.0f;
        
        // Baz düşürme hızı (tam güçte ne kadar düşürecek)
        float base_reduction = DUTY_REDUCTION_RATE_FAST;
        
        // Final duty reduction request - boost mode sırasında %10
        thermal_limiter.duty_reduction_request = base_reduction * combined_mult * boost_multiplier;
        
        // Slew damping: boost mode sırasında tam hız (1.0)
        if (load_connect_boost_active) {
            thermal_limiter.slew_damping = 1.0f;  // Boost mode: tam slew rate
        } else {
            thermal_limiter.slew_damping = 1.0f - combined_mult;  // Normal: akım/zaman bazlı
            thermal_limiter.slew_damping = Clamp_Float(thermal_limiter.slew_damping, 0.1f, 1.0f);
        }
        
        // Hold reduction sadece çok yüksek kombinasyonlarda aktif
        thermal_limiter.hold_reduction = (combined_mult > 0.5f) ? 1 : 0;
        
        // Prediction tracking
        thermal_limiter.prediction_valid = 1;
        thermal_limiter.projected_time_to_safe = 0.0f;
        float di_dt = thermal_limiter.di_dt_filtered;
        if (di_dt < -100.0f) {
            float delta = filt_iind - I_CONTINUOUS_MAX;
            thermal_limiter.projected_time_to_safe = delta / Abs_Float(di_dt);
        }
    } else {
        // 10A altında veya clamp aktif - slew damping sıfırla
        thermal_limiter.slew_damping = 1.0f;
    }
#else
    // OCP devre dışı - slew damping normal
    thermal_limiter.slew_damping = 1.0f;
#endif
    
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // 6.5: SOFT RECOVERY (Klamptan cikis)
    // ────────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // I_L < 10A VE thermal_clamp_active: Yavasca normal duty'ye rampa
    
    if (thermal_limiter.soft_recovery_active) {
        thermal_limiter.recovery_timer += TS;
        
        // Yalnizca akim guvenli bolgedeyse recovery'ye izin ver
        if (filt_iind < I_CONTINUOUS_MAX * 0.9f) {  // 9A altinda
            // Yavas rampa ile duty artir
            thermal_limiter.recovery_duty += DUTY_RECOVERY_RATE;
            
            // Hedef duty'ye ulasti mi?
            if (thermal_limiter.recovery_duty >= timer_a_duty) {
                // Recovery tamamlandi
                thermal_limiter.soft_recovery_active = 0;
                thermal_limiter.thermal_clamp_active = 0;
                thermal_limiter.recovery_duty = 0.0f;
                thermal_limiter.time_in_clamp = 0.0f;
            }
        }
        else {
            // Akim hala yuksek - recovery'yi durdur
            thermal_limiter.recovery_duty = DUTY_DISCHARGE;
        }
    }
    
    // Klamp suresini takip et
    if (thermal_limiter.thermal_clamp_active) {
        thermal_limiter.time_in_clamp += TS;
    }
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 7: PROTECTION CHECKS                                                         ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 7.1: PEAK CURRENT I2t TRACKER
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

{
    // Catastrophic check (40A) - ANINDA KAPAT
    if (filt_iind > IIND_PEAK_ABSOLUTE || filt_iin > IIN_CATASTROPHIC) {
        peak_tracker.instant_fault = 1;
        fault.fault_active = 1;
        fault.fault_type = FAULT_OCP;
        fault.fault_value = Max_Float(filt_iind, filt_iin);
        fault.ocp_trip = 1;
        diag.ocp_events++;
    }
    
    // Peak limit (30A) - I2t korumali
    if (filt_iind > IIND_PEAK_LIMIT) {
        peak_tracker.peak_active = 1;
        peak_tracker.peak_timer += TS;
        
        if (peak_tracker.peak_timer > IIND_PEAK_TIME_LIMIT) {
            peak_tracker.peak_fault = 1;
            fault.fault_active = 1;
            fault.fault_type = FAULT_PEAK_I2T;
            fault.fault_value = filt_iind;
            diag.ocp_events++;
        }
    } else {
        peak_tracker.peak_active = 0;
        if (peak_tracker.peak_timer > 0.0f) {
            peak_tracker.peak_timer -= TS * 0.5f;
            if (peak_tracker.peak_timer < 0.0f) peak_tracker.peak_timer = 0.0f;
        }
    }
}

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 7.2: OVP CHECK
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

{
    static int ovp_debounce = 0;
    
    if (filt_vout > OVP_ABSOLUTE) {
        // Mutlak OVP - anında trip
        fault.fault_active = 1;
        fault.fault_type = FAULT_OVP;
        fault.fault_value = filt_vout;
        fault.ovp_trip = 1;
        diag.ovp_events++;
        
    } else if (filt_vout > OVP_TRIP) {
        ovp_debounce++;
        if (ovp_debounce > FAULT_DEBOUNCE_OVP) {
            fault.fault_active = 1;
            fault.fault_type = FAULT_OVP;
            fault.fault_value = filt_vout;
            fault.ovp_trip = 1;
            diag.ovp_events++;
        }
    } else {
        if (ovp_debounce > 0) ovp_debounce--;
    }
}

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 7.3: INPUT OCP & SHOOT-THROUGH
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

{
    // Peak tracking
    if (filt_iin > input_ocp.iin_peak) {
        input_ocp.iin_peak = filt_iin;
    }
    
    // Warning
    input_ocp.warning_active = (filt_iin > IIN_OCP_THRESHOLD * 0.8f) ? 1 : 0;
    
    // OCP detection
    if (filt_iin > IIN_OCP_THRESHOLD) {
        input_ocp.ocp_suspected = 1;
        input_ocp.debounce_counter++;
        
        if (input_ocp.debounce_counter > FAULT_DEBOUNCE_IIN) {
            input_ocp.ocp_proven = 1;
            input_ocp.iin_at_fault = filt_iin;
            fault.fault_active = 1;
            fault.fault_type = FAULT_IIN_OCP;
            fault.iin_ocp_trip = 1;
            diag.iin_ocp_events++;
        }
    } else {
        input_ocp.ocp_suspected = 0;
        if (input_ocp.debounce_counter > 0) input_ocp.debounce_counter--;
    }
    
    // Shoot-through detection
    int st_conditions = 0;
    if (filt_iin > SHOOTTHROUGH_IIN_THRESHOLD) st_conditions++;
    if (filt_vout < SHOOTTHROUGH_VOUT_LOW) st_conditions++;
    
    if (st_conditions >= 2) {
        input_ocp.shootthrough_suspected = 1;
        input_ocp.shootthrough_evidence_timer += TS;
        
        if (input_ocp.shootthrough_evidence_timer > SHOOTTHROUGH_PROOF_TIME) {
            input_ocp.shootthrough_proven = 1;
            fault.fault_active = 1;
            fault.fault_type = FAULT_SHOOTTHROUGH;
            fault.shootthrough_trip = 1;
            diag.shootthrough_events++;
        }
    } else {
        input_ocp.shootthrough_suspected = 0;
        if (input_ocp.shootthrough_evidence_timer > 0.0f) {
            input_ocp.shootthrough_evidence_timer -= TS * 0.5f;
        }
    }
}

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 7.4: I2t THERMAL INTEGRATOR
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

{
    // Enerji birikimi
    if (filt_iind > I2T_CURRENT_NOMINAL) {
        float excess = filt_iind - I2T_CURRENT_NOMINAL;
        float energy_rate = excess * excess * 0.05f;  // I²R
        
        if (filt_iind > I2T_CURRENT_CRITICAL) {
            thermal_int.energy_accumulated += energy_rate * TS * 2.0f;
        } else if (filt_iind > I2T_CURRENT_WARNING) {
            thermal_int.energy_accumulated += energy_rate * TS;
        } else {
            thermal_int.energy_accumulated += energy_rate * TS * 0.5f;
        }
        
        thermal_int.cooling_trend = 0;
    }
    
    // Enerji azalması
    if (filt_iind < I2T_CURRENT_WARNING) {
        float decay = (filt_iind < I2T_CURRENT_NOMINAL) ? I2T_DECAY_FAST : I2T_DECAY_RATE;
        thermal_int.energy_accumulated -= decay;
        if (thermal_int.energy_accumulated < 0.0f) thermal_int.energy_accumulated = 0.0f;
        thermal_int.cooling_trend = 1;
    }
    
    // Peak
    if (thermal_int.energy_accumulated > thermal_int.energy_peak) {
        thermal_int.energy_peak = thermal_int.energy_accumulated;
    }
    
    // Warning/Critical flags
    thermal_int.warning_active = (thermal_int.energy_accumulated > I2T_ENERGY_WARNING) ? 1 : 0;
    thermal_int.critical_active = (thermal_int.energy_accumulated > I2T_ENERGY_MAX_SUSTAINED) ? 1 : 0;
    
    // Trip
    if (thermal_int.energy_accumulated > I2T_ENERGY_MAX_BURST) {
        thermal_int.trip_proven = 1;
        fault.fault_active = 1;
        fault.fault_type = FAULT_THERMAL;
        fault.thermal_trip = 1;
        diag.thermal_events++;
    }
}

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 7.5: IMPEDANCE OBSERVER
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

{
    if (filt_iout > 0.5f && filt_vout > 0.5f) {
        impedance_obs.r_load_instant = filt_vout / filt_iout;
        impedance_obs.r_load_instant = Clamp_Float(impedance_obs.r_load_instant, 0.5f, 1000.0f);
        
        impedance_obs.r_load_filtered = impedance_obs.r_load_filtered * 0.999f + 
                                         impedance_obs.r_load_instant * 0.001f;
        
        impedance_obs.confidence_timer += TS;
        if (impedance_obs.confidence_timer > 0.005f) {
            impedance_obs.confidence = Min_Float(impedance_obs.confidence + 0.001f, 1.0f);
            impedance_obs.is_valid = 1;
        }
    } else {
        impedance_obs.confidence_timer = 0.0f;
        impedance_obs.confidence *= 0.999f;
        if (impedance_obs.confidence < 0.3f) impedance_obs.is_valid = 0;
    }
}

// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────
// 7.6: POWER & EFFICIENCY
// ────────────────────────────────────────────────────────────────────────────────────────────────────────────────────

{
    power_eff.pin_instant = filt_vin * filt_iin;
    power_eff.pin_instant = Clamp_Float(power_eff.pin_instant, 0.0f, 500.0f);
    
    power_eff.pout_instant = filt_vout * filt_iout;
    power_eff.pout_instant = Clamp_Float(power_eff.pout_instant, 0.0f, 100.0f);
    
    power_eff.pin_filtered = power_eff.pin_filtered * 0.99f + power_eff.pin_instant * 0.01f;
    power_eff.pout_filtered = power_eff.pout_filtered * 0.99f + power_eff.pout_instant * 0.01f;
    
    if (power_eff.pin_filtered > 1.0f) {
        power_eff.efficiency_instant = power_eff.pout_filtered / power_eff.pin_filtered;
        power_eff.efficiency_instant = Clamp_Float(power_eff.efficiency_instant, 0.0f, 1.0f);
    }
    
    power_eff.efficiency_filtered = power_eff.efficiency_filtered * 0.999f + 
                                     power_eff.efficiency_instant * 0.001f;
    power_eff.efficiency_percent = power_eff.efficiency_filtered * 100.0f;
    
    power_eff.energy_in_total += power_eff.pin_instant * TS;
    power_eff.energy_out_total += power_eff.pout_instant * TS;
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 8: SOFT-START STATE MACHINE (9-Phase)                                        ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝

if (sys_state == STATE_SOFTSTART) {
    // V8.6 Correction: Update timers!
    ss.phase_timer += TS;
    ss.total_timer += TS;
    
    switch (ss.phase) {
        
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_INIT: V8.6 - 250µs Algılama Penceresi Başlangıcı
        // Her iki taraf MIN duty ile başlar, mod henüz belirlenmez
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_INIT:
            // V8.9: FREEWHEELING başlangıç - mod algılanana kadar güvenli mod
            // Buck HO OFF (LO ON), Boost HO OFF (LO ON) → Induktör freewheeling
            ss.duty_buck_current = DUTY_MIN;   // Buck HO OFF → LO ON
            ss.duty_boost_current = DUTY_MAX;  // V8.9 FIX: DUTY_MIN → DUTY_MAX (Boost LO ON)
            prev_timer_a = DUTY_MIN;
            prev_timer_b = DUTY_MAX;           // V8.9 FIX: DUTY_MIN → DUTY_MAX
            timer_a_duty = DUTY_MIN;           // Output'a da yaz
            timer_b_duty = DUTY_MAX;           // Output'a da yaz
            
            // Mod henüz belirsiz - SS_PHASE_MEASURE sonunda belirlenecek
            ss.operating_mode = MODE_BOOST;  // Güvenli başlangıç
            
            ss.iind_limit_active = SS_IIND_LIMIT_MEASURE;
            slew_limit = SLEW_PRECHARGE;
            
            ss.phase = SS_PHASE_MEASURE;
            ss.phase_timer = 0.0f;
            ss.vin_sample_count = 0;
            ss.vin_measured = filt_vin;
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_MEASURE: V8.8 - 100µs FREEWHEELING Mod Algılama
        // Her iki LO tam iletimde (Buck LO + Boost LO ON), HO'lar OFF
        // Bu sayede induktör freewheeling yapar, Vin güvenle ölçülür
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_MEASURE:
            // V8.8: FREEWHEELING - Her iki LO tam iletim
            // Buck:  timer_a = MIN → Q1(HO) OFF, Q2(LO) ON
            // Boost: timer_b = MAX → Q3(LO) ON, Q4(HO) OFF
            ss.duty_buck_current = DUTY_MIN;
            ss.duty_boost_current = DUTY_MAX;   // V8.8: Boost LO ON for freewheeling
            prev_timer_a = DUTY_MIN;
            prev_timer_b = DUTY_MAX;            // V8.8: Boost LO ON for freewheeling
            
            ss.iind_limit_active = SS_IIND_LIMIT_MEASURE;
            slew_limit = SLEW_PRECHARGE;
            
            // Vin sample toplama
            if (ss.vin_sample_count < 64) {
                ss.vin_samples[ss.vin_sample_count] = filt_vin;
                ss.vin_sample_count++;
            }
            
            // 250µs doldu mu? MOD KARARI VER
            if (ss.phase_timer >= SS_MEASURE_TIME) {
                // Ortalama hesapla
                if (ss.vin_sample_count > 0) {
                    float sum = 0.0f;
                    for (int i = 0; i < ss.vin_sample_count; i++) {
                        sum += ss.vin_samples[i];
                    }
                    ss.vin_measured = sum / (float)ss.vin_sample_count;
                } else {
                    ss.vin_measured = filt_vin;
                }
                
                // ═══════════════════════════════════════════════════════════════════════
                // V8.6: KESİN MOD BELİRLEME - 250µs Vin ortalamasına göre
                // ═══════════════════════════════════════════════════════════════════════
                if (ss.vin_measured > VIN_BUCK_THRESHOLD) {
                    ss.operating_mode = MODE_BUCK;
                    // BUCK: Boost = MIN (OFF), Buck = soft-start yapacak (2.5ms)
                    ss.duty_buck_current = DUTY_MIN;   // Buck soft-start başlangıcı
                    ss.duty_boost_current = DUTY_MIN;  // TITAN: Boost LO OFF (Buck mode)
                    prev_timer_a = DUTY_MIN;
                    prev_timer_b = DUTY_MIN;
                } else if (ss.vin_measured < VIN_BOOST_THRESHOLD) {
                    ss.operating_mode = MODE_BOOST;
                    // BOOST: Buck = MIN'den MAX'a ramp (250µs), Boost = soft-start (2.5ms)
                    // V8.9 FIX: Buck da MIN'den başlasın, rampa sonraki fazlarda yapılacak
                    ss.duty_buck_current = DUTY_MIN;   // V8.9: Buck ramp başlangıcı
                    ss.duty_boost_current = DUTY_MIN;  // Boost soft-start başlangıcı
                    prev_timer_a = DUTY_MIN;           // V8.9: Ramp için MIN'den başla
                    prev_timer_b = DUTY_MIN;
                } else {
                    ss.operating_mode = MODE_BUCKBOOST;  // V8.17: Varsayılan Buck-Boost
                    // BUCK-BOOST: Her ikisi de soft-start
                    ss.duty_buck_current = DUTY_MIN;
                    ss.duty_boost_current = DUTY_MIN;
                    prev_timer_a = DUTY_MIN;
                    prev_timer_b = DUTY_MIN;
                }
                
                // Sonraki faza geç
                ss.phase = SS_PHASE_CHARACTERIZE;
                ss.phase_timer = 0.0f;
            }
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_CHARACTERIZE: V8.6 - Mod kararı sonrası karakterizasyon
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_CHARACTERIZE:
            // V8.9: Her iki tarafın prev_timer değerini mevcut duty'ye senkronize et
            // Ramp sonraki fazlarda yapılacak - burada sadece senkronizasyon
            if (ss.operating_mode == MODE_BOOST) {
                // BOOST: Buck ve Boost - mevcut değerlerle devam et
                // V8.9 FIX: Buck'ı anında MAX yapmak yerine, ramp değerini koru
                prev_timer_a = ss.duty_buck_current;   // Ramp devam edecek
                prev_timer_b = ss.duty_boost_current;
            } else if (ss.operating_mode == MODE_BUCK) {
                // BUCK: Boost = MIN (OFF), Buck = soft-start
                ss.duty_boost_current = DUTY_MIN;  // TITAN: Boost LO OFF
                timer_b_duty = DUTY_MIN;
                prev_timer_b = DUTY_MIN;
                // Buck soft-start - mevcut değerle senkronize et
                prev_timer_a = ss.duty_buck_current;
            } else {
                // BUCK-BOOST: Her ikisi soft-start
                prev_timer_a = ss.duty_buck_current;
                prev_timer_b = ss.duty_boost_current;
            }
            
            ss.iind_limit_active = SS_IIND_LIMIT_CHAR;
            slew_limit = SLEW_PRECHARGE;
            
            // Hedef duty hesaplama
            if (ss.operating_mode == MODE_BUCK) {
                // Buck: D = Vout/Vin
                ss.duty_buck_target = pid_target_dynamic / ss.vin_measured;
                ss.duty_buck_target = Clamp_Float(ss.duty_buck_target, 0.3f, DUTY_BUCK_MAX);
                ss.duty_boost_target = DUTY_MIN;   // TITAN: Boost LO OFF (Buck mode)
                
            } else if (ss.operating_mode == MODE_BOOST) {
                // Boost: D_buck = max, D_boost hedef hesapla
                ss.duty_buck_target = DUTY_MAX;
                ss.duty_boost_target = 1.0f - (ss.vin_measured / pid_target_dynamic);  // V8.9: dynamic target
                ss.duty_boost_target = Clamp_Float(ss.duty_boost_target, DUTY_BOOST_MIN, 0.5f);
                
            } else {
                // Buck-Boost
                ss.duty_buck_target = pid_target_dynamic / ss.vin_measured;  // V8.9: dynamic target
                ss.duty_buck_target = Clamp_Float(ss.duty_buck_target, 0.7f, DUTY_BUCK_MAX);
                
                float boost_need = (pid_target_dynamic - ss.vin_measured) / pid_target_dynamic;  // V8.9
                if (boost_need > 0.0f) {
                    ss.duty_boost_target = DUTY_MIN + (boost_need * 0.3f);  // TITAN: LO artış
                } else {
                    ss.duty_boost_target = DUTY_MIN;  // TITAN: Boost LO OFF
                }
            }
            
            // Başlangıç duty'leri
            ss.duty_buck_start = DUTY_MIN;
            ss.duty_boost_start = DUTY_MIN;    // TITAN: Boost LO OFF başla
            
            // Multi-slope ramp hesaplama
            float total_ramp_time = SS_RAMP_SLOW_TIME + SS_RAMP_FAST_TIME + SS_RAMP_SETTLE_TIME;
            float delta_buck = ss.duty_buck_target - ss.duty_buck_start;
            
            // Toplam cycles
            float total_cycles = total_ramp_time / TS;
            
            // Ramp rate (ortalama)
            float base_rate_buck = delta_buck / total_cycles;
            
            // Faz bazinda oranlar
            ss.ramp_rate_slow = base_rate_buck * RAMP_SLOW_RATIO;
            ss.ramp_rate_fast = base_rate_buck * RAMP_FAST_RATIO * 1.3f;  // Biraz daha hizli
            ss.ramp_rate_settle = base_rate_buck * RAMP_SETTLE_RATIO;
            
            // Yuk tahmini (Vout / varsayilan akim)
            ss.load_estimate = R_LOAD_NOMINAL;
            
            if (ss.phase_timer >= SS_CHARACTERIZE_TIME) {
                ss.phase = SS_PHASE_PRECHARGE;
                ss.phase_timer = 0.0f;
                ss.vout_at_start = filt_vout;
            }
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_PRECHARGE: 20µs Yavas on-sarj (V8.5 Mode-Aware)
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_PRECHARGE:
            ss.iind_limit_active = SS_IIND_LIMIT_PRECHARGE;
            slew_limit = SLEW_PRECHARGE;
            
            // V8.9 USER SPEC: 100µs ölçme, 250µs karşı bacak, 2.5ms ana bacak
            // BOOST modunda: Buck = MIN→MAX (250µs), Boost = MIN→hedef (2.5ms)
            // BUCK modunda: Boost = MIN (OFF), Buck = MIN→hedef (2.5ms)
            if (ss.operating_mode == MODE_BOOST) {
                // BOOST MODE: Buck ramp (MIN → MAX, 250µs'de tamamlanır)
                ss.duty_buck_current += SS_OPPOSITE_RAMP_RATE;  // 0.004 = 250µs'de MAX
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, DUTY_MAX);
                // Boost ramp (MIN → hedef, 2.5ms'de tamamlanır)
                ss.duty_boost_current += SS_MAIN_RAMP_RATE;     // 0.0004 = 2.5ms'de hedef
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, 
                    DUTY_MIN, ss.duty_boost_target);
            } else {
                // BUCK veya BUCK-BOOST MODE: Boost = MIN (OFF), Buck ramp (2.5ms)
                ss.duty_boost_current = DUTY_MIN;  // TITAN: Boost LO OFF - ANINDA
                prev_timer_b = DUTY_MIN;
                timer_b_duty = DUTY_MIN;
                ss.duty_buck_current += SS_MAIN_RAMP_RATE;      // 0.0004 = 2.5ms'de hedef
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, 
                    DUTY_MIN, ss.duty_buck_target);
            }
            
            // Akim kontrolu (thermal limiter)
            if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_warning) {
                if (ss.operating_mode == MODE_BOOST) {
                    ss.duty_boost_current = DUTY_MIN;  // TITAN: Boost'u kapat (LO OFF)
                } else {
                    ss.duty_buck_current = DUTY_MIN;   // Buck'ı kapat
                }
            }
            
            if (ss.phase_timer >= SS_PRECHARGE_TIME) {
                ss.phase = SS_PHASE_RAMP_SLOW;
                ss.phase_timer = 0.0f;
                ss.vout_at_precharge = filt_vout;
            }
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_RAMP_SLOW: 30µs Yavas ramp baslangici (V8.5 Mode-Aware)
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_RAMP_SLOW:
            ss.iind_limit_active = SS_IIND_LIMIT_RAMP_SLOW;
            slew_limit = SLEW_RAMP_SLOW;
            ss.ramp_rate_current = ss.ramp_rate_slow;
            
            // V8.9 USER SPEC: RAMP SLOW
            if (ss.operating_mode == MODE_BOOST) {
                // ═══════════════════════════════════════════════════════════════════
                // BOOST MODE: Buck ramp → MAX (250µs), Boost ramp → hedef (2.5ms)
                // ═══════════════════════════════════════════════════════════════════
                ss.duty_buck_current += SS_OPPOSITE_RAMP_RATE;  // 250µs'de MAX
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, DUTY_MAX);
                
                // Termal limiter kontrolu
                if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                    ss.duty_boost_current = DUTY_MIN;  // Boost kapat
                } else if (thermal_limiter.peak_warning) {
                    ss.duty_boost_current -= thermal_limiter.duty_reduction_request;
                } else {
                    // Normal ramp - Boost duty ARTAR (2.5ms'de hedefe)
                    ss.duty_boost_current += SS_MAIN_RAMP_RATE;
                }
                
                // Clamp - hedefin %60'ına kadar bu fazda
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, 
                    DUTY_MIN, ss.duty_boost_target * 0.6f);
                    
            } else if (ss.operating_mode == MODE_BUCK) {
                // ═══════════════════════════════════════════════════════════════════
                // BUCK MODE: Buck ramp, Boost = DUTY_MIN (OFF) - V8.9: ANINDA
                // ═══════════════════════════════════════════════════════════════════
                ss.duty_boost_current = DUTY_MIN;  // TITAN: Boost LO OFF
                prev_timer_b = DUTY_MIN;           // V8.9: Slew bypass
                timer_b_duty = DUTY_MIN;           // V8.9: Output'a da yaz
                
                // Termal limiter kontrolu
                if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                    ss.duty_buck_current = DUTY_DISCHARGE;
                } else if (thermal_limiter.peak_warning) {
                    ss.duty_buck_current -= thermal_limiter.duty_reduction_request;
                } else if (thermal_limiter.modulation_active && !pred.natural_return_ok) {
                    if (thermal_limiter.hold_reduction) {
                        ss.duty_buck_current += ss.ramp_rate_current * 0.3f;
                    } else {
                        ss.duty_buck_current -= thermal_limiter.duty_reduction_request;
                    }
                } else {
                    // Normal ramp
                    ss.duty_buck_current += ss.ramp_rate_current;
                }
                
                // Clamp - bu fazda hedefin %60'ına kadar
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, ss.duty_buck_target * 0.6f);
                
            } else {
                // ═══════════════════════════════════════════════════════════════════
                // BUCK-BOOST MODE: Her ikisi de ramp
                // ═══════════════════════════════════════════════════════════════════
                // Termal limiter kontrolu
                if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                    ss.duty_buck_current = DUTY_DISCHARGE;
                    ss.duty_boost_current = DUTY_MIN;  // TITAN: Boost LO OFF
                } else if (thermal_limiter.peak_warning) {
                    ss.duty_buck_current -= thermal_limiter.duty_reduction_request;
                } else {
                    // Normal ramp - her ikisi de
                    ss.duty_buck_current += ss.ramp_rate_current;
                    ss.duty_boost_current -= ss.ramp_rate_current * 0.5f;
                }
                
                // Clamp - TITAN topolojisi: Boost LO artar (DUTY_MIN'den yukarı)
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, ss.duty_buck_target * 0.6f);
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, ss.duty_boost_target);
            }
            
            if (ss.phase_timer >= SS_RAMP_SLOW_TIME) {
                ss.phase = SS_PHASE_RAMP_FAST;
                ss.phase_timer = 0.0f;
            }
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_RAMP_FAST: 50µs Hizli ramp ortasi (V8.5 Mode-Aware)
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_RAMP_FAST:
            ss.iind_limit_active = SS_IIND_LIMIT_RAMP_FAST;
            slew_limit = SLEW_RAMP_FAST;
            ss.ramp_rate_current = ss.ramp_rate_fast;
            
            // V8.12 USER SPEC: RAMP FAST - Mode-aware with Buck mode restored
            if (ss.operating_mode == MODE_BUCK) {
                // ═══════════════════════════════════════════════════════════════════
                // BUCK MODE: Buck ramp → hedef (2.5ms), Boost = DUTY_MAX (LO ON)
                // V8.12: Boost modunun aynası, roller ters
                // ═══════════════════════════════════════════════════════════════════
                ss.duty_boost_current = DUTY_MAX;  // Boost LO ON (pasif)
                
                if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                    ss.duty_buck_current = DUTY_DISCHARGE;  // Buck kapat
                } else if (thermal_limiter.peak_warning) {
                    ss.duty_buck_current -= thermal_limiter.duty_reduction_request;
                } else {
                    // V8.13: Yüksek Vin yavaşlatması KALDIRILDI (7ms→3ms hedefe ulaşmak için)
                    // Orijinal: Vin>24V → %25 yavaş ramp - artık devre dışı
                    float buck_ramp_rate = SS_MAIN_RAMP_RATE;  // Sabit rate - yavaşlatma yok
                    ss.duty_buck_current += buck_ramp_rate;
                }
                
                // V8.12: VOUT TRACKING - OVP önleme
                // Eğer Vout hedefin üzerine çıkarsa duty'yi azalt
                if (filt_vout > pid_target_dynamic * 1.05f) {
                    // Vout hedefin %5 üzerinde - duty hızla azalt
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE * 3.0f;
                } else if (filt_vout > pid_target_dynamic) {
                    // Vout hedefin üzerinde - duty yavaşça azalt
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE;
                }
                
                // Clamp - hedefe yaklaş (%98)
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, 
                    DUTY_MIN, ss.duty_buck_target * 0.98f);
                    
            } else if (ss.operating_mode == MODE_BOOST) {
                // ═══════════════════════════════════════════════════════════════════
                // BOOST MODE: Boost ramp → hedef (2.5ms)
                // ═══════════════════════════════════════════════════════════════════
                ss.duty_buck_current += SS_OPPOSITE_RAMP_RATE;  // 250µs'de MAX
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, DUTY_MAX);
                
                if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                    ss.duty_boost_current = DUTY_MIN;  // Boost kapat
                } else if (thermal_limiter.peak_warning) {
                    ss.duty_boost_current -= thermal_limiter.duty_reduction_request;
                } else {
                    // Normal ramp - Boost duty ARTAR (2.5ms'de hedefe)
                    ss.duty_boost_current += SS_MAIN_RAMP_RATE;
                }
                
                // Clamp - hedefe yaklaş (%98)
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, 
                    DUTY_MIN, ss.duty_boost_target * 0.98f);
                    
            } else {
                // ═══════════════════════════════════════════════════════════════════
                // BUCK-BOOST MODE: Her ikisi de ramp
                // ═══════════════════════════════════════════════════════════════════
                if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                    ss.duty_buck_current = DUTY_DISCHARGE;
                    ss.duty_boost_current = DUTY_MAX;
                } else if (thermal_limiter.peak_warning) {
                    ss.duty_buck_current -= thermal_limiter.duty_reduction_request;
                } else {
                    // Normal ramp - her ikisi de
                    ss.duty_buck_current += ss.ramp_rate_current;
                    ss.duty_boost_current -= ss.ramp_rate_current * 0.3f;
                }
                
                // Clamp
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, ss.duty_buck_target * 0.98f);
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, ss.duty_boost_target, DUTY_MAX);
            }
            
            // Peak tracking
            if (filt_vout > ss.vout_peak) ss.vout_peak = filt_vout;
            if (filt_iind > ss.iind_peak_during_ss) ss.iind_peak_during_ss = filt_iind;
            
            if (ss.phase_timer >= SS_RAMP_FAST_TIME) {
                ss.phase = SS_PHASE_RAMP_SETTLE;
                ss.phase_timer = 0.0f;
            }
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_RAMP_SETTLE: 30µs Yavas ramp bitisi (V8.5 Mode-Aware)
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_RAMP_SETTLE:
            ss.iind_limit_active = SS_IIND_LIMIT_RAMP_SETTLE;
            slew_limit = SLEW_RAMP_SLOW;
            ss.ramp_rate_current = ss.ramp_rate_settle;
            
            // V8.12 USER SPEC: RAMP SETTLE - Mode-aware with Buck mode restored
            if (ss.operating_mode == MODE_BUCK) {
                // ═══════════════════════════════════════════════════════════════════
                // BUCK MODE: Boost YAVAS ramp → MAX, Buck ramp → hedef
                // V8.12: Boost modunun aynası, roller ters
                // ═══════════════════════════════════════════════════════════════════
                ss.duty_boost_current = DUTY_MAX;  // Boost LO ON (pasif)
                
                if (!thermal_limiter.thermal_clamp_active && !thermal_limiter.peak_warning) {
                    if (ss.duty_buck_current < ss.duty_buck_target) {
                        // V8.12: Yüksek Vin'de (>24V) ramp hızını %25 azalt
                        float buck_ramp_rate = SS_MAIN_RAMP_RATE;
                        if (filt_vin > 24.0f) {
                            buck_ramp_rate = SS_MAIN_RAMP_RATE * 0.75f;
                        }
                        ss.duty_buck_current += buck_ramp_rate;
                    }
                }
                
                // Termal limiter kontrolu
                if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                    ss.duty_buck_current = DUTY_DISCHARGE;  // Buck kapat
                } else if (thermal_limiter.peak_warning) {
                    ss.duty_buck_current -= thermal_limiter.duty_reduction_request;
                }
                
                // V8.12: VOUT TRACKING - OVP önleme
                if (filt_vout > pid_target_dynamic * 1.05f) {
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE * 3.0f;
                } else if (filt_vout > pid_target_dynamic) {
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE;
                }
                
                // Clamp - tam hedefe izin ver
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, ss.duty_buck_target);
                
            } else if (ss.operating_mode == MODE_BOOST) {
                // ═══════════════════════════════════════════════════════════════════
                // BOOST MODE: Buck YAVAS ramp → MAX, Boost ramp → hedef
                // V8.9 FIX: SS_OPPOSITE_RAMP_RATE → SS_MAIN_RAMP_RATE (60A spike önle)
                // ═══════════════════════════════════════════════════════════════════
                ss.duty_buck_current += SS_MAIN_RAMP_RATE;  // Yavaş ramp (60A önle)
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, DUTY_MAX);
                
                if (!thermal_limiter.thermal_clamp_active && !thermal_limiter.peak_warning) {
                    if (ss.duty_boost_current < ss.duty_boost_target) {
                        ss.duty_boost_current += SS_MAIN_RAMP_RATE;  // 2.5ms'de hedefe
                    }
                }
                
                // Termal limiter kontrolu
                if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                    ss.duty_boost_current = DUTY_MIN;  // Boost kapat
                } else if (thermal_limiter.peak_warning) {
                    ss.duty_boost_current -= thermal_limiter.duty_reduction_request;
                }
                
                // Clamp - tam hedefe izin ver
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, ss.duty_boost_target);
                
            } else {
                // ═══════════════════════════════════════════════════════════════════
                // BUCK-BOOST MODE: Her ikisi de tam hedefe yaklaş
                // ═══════════════════════════════════════════════════════════════════
                if (!thermal_limiter.thermal_clamp_active && !thermal_limiter.peak_warning) {
                    if (ss.duty_buck_current < ss.duty_buck_target) {
                        ss.duty_buck_current += ss.ramp_rate_current;
                    }
                    if (ss.duty_boost_current > ss.duty_boost_target) {
                        ss.duty_boost_current -= ss.ramp_rate_current * 0.5f;
                    }
                }
                
                // Termal limiter kontrolu
                if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                    ss.duty_buck_current = DUTY_DISCHARGE;
                    ss.duty_boost_current = DUTY_MAX;
                } else if (thermal_limiter.peak_warning) {
                    ss.duty_buck_current -= thermal_limiter.duty_reduction_request;
                }
                
                // Clamp
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, ss.duty_buck_target);
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, ss.duty_boost_target, DUTY_MAX);
            }
            
            if (ss.phase_timer >= SS_RAMP_SETTLE_TIME) {
                ss.phase = SS_PHASE_BLEND;
                ss.phase_timer = 0.0f;
                ss.vout_at_ramp_end = filt_vout;
                
                // Integrator seed hazirla
                if (ss.operating_mode == MODE_BUCK || ss.operating_mode == MODE_BUCKBOOST) {
                    float steady_state = pid_target_dynamic / filt_vin;
                    /* V8.18 FIX: Unclamped steady_state > 1.0 creates FALSE negative error against clamped duty */
                    steady_state = Clamp_Float(steady_state, DUTY_MIN, DUTY_MAX);
                    
                    ss.integrator_seed = (ss.duty_buck_current - steady_state) * 0.3f;
                    ss.integrator_seed = Clamp_Float(ss.integrator_seed, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS * 0.3f);
                } else {
                    ss.integrator_seed = 0.0f;
                }
                
                // PID baslangic
                prev_error = pid_target_dynamic - filt_vout;
                prev_prev_error = prev_error;
            }
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_BLEND: 5ms Sigmoid PID gecisi
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_BLEND:
            ss.iind_limit_active = SS_IIND_LIMIT_BLEND;
            slew_limit = SLEW_BLEND;
            
            // Sigmoid progress (0 -> 1)
            ss.sigmoid_progress = ss.phase_timer / SS_BLEND_TIME;
            ss.sigmoid_progress = Clamp_Float(ss.sigmoid_progress, 0.0f, 1.0f);
            
            // S-curve blend: 6t^5 - 15t^4 + 10t^3 (smootherstep)
            float t = ss.sigmoid_progress;
            ss.pid_blend_factor = t * t * t * (t * (t * 6.0f - 15.0f) + 10.0f);
            
            // PID hesapla (dusuk kazanclarla)
            error = pid_target_dynamic - filt_vout;
            error = Clamp_Float(error, -ERROR_CLAMP_STARTUP, ERROR_CLAMP_STARTUP);
            
            error_derivative = (error - prev_error) / TS;
            prev_prev_error = prev_error;
            prev_error = error;
            
            // Integrator (blend'li)
            float ki_blend = KI_BASE * ss.pid_blend_factor * 0.5f;
            integrator = ss.integrator_seed * (1.0f - ss.pid_blend_factor) + 
                         (integrator + ki_blend * error * TS);
            integrator = Clamp_Float(integrator, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS);
            
            // PID output
            float kp_blend = KP_BASE * ss.pid_blend_factor * 0.7f;
            float kd_blend = KD_BASE * ss.pid_blend_factor;
            
            float pid_out = kp_blend * error + integrator + kd_blend * error_derivative;
            
            // Feedforward + PID blend (blended_duty için gelecekte kullanılabilir)
            float feedforward_duty = pid_target_dynamic / filt_vin;
            feedforward_duty = Clamp_Float(feedforward_duty, DUTY_MIN, DUTY_MAX);
            (void)pid_out;  // Suppress unused warning - available for future PID blend use
            
            // V8.12: MODE-AWARE BLEND - Ramp devam etmeli, anında atlamak yok
            if (ss.operating_mode == MODE_BUCK) {
                // BUCK: Buck ramp → hedef (10ms), Boost = DUTY_MAX
                // V8.12: Boost modunun aynası, roller ters
                ss.duty_boost_current = DUTY_MAX;  // Boost LO ON (pasif)
                
                // Buck: PID blend ile soft hedef
                float buck_target_blend = ss.duty_buck_target * ss.pid_blend_factor + 
                                          ss.duty_buck_current * (1.0f - ss.pid_blend_factor);
                if (ss.duty_buck_current < buck_target_blend) {
                    // V8.12: Yüksek Vin'de (>24V) ramp hızını %25 azalt
                    float buck_ramp_rate = SS_MAIN_RAMP_RATE;
                    if (filt_vin > 24.0f) {
                        buck_ramp_rate = SS_MAIN_RAMP_RATE * 0.75f;
                    }
                    ss.duty_buck_current += buck_ramp_rate;
                }
                
                // V8.12: VOUT TRACKING - OVP önleme
                if (filt_vout > pid_target_dynamic * 1.05f) {
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE * 3.0f;
                } else if (filt_vout > pid_target_dynamic) {
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE;
                }
                
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, ss.duty_buck_target);
                
            } else if (ss.operating_mode == MODE_BOOST) {
                // BOOST: Boost ramp → hedef (10ms)
                // V8.9 FIX: Anında MAX yerine rampa devam
                // Boost: PID blend ile soft hedef
                float boost_target_blend = ss.duty_boost_target * ss.pid_blend_factor + 
                                           ss.duty_boost_current * (1.0f - ss.pid_blend_factor);
                
                // V8.18 FIX: OVP Tracking - Boost'ta da Vout kontrolü ekle
                if (filt_vout > pid_target_dynamic) {
                    // OVP risk - duty AZALT (ters ramp)
                    ss.duty_boost_current -= SS_MAIN_RAMP_RATE;
                } else if (ss.duty_boost_current < boost_target_blend) {
                    ss.duty_boost_current += SS_MAIN_RAMP_RATE;
                }
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, ss.duty_boost_target);
                
            } else {
                // BUCK-BOOST: Her ikisi de blend
                // V8.18 FIX: OVP Tracking
                if (filt_vout > pid_target_dynamic) {
                    // OVP risk - Buck azalt, Boost azalt (Boost duty azaltmak = LO ON süresini artırmak = Output düşer mi? Hayır, Boost duty artınca Output artar. Boost duty AZALTILMALI)
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE;
                    ss.duty_boost_current -= SS_MAIN_RAMP_RATE;
                } else {
                    if (ss.duty_buck_current < ss.duty_buck_target) {
                        ss.duty_buck_current += SS_MAIN_RAMP_RATE;
                    }
                    if (ss.duty_boost_current < ss.duty_boost_target) {
                        ss.duty_boost_current += SS_MAIN_RAMP_RATE;
                    }
                }
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, ss.duty_buck_target);
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, ss.duty_boost_target);
            }
            
            // Akim koruma (thermal limiter)
            if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_absolute) {
                ss.duty_buck_current = DUTY_DISCHARGE;
            } else if (thermal_limiter.peak_warning || thermal_limiter.modulation_active) {
                ss.duty_buck_current -= thermal_limiter.duty_reduction_request;
            }
            
            // Clamp
            ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, DUTY_MAX);
            ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, DUTY_MAX);
            
            if (ss.phase_timer >= SS_BLEND_TIME) {
                ss.phase = SS_PHASE_STABILIZE;
                ss.phase_timer = 0.0f;
            }
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_STABILIZE: 3ms Stabilizasyon
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_STABILIZE:
            ss.iind_limit_active = SS_IIND_LIMIT_STABILIZE;
            slew_limit = SLEW_NORMAL;
            ss.pid_blend_factor = 1.0f;
            
            // Tam PID kontrolu
            error = pid_target_dynamic - filt_vout;
            error = Clamp_Float(error, -ERROR_CLAMP_NORMAL, ERROR_CLAMP_NORMAL);
            
            error_derivative = (error - prev_error) / TS;
            prev_prev_error = prev_error;
            prev_error = error;
            
            // Integrator (thermal limiter check)
            if (!thermal_limiter.thermal_clamp_active) {
                integrator += KI_BASE * error * TS;
            }
            integrator = Clamp_Float(integrator, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS);
            
            // PID output
            float pid_output = KP_BASE * error + integrator + KD_BASE * error_derivative;
            (void)pid_output;  // Suppress unused warning - available for future use
            
            // Feedforward
            float feedforward_duty_stabilize = pid_target_dynamic / filt_vin;
            feedforward_duty_stabilize = Clamp_Float(feedforward_duty_stabilize, DUTY_MIN, DUTY_MAX);
            (void)feedforward_duty_stabilize;  // Suppress unused warning
            
            // V8.12: MODE-AWARE STABILIZE - Ramp devam etmeli
            if (ss.operating_mode == MODE_BUCK) {
                // BUCK: Buck ramp → hedef (10ms), Boost = DUTY_MAX
                // V8.12: Boost modunun aynası, roller ters
                ss.duty_boost_current = DUTY_MAX;  // Boost LO ON (pasif)
                
                // Buck: Yavaş rampa hedefe
                if (ss.duty_buck_current < ss.duty_buck_target) {
                    // V8.12: Yüksek Vin'de (>24V) ramp hızını %25 azalt
                    float buck_ramp_rate = SS_MAIN_RAMP_RATE;
                    if (filt_vin > 24.0f) {
                        buck_ramp_rate = SS_MAIN_RAMP_RATE * 0.75f;
                    }
                    ss.duty_buck_current += buck_ramp_rate;
                }
                
                // V8.12: VOUT TRACKING - OVP önleme
                if (filt_vout > pid_target_dynamic * 1.05f) {
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE * 3.0f;
                } else if (filt_vout > pid_target_dynamic) {
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE;
                }
                
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, ss.duty_buck_target);
                
            } else if (ss.operating_mode == MODE_BOOST) {
                // BOOST: Boost ramp → hedef (10ms)
                // Boost: Yavaş rampa hedefe
                
                // V8.18 FIX: OVP Tracking
                if (filt_vout > pid_target_dynamic) {
                    ss.duty_boost_current -= SS_MAIN_RAMP_RATE;
                } else if (ss.duty_boost_current < ss.duty_boost_target) {
                    ss.duty_boost_current += SS_MAIN_RAMP_RATE;
                }
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, ss.duty_boost_target);
                
            } else {
                // BUCK-BOOST: Her ikisi de soft kontrol
                // V8.18 FIX: OVP Tracking
                if (filt_vout > pid_target_dynamic) {
                    ss.duty_buck_current -= SS_MAIN_RAMP_RATE;
                    ss.duty_boost_current -= SS_MAIN_RAMP_RATE;
                } else {
                    if (ss.duty_buck_current < ss.duty_buck_target) {
                        ss.duty_buck_current += SS_MAIN_RAMP_RATE;
                    }
                    if (ss.duty_boost_current < ss.duty_boost_target) {
                        ss.duty_boost_current += SS_MAIN_RAMP_RATE;
                    }
                }
                ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, ss.duty_buck_target);
                ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, ss.duty_boost_target);
            }
            
            // Akim koruma (thermal limiter)
            if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_warning) {
                ss.duty_buck_current -= thermal_limiter.duty_reduction_request;
            }
            
            // Clamp
            ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, DUTY_MAX);
            ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, DUTY_MAX);
            
            // Stabilizasyon kontrolu
            float vout_error = Abs_Float(filt_vout - pid_target_dynamic);
            ss.vout_settle = ss.vout_settle * 0.99f + vout_error * 0.01f;
            
            if (ss.phase_timer >= SS_STABILIZE_TIME) {
                // Basarili mi?
                if (vout_error < 1.0f && !fault.fault_active) {
                    ss.success = 1;
                    ss.final_error = vout_error;
                }
                
                ss.phase = SS_PHASE_COMPLETE;
                ss.phase_timer = 0.0f;
            }
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_COMPLETE: RUN'a gec
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_COMPLETE:
            ss.complete = 1;
            diag.softstart_completions++;
            
            // ═══════════════════════════════════════════════════════════════════════
            // V8.19: BUMPLESS TRANSFER - Integratörü hedefe yakınken akıllıca sıfırla
            // Problem: Soft-start'ta biriken integratör, RUN modunda Vout overshoot yapıyor
            // Çözüm: Vout hedefteyse integratör=0, değilse mevcut duty'yi koruyacak değer
            // ═══════════════════════════════════════════════════════════════════════
            {
                float current_error_complete = pid_target_dynamic - filt_vout;
                
                if (Abs_Float(current_error_complete) < 0.5f) {
                    // Vout hedefteyse: İntegratörü sıfırla - temiz başlangıç
                    integrator = 0.0f;
                } else {
                    // Hata varsa: Mevcut duty'yi koruyacak integratör değeri hesapla
                    // integrator = duty - feedforward - Kp*error
                    float ff_complete = pid_target_dynamic / filt_vin;
                    ff_complete = Clamp_Float(ff_complete, DUTY_MIN, DUTY_MAX);
                    float target_kp = (ss.operating_mode == MODE_BUCK) ? KP_BUCK :
                                     (ss.operating_mode == MODE_BOOST) ? KP_BOOST : KP_BUCKBOOST;
                    integrator = ss.duty_buck_current - ff_complete - (target_kp * current_error_complete);
                    integrator = Clamp_Float(integrator, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS * 0.5f);
                }
                
                // Hata değişkenlerini sıfırla - temiz başlangıç
                prev_error = 0.0f;
                prev_prev_error = 0.0f;
            }
            
            // V8.12: Moda gore state gec - MODE_BUCK restored
            if (ss.operating_mode == MODE_BUCK) {
                sys_state = STATE_RUN_BUCK;
                kp = KP_BUCK;
                ki = KI_BUCK;
                kd = KD_BUCK;
            } else if (ss.operating_mode == MODE_BOOST) {
                sys_state = STATE_RUN_BOOST;
                kp = KP_BOOST;
                ki = KI_BOOST;
                kd = KD_BOOST;
            } else {
                // V8.17: Buck-Boost mode
                sys_state = STATE_RUN_BUCKBOOST;
                kp = KP_BUCKBOOST;
                ki = KI_BUCKBOOST;
                kd = KD_BUCKBOOST;
            }
            
            state_timer = 0.0f;
            break;
            
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        // SS_PHASE_FAULT
        // ════════════════════════════════════════════════════════════════════════════════════════════════════════════
        case SS_PHASE_FAULT:
            ss.duty_buck_current = DUTY_MIN;
            ss.duty_boost_current = DUTY_MAX;
            diag.softstart_faults++;
            
            sys_state = STATE_FAULT;
            state_timer = 0.0f;
            break;
            
        default:
            ss.phase = SS_PHASE_INIT;
            break;
    }
    
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    // SOFT-START FAULT CHECK
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    
    if (fault.fault_active) {
        ss.phase = SS_PHASE_FAULT;
    }
    
    // Timeout
    if (ss.total_timer > SS_MAX_TIME) {
        if (filt_vout < pid_target_dynamic * 0.5f) {
            fault.fault_active = 1;
            fault.fault_type = FAULT_CONTROL;
            ss.phase = SS_PHASE_FAULT;
        } else {
            ss.phase = SS_PHASE_COMPLETE;
        }
    }
}
// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 9: MAIN STATE MACHINE                                                        ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝

prev_state = sys_state;

// Feedforward hesapla (tum RUN states icin)
feedforward_buck = pid_target_dynamic / filt_vin;

// AUXILIARY DUTY ELEMENTS DESTROYED AS PER USER ORDER
// feedforward_buck += boot_compensation; // DESTROYED
// feedforward_buck += instant_duty_correction; // DESTROYED
// V7.7 & V7.8 LOGIC DESTROYED
feedforward_buck = Clamp_Float(feedforward_buck, duty_min_dynamic, duty_max_dynamic);

feedforward_boost = 1.0f - (filt_vin / pid_target_dynamic);
// Boost tarafında da kompanzasyon (ters yönde)
feedforward_boost = Clamp_Float(feedforward_boost, 0.0f, 1.0f - duty_min_dynamic);

// PREDICTIVE FEEDFORWARD ENHANCEMENT DESTROYED
// if (pred.buffer_filled && pred.feedforward_boost_active && pred.vin_slew_class >= 2) { ... }

// Effective PID gains (scenario engine'den)
float damping_for_pid = behavior.damping_factor;
float effective_kp = scenario.active_kp * damping_for_pid;
float effective_ki = scenario.active_ki * damping_for_pid;
float effective_kd = scenario.active_kd;

// ════════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V7.1: ADAPTIVE PID FROM PREDICTION ENGINE
// Osilasyon tespit edildiginde veya stabil durumda PID kazanclarini otomatik ayarla
// GUVENLIK: Sadece buffer dolduktan sonra adaptif PID uygula
// ════════════════════════════════════════════════════════════════════════════════════════════════════════════════════
if (pred.buffer_filled && pred.pid_update_pending && sys_state >= STATE_RUN_BUCK) {
    // Osilasyon durumuna gore PID'i guncelle
    if (pred.oscillation_type >= OSC_TYPE_SUSTAINED) {
        // Osilasyon tespit edildi - yumusatici PID
        effective_kp = pred.recommended_kp * behavior.damping_factor;
        effective_ki = pred.recommended_ki * behavior.damping_factor;
        effective_kd = pred.recommended_kd;
    } else if (pred.oscillation_type == OSC_TYPE_NONE) {
        // Stabil sistem - daha agresif PID (opsiyonel)
        effective_kp = pred.recommended_kp * behavior.damping_factor;
        effective_ki = pred.recommended_ki * behavior.damping_factor;
        effective_kd = pred.recommended_kd;
    }
    
    // PID guncelleme istegini sifirla (bir kez uygula)
    pred.pid_update_pending = 0;
    pred.pid_adaptation_timer = 0.0f;
}

// High Vin attenuation (24V+ icin ekstra azaltma)
if (filt_vin > HIGH_VIN_THRESHOLD) {
    float vin_ratio = (filt_vin - HIGH_VIN_THRESHOLD) / (VIN_MAX - HIGH_VIN_THRESHOLD);
    vin_ratio = Clamp_Float(vin_ratio, 0.0f, 1.0f);
    
    effective_kp *= (1.0f - vin_ratio * (1.0f - HIGH_VIN_KP_FACTOR));
    effective_ki *= (1.0f - vin_ratio * (1.0f - HIGH_VIN_KI_FACTOR));
}

// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V8.9: RECOVERY GAIN BOOST - Yükten kopunca PID 7x daha agresif
// Hedef: 35ms yerine 3-4ms'de yüke geri bağlanma
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
if (load_state == LOAD_STATE_YUKSUZ) {
    // Yükten kopmuş - hızlı recovery için PID kazançlarını artır
    effective_kp *= RECOVERY_GAIN_MULTIPLIER;
    effective_ki *= RECOVERY_GAIN_MULTIPLIER;
    effective_kd *= (RECOVERY_GAIN_MULTIPLIER * 0.5f);  // D için daha az (stabilite)
}

// ════════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V7.1: MODEL OBSERVER DUTY CORRECTION - DESTROYED (feedforward_buck removed)
// Model observer correction logic has been removed along with Buck mode
// ════════════════════════════════════════════════════════════════════════════════════════════════════════════════════

switch (sys_state) {
    
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    // STATE_INIT
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    case STATE_INIT:
        // V8.9: FREEWHEELING başlangıç - mod algılanana kadar güvenli mod
        // Buck HO OFF (LO ON), Boost HO OFF (LO ON) → Induktör freewheeling
        ss.duty_buck_current = DUTY_MIN;   // Buck HO OFF → LO ON
        ss.duty_boost_current = DUTY_MAX;  // V8.9 FIX: Boost LO ON → HO OFF (FREEWHEELING)
        prev_timer_a = DUTY_MIN;
        prev_timer_b = DUTY_MAX;           // V8.9 FIX: Boost LO ON
        timer_a_duty = DUTY_MIN;           // Output'a da yaz
        timer_b_duty = DUTY_MAX;           // Output'a da yaz
        slew_limit = SLEW_PRECHARGE;
        
        // Reset all faults
        fault.fault_active = 0;
        fault.fault_type = FAULT_NONE;
        fault.ovp_trip = 0;
        fault.ocp_trip = 0;
        fault.uvlo_trip = 0;
        fault.thermal_trip = 0;
        fault.short_trip = 0;
        fault.iin_ocp_trip = 0;
        fault.shootthrough_trip = 0;
        fault.dcm_trip = 0;
        
        peak_tracker.instant_fault = 0;
        peak_tracker.peak_fault = 0;
        peak_tracker.peak_timer = 0.0f;
        
        input_ocp.ocp_proven = 0;
        input_ocp.shootthrough_proven = 0;
        input_ocp.debounce_counter = 0;
        
        thermal_int.trip_proven = 0;
        thermal_int.energy_accumulated = 0.0f;
        
        // Reset soft-start
        ss.phase = SS_PHASE_INIT;
        ss.phase_timer = 0.0f;
        ss.total_timer = 0.0f;
        ss.vin_sample_count = 0;
        ss.complete = 0;
        ss.pid_blend_factor = 0.0f;
        
        // Reset behavior
        behavior.oscillation_detected = 0;
        behavior.damping_factor = 1.0f;
        behavior.sign_changes = 0;
        
        // Reset scenario
        scenario.current_scenario = SCENARIO_NORMAL;
        scenario.scenario_blend = 1.0f;
        
        // PID reset
        integrator = 0.0f;
        prev_error = 0.0f;
        prev_prev_error = 0.0f;
        
        // History reset
        history.head = 0;
        history.count = 0;
        
        // UVLO check
        if (filt_vin >= VIN_MIN) {
            sys_state = STATE_SOFTSTART;
            state_timer = 0.0f;
        }
        break;
        
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    // STATE_SOFTSTART (Module 8'de islendi)
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    case STATE_SOFTSTART:
        // Soft-start logic Module 8'de
        break;
        
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    // STATE_RUN_BUCK - V8.12: Boost modunun aynası, roller ters
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    case STATE_RUN_BUCK:
        slew_limit = scenario.active_slew;
        
        // Error - V8.12: pid_target_dynamic kullan (yüksüz=14V, yüklü=12V)
        error = pid_target_dynamic - filt_vout;
        error = Clamp_Float(error, -ERROR_CLAMP_NORMAL, ERROR_CLAMP_NORMAL);
        
        // Derivative
        float raw_deriv_buck = (error - prev_error) / TS;
        error_derivative = error_derivative * 0.8f + raw_deriv_buck * 0.2f;
        prev_prev_error = prev_error;
        prev_error = error;
        
        // Integrator (V8.12: Deadband içinde integrator durur - Boost ile aynı)
        if (!thermal_limiter.thermal_clamp_active) {
            // V8.12: Deadband içinde integrator birikimini durdur (üçgen osilasyon önleme)
            if (Abs_Float(error) >= ERROR_DEADBAND) {
                integrator += effective_ki * error * TS;
            }
            // Anti-windup
            if ((integrator > INTEGRATOR_LIMIT_POS && error > 0) ||
                (integrator < INTEGRATOR_LIMIT_NEG && error < 0)) {
                integrator = Clamp_Float(integrator, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS);
            }
        } else {
            integrator *= INTEGRATOR_DECAY_RATE;
        }
        integrator = Clamp_Float(integrator, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS);
        
        // PID output - V8.12: Boost ile aynı yapı
        float d_term_buck = effective_kd * error_derivative * 0.1f;  // 10x düşük
        float pid_out_buck = effective_kp * error + integrator + d_term_buck;
        
        // V8.12: Debug kopyaları (Buck modu için)
        debug_p_term = effective_kp * error;
        debug_i_term = integrator;
        debug_d_term = d_term_buck;

        
        // V8.12: ERROR DEADBAND - Küçük hatalarda PID pasif (duty osilasyonu önle)
        if (Abs_Float(error) < ERROR_DEADBAND) {
            pid_out_buck = 0.0f;  // Deadband içinde - duty değiştirme
        }
        
        // V8.12: Boost = DUTY_MAX (LO ON, pasif) - Boost modundaki Buck'un tersi
        ss.duty_boost_current = DUTY_MAX;
        
        // Buck: V8.12 PID kontrolü (Boost modundaki Boost duty'nin aynısı)
        // Feedforward + PID
        float buck_feedforward = pid_target_dynamic / filt_vin;
        buck_feedforward = Clamp_Float(buck_feedforward, DUTY_MIN, DUTY_MAX);
        float buck_target = buck_feedforward + pid_out_buck;  // V8.16: PID tam etki (0.01f → 1.0)
        
        // DCM prevention
        if (dcm_prev.prevention_active) {
            buck_target += dcm_prev.duty_boost_request;
        }
        
        // Thermal limiting
        if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_warning) {
            buck_target -= thermal_limiter.duty_reduction_request * 0.5f;
        }
        
        // ═══════════════════════════════════════════════════════════════════════
        // V8.19: OVP FEEDBACK LOOP - Vout > hedef iken integrator sınırla
        // Problem: Pozitif integratör birikimi Vout'u hedefe rağmen yükseltiyor
        // Çözüm: Overshoot durumunda integratörün pozitif katkısını engelle
        // ═══════════════════════════════════════════════════════════════════════
        if (filt_vout > pid_target_dynamic + 0.1f) {
            // Overshoot tespit - pozitif integratör yok
            integrator = Min_Float(integrator, 0.0f);
            // Duty'yi azalt
            buck_target -= 0.01f;
        }
        
        // V8.12: PID çıkışı direkt uygulanır
        ss.duty_buck_current = buck_target;
        
        // Clamp
        ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, DUTY_MAX);
        ss.duty_boost_current = DUTY_MAX;  // Buck modunda Boost her zaman DUTY_MAX
        
        // Fault check
        if (fault.fault_active) {
            sys_state = STATE_FAULT;
            state_timer = 0.0f;
            fault.fault_count++;
        }
        
        // Mode transition: Vin düşerse Buck-Boost'a geç
        if (filt_vin < VIN_BUCK_EXIT) {
            sys_state = STATE_RUN_BUCKBOOST;
            ss.operating_mode = MODE_BUCKBOOST;
            integrator = 0.0f;
        } else {
            ss.operating_mode = MODE_BUCK;
        }
        break;
        
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    // STATE_RUN_BOOST
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    case STATE_RUN_BOOST:
        slew_limit = scenario.active_slew;
        
        // Error - V8.9: pid_target_dynamic kullan (yüksüz=14V, yüklü=12V)
        error = pid_target_dynamic - filt_vout;
        error = Clamp_Float(error, -ERROR_CLAMP_NORMAL, ERROR_CLAMP_NORMAL);
        
        // Derivative
        float raw_deriv_boost = (error - prev_error) / TS;
        error_derivative = error_derivative * 0.8f + raw_deriv_boost * 0.2f;
        prev_prev_error = prev_error;
        prev_error = error;
        
        // Integrator (V8.9: Deadband içinde integrator durur)
        if (!thermal_limiter.thermal_clamp_active) {
            // V8.9: Deadband içinde integrator birikimini durdur (üçgen osilasyon önleme)
            if (Abs_Float(error) >= ERROR_DEADBAND) {
                integrator += effective_ki * error * TS;
            }
            // Anti-windup
            if ((integrator > INTEGRATOR_LIMIT_POS && error > 0) ||
                (integrator < INTEGRATOR_LIMIT_NEG && error < 0)) {
                integrator = Clamp_Float(integrator, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS);
            }
        } else {
            integrator *= INTEGRATOR_DECAY_RATE;
        }
        integrator = Clamp_Float(integrator, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS);
        
        // PID output - V8.10: D-term düşük agresiflikle geri eklendi (0.1x)
        float d_term_boost = effective_kd * error_derivative * 0.1f;  // 10x düşük
        float pid_out_boost = effective_kp * error + integrator + d_term_boost;
        
        // V8.10: Debug kopyaları (Boost modu için)
        debug_p_term = effective_kp * error;
        debug_i_term = integrator;
        debug_d_term = d_term_boost;

        
        // V8.9: ERROR DEADBAND - Küçük hatalarda PID pasif (duty osilasyonu önle)
        if (Abs_Float(error) < ERROR_DEADBAND) {
            pid_out_boost = 0.0f;  // Deadband içinde - duty değiştirme
        }
        
        // V8.9 FIX: Buck ve Boost soft-start değerlerini koru, PID küçük ayarlama yapsın
        // V8.18 FIX: Double Integrator REMOVED. Use Feedforward + PID (Position Form)
        // Buck: Always Max in Boost Mode
        ss.duty_buck_current = DUTY_MAX;
        
        // Boost: Feedforward (1 - Vin/Vout) + PID
        float boost_ff = 1.0f - (filt_vin / pid_target_dynamic);
        boost_ff = Clamp_Float(boost_ff, 0.0f, 0.6f);
        
        // Target = FF + PID
        float boost_target = boost_ff + pid_out_boost;
        
        // DCM prevention
        if (dcm_prev.prevention_active) {
            boost_target += dcm_prev.duty_boost_request;
        }
        
        // Thermal limiting
        if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_warning) {
            boost_target -= thermal_limiter.duty_reduction_request * 0.5f;
        }

        // V8.18 FIX: HARD OVP OVERRIDE
        // Eğer Vout, hedefin 0.5V üzerindeyse (örn. 12.5V > 12.0V), Boost'u zorla KAPAT.
        // Bu, PID veya Feedforward hatalarından bağımsız kesin koruma sağlar.
        if (filt_vout > pid_target_dynamic + 0.5f) {
            boost_target = 0.0f;
        }
        
        // Apply directly
        ss.duty_boost_current = boost_target;
        
        // Clamp
        ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, DUTY_MIN, DUTY_MAX);
        ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, 0.5f);  // Max %50
        
        // Fault check
        if (fault.fault_active) {
            sys_state = STATE_FAULT;
            state_timer = 0.0f;
            fault.fault_count++;
        }
        
        // Mode transition: Vin yükselirse Buck-Boost'a geç
        if (filt_vin > VIN_BOOST_EXIT) {
            sys_state = STATE_RUN_BUCKBOOST;
            ss.operating_mode = MODE_BUCKBOOST;
            integrator = 0.0f;
        } else {
            ss.operating_mode = MODE_BOOST;
        }
        // V8.9: timer_b_duty zaten doğru değerde (soft-start'tan), üzerine yazma!
        break;
        
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    // STATE_RUN_BUCKBOOST - V8.17: Buck ve Boost modlarının kombinasyonu
    // Her iki bacak da aktif kontrol altında (11V < Vin < 13V aralığı)
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    case STATE_RUN_BUCKBOOST:
        slew_limit = scenario.active_slew;
        
        // Error - pid_target_dynamic kullan
        error = pid_target_dynamic - filt_vout;
        error = Clamp_Float(error, -ERROR_CLAMP_NORMAL, ERROR_CLAMP_NORMAL);
        
        // Derivative
        float raw_deriv_bb = (error - prev_error) / TS;
        error_derivative = error_derivative * 0.8f + raw_deriv_bb * 0.2f;
        prev_prev_error = prev_error;
        prev_error = error;
        
        // Integrator (Deadband içinde durur)
        if (!thermal_limiter.thermal_clamp_active) {
            if (Abs_Float(error) >= ERROR_DEADBAND) {
                integrator += effective_ki * error * TS;
            }
            // Anti-windup
            if ((integrator > INTEGRATOR_LIMIT_POS && error > 0) ||
                (integrator < INTEGRATOR_LIMIT_NEG && error < 0)) {
                integrator = Clamp_Float(integrator, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS);
            }
        } else {
            integrator *= INTEGRATOR_DECAY_RATE;
        }
        integrator = Clamp_Float(integrator, INTEGRATOR_LIMIT_NEG, INTEGRATOR_LIMIT_POS);
        
        // PID output
        float d_term_bb = effective_kd * error_derivative * 0.1f;  // 10x düşük
        float pid_out_bb = effective_kp * error + integrator + d_term_bb;
        
        // Debug kopyaları
        debug_p_term = effective_kp * error;
        debug_i_term = integrator;
        debug_d_term = d_term_bb;
        
        // ERROR DEADBAND
        if (Abs_Float(error) < ERROR_DEADBAND) {
            pid_out_bb = 0.0f;
        }
        
        // ═══════════════════════════════════════════════════════════════════════════
        // BUCK-BOOST SPESİFİK DUTY HESAPLAMA
        // Her iki bacak da aktif kontrol altında
        // ═══════════════════════════════════════════════════════════════════════════
        
        // Buck duty: Feedforward + PID
        float buck_ff = pid_target_dynamic / filt_vin;
        buck_ff = Clamp_Float(buck_ff, DUTY_MIN, DUTY_MAX);
        float buck_target_bb = buck_ff + pid_out_bb;
        
        // Boost duty: 1 - (Vin/Vout) formülü + PID katkısı
        float boost_ratio = 1.0f - (filt_vin / pid_target_dynamic);
        boost_ratio = Clamp_Float(boost_ratio, 0.0f, 0.5f);
        float boost_target_bb = boost_ratio + pid_out_bb * 0.5f;  // PID'in yarısı boost'a
        
        // DCM prevention
        if (dcm_prev.prevention_active) {
            buck_target_bb += dcm_prev.duty_boost_request * 0.5f;
            boost_target_bb += dcm_prev.duty_boost_request * 0.5f;
        }
        
        // Thermal limiting
        if (thermal_limiter.thermal_clamp_active || thermal_limiter.peak_warning) {
            buck_target_bb -= thermal_limiter.duty_reduction_request * 0.3f;
            boost_target_bb -= thermal_limiter.duty_reduction_request * 0.3f;
        }

        // V8.18 FIX: HARD OVP OVERRIDE
        if (filt_vout > pid_target_dynamic + 0.5f) {
            boost_target_bb = 0.0f; 
        }
        
        // Duty uygula
        ss.duty_buck_current = buck_target_bb;
        ss.duty_boost_current = boost_target_bb;
        
        // Clamp
        ss.duty_buck_current = Clamp_Float(ss.duty_buck_current, 0.3f, DUTY_MAX);  // Min %30
        ss.duty_boost_current = Clamp_Float(ss.duty_boost_current, DUTY_MIN, 0.5f);  // Max %50
        
        // Fault check
        if (fault.fault_active) {
            sys_state = STATE_FAULT;
            state_timer = 0.0f;
            fault.fault_count++;
        }
        
        // Mode transitions
        if (filt_vin > VIN_BUCK_ENTER) {
            sys_state = STATE_RUN_BUCK;
            ss.operating_mode = MODE_BUCK;
            ss.duty_boost_current = DUTY_MAX;  // Boost pasif
            integrator = 0.0f;
        } else if (filt_vin < VIN_BOOST_ENTER) {
            sys_state = STATE_RUN_BOOST;
            ss.operating_mode = MODE_BOOST;
            ss.duty_buck_current = DUTY_MAX;   // Buck pasif
            integrator = 0.0f;
        } else {
            ss.operating_mode = MODE_BUCKBOOST;
        }
        break;
        

        
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    // STATE_FAULT
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    case STATE_FAULT:
        // Guvenli cikislar
        ss.duty_buck_current = DUTY_MIN;
        ss.duty_boost_current = DUTY_MIN;  /* TITAN: Boost LO OFF (Fault safe) */
        slew_limit = SLEW_FAULT_RECOVERY;
        
        // Integrator reset
        integrator = 0.0f;
        fault.fault_latched = 1;
        
        // Behavior reset
        behavior.oscillation_detected = 0;
        behavior.damping_factor = 1.0f;
        
        // Recovery timer
        fault.recovery_timer += TS;
        
        // ═══════════════════════════════════════════════════════════════════════════════════════════════════════════
        // V8.19: AUTO-RESTART - 2ms sonra Vout ve I_L sıfırsa otomatik soft-start
        // Problem: Fault durumunda sistem kalıcı olarak kapanıyordu
        // Çözüm: Devre soğudu (Vout≈0, I_L≈0) ve Vin OK ise yeniden başlat
        // ═══════════════════════════════════════════════════════════════════════════════════════════════════════════
        if (fault.recovery_timer > 0.002f) {  // V8.19: 2s → 2ms (hızlı restart)
            // Restart koşulları:
            // 1. Shootthrough değilse (donanım hasarı riski)
            // 2. Anlık fault değilse
            // 3. Vin geçerli aralıkta
            // 4. Vout sıfıra yakın (devre boşalmış)
            // 5. Inductor akımı sıfıra yakın (enerji kalmamış)
            if (!fault.shootthrough_trip && 
                !peak_tracker.instant_fault &&
                filt_vin > VIN_MIN &&
                filt_vout < 1.0f &&   // V8.19: Vout ≈ 0 (1V altı)
                filt_iind < 0.5f) {   // V8.19: I_L ≈ 0 (0.5A altı)
                
                // Tüm fault flag'lerini temizle
                fault.fault_active = 0;
                fault.fault_type = FAULT_NONE;
                fault.fault_latched = 0;
                fault.recovery_timer = 0.0f;
                fault.ovp_trip = 0;
                fault.ocp_trip = 0;
                fault.uvlo_trip = 0;
                fault.thermal_trip = 0;
                fault.short_trip = 0;
                fault.iin_ocp_trip = 0;
                fault.dcm_trip = 0;
                
                // Debounce counter'ları sıfırla
                ovp_debounce = 0;
                ocp_debounce = 0;
                input_ocp.debounce_counter = 0;
                input_ocp.ocp_proven = 0;
                thermal_int.trip_proven = 0;
                thermal_int.energy_accumulated = 0.0f;
                
                // PID sıfırla
                integrator = 0.0f;
                prev_error = 0.0f;
                prev_prev_error = 0.0f;
                
                // Soft-start'a geri dön
                sys_state = STATE_INIT;
                state_timer = 0.0f;
                
                // diag.auto_restarts++;  // V8.19: Debug restart sayacı (diag struct'a eklenirse aktif et)
            }
        }
        break;
        
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    // STATE_RECOVERY
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    case STATE_RECOVERY:
        sys_state = STATE_INIT;
        break;
        
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    // DEFAULT
    // ════════════════════════════════════════════════════════════════════════════════════════════════════════════════
    default:
        sys_state = STATE_INIT;
        break;
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 10: SLEW RATE LIMITING & OUTPUT                                              ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝

// Target duties
float target_buck = ss.duty_buck_current;
float target_boost = ss.duty_boost_current;

// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V8.9: OPPOSITE LEG FAST RAMP LOGIC
// Mod algılandıktan sonra karşı bacak (tam açık/kapalı olacak) için hızlı rampa
// BOOST mode: Buck → DUTY_MAX (hızlı)
// BUCK mode:  Boost → DUTY_MIN (hızlı)
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
#if OPPOSITE_LEG_FAST_RAMP_ENABLE

// Karşı bacak için kullanılacak slew limit'i belirle
float slew_limit_buck = slew_limit;   // Active leg: normal slew
float slew_limit_boost = slew_limit;  // Active leg: normal slew

// Soft-start veya run mode sırasında mod algılandıysa
if (ss.phase >= SS_PHASE_CHARACTERIZE || sys_state >= STATE_RUN_BUCK) {
    
    #if OPPOSITE_LEG_INSTANT_MODE
    // INSTANT MODE: Karşı bacak anında hedef değere geçer (slew bypass)
    if (ss.operating_mode == MODE_BOOST) {
        // BOOST: Buck = DUTY_MAX anında
        target_buck = DUTY_MAX;
        prev_timer_a = DUTY_MAX;  // Slew bypass
    } else if (ss.operating_mode == MODE_BUCK) {
        // BUCK: Boost = DUTY_MIN anında
        target_boost = DUTY_MIN;
        prev_timer_b = DUTY_MIN;  // Slew bypass
    }
    // BUCK-BOOST: Her iki bacak da aktif, instant mode yok
    
    #else
    // FAST RAMP MODE: Karşı bacak hızlı rampalar (10x)
    if (ss.operating_mode == MODE_BOOST) {
        // BOOST: Buck için hızlı slew kullan
        slew_limit_buck = OPPOSITE_LEG_FAST_SLEW;
    } else if (ss.operating_mode == MODE_BUCK) {
        // BUCK: Boost için hızlı slew kullan
        slew_limit_boost = OPPOSITE_LEG_FAST_SLEW;
    }
    #endif
}

// Slew rate limiting (mode-aware)
float delta_buck = target_buck - prev_timer_a;
float delta_boost = target_boost - prev_timer_b;

// Apply slew limit (karşı bacak için hızlı)
if (delta_buck > slew_limit_buck) delta_buck = slew_limit_buck;
if (delta_buck < -slew_limit_buck) delta_buck = -slew_limit_buck;
if (delta_boost > slew_limit_boost) delta_boost = slew_limit_boost;
if (delta_boost < -slew_limit_boost) delta_boost = -slew_limit_boost;

#else
// V8.8 ve öncesi: Normal slew rate limiting (her iki bacak aynı)
float delta_buck = target_buck - prev_timer_a;
float delta_boost = target_boost - prev_timer_b;

// Apply slew limit
if (delta_buck > slew_limit) delta_buck = slew_limit;
if (delta_buck < -slew_limit) delta_buck = -slew_limit;
if (delta_boost > slew_limit) delta_boost = slew_limit;
if (delta_boost < -slew_limit) delta_boost = -slew_limit;
#endif

// Update duties
timer_a_duty = prev_timer_a + delta_buck;
timer_b_duty = prev_timer_b + delta_boost;

// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V8.9: AGGRESSIVE OVERCURRENT PROTECTION (10A threshold, 5x duty reduction)
// Akım 10A'yı aşarsa duty 5x daha hızlı azaltılır - 225µs freewheel'den önce düşürülür
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
{
    const float OCP_THRESHOLD = 10.0f;      // Akım eşiği [A]
    const float OCP_DUTY_REDUCTION = 0.025f; // Normal slew'in 5 katı
    
    float abs_iind = Abs_Float(raw_iind);
    float abs_iin = Abs_Float(raw_iin);
    float max_current = (abs_iind > abs_iin) ? abs_iind : abs_iin;
    
    // V8.16: Load connect boost mode sırasında bu OCP'yi ATLAT
    if (max_current > OCP_THRESHOLD && !load_connect_boost_active) {
        // Akım aşırı yüksek - duty'yi agresif azalt
        float excess = max_current - OCP_THRESHOLD;  // Aşım miktarı
        float reduction = OCP_DUTY_REDUCTION * (1.0f + excess * 0.5f);  // Aşım arttıkça daha agresif
        
        // Buck modunda: buck duty azalt
        if (ss.operating_mode == MODE_BUCK || ss.operating_mode == MODE_BUCKBOOST) {
            timer_a_duty -= reduction;
        }
        // Boost modunda: boost duty azalt (LO OFF → akım düşer)
        if (ss.operating_mode == MODE_BOOST || ss.operating_mode == MODE_BUCKBOOST) {
            timer_b_duty -= reduction * 0.5f;  // Boost için daha az (LO ON tutma)
        }
    }
}

// Final clamp
timer_a_duty = Clamp_Float(timer_a_duty, DUTY_MIN, DUTY_MAX);
timer_b_duty = Clamp_Float(timer_b_duty, DUTY_MIN, DUTY_MAX);

// Store for next cycle
prev_timer_a = timer_a_duty;
prev_timer_b = timer_b_duty;

// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V8.9: FINAL MODE-AWARE OVERRIDE - Soft-Start sırasında sabit tarafı zorla
// Bu blok, tüm önceki atamaları override eder ve beklenen davranışı garanti eder
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
if (sys_state == STATE_SOFTSTART) {
    
    // V8.9: SS_PHASE_INIT ve SS_PHASE_MEASURE sırasında FREEWHEELING MODU
    // Mod algılanmadan önce güvenli mod: Buck HO OFF (LO ON), Boost HO OFF (LO ON)
    if (ss.phase <= SS_PHASE_MEASURE) {
        // FREEWHEELING: Her iki LO ON, HO'lar OFF
        timer_a_duty = DUTY_MIN;   // Buck HO OFF → LO ON
        timer_b_duty = DUTY_MAX;   // Boost LO ON → HO OFF
        prev_timer_a = DUTY_MIN;
        prev_timer_b = DUTY_MAX;
    }
    // V8.9: Mod algılandıktan sonra (SS_PHASE_CHARACTERIZE ve sonrası)
    else if (ss.operating_mode == MODE_BOOST) {
        // BOOST: Buck (timer_a) = soft-start 10x hızlı, Boost (timer_b) = soft-start yavaş
        timer_a_duty = ss.duty_buck_current;   // V8.9: Rampa değeri (MAX'a doğru 10x hızlı)
        prev_timer_a = ss.duty_buck_current;
        timer_b_duty = ss.duty_boost_current;  // Soft-start (MIN'den yukarı rampalanır)
        prev_timer_b = ss.duty_boost_current;
    } else if (ss.operating_mode == MODE_BUCK) {
        // BUCK: Boost (timer_b) = MIN (LO OFF), Buck (timer_a) = PWM soft-start
        timer_b_duty = DUTY_MIN;   // TITAN: Boost LO OFF
        prev_timer_b = DUTY_MIN;   // TITAN: Boost LO OFF
        timer_a_duty = ss.duty_buck_current;  // Soft-start (MIN'den yukarı rampalanır)
        prev_timer_a = ss.duty_buck_current;
    }
    // BUCK-BOOST: Her ikisi de değişken, override yok
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                    MODULE 10.5: BOOTSTRAP REFRESH (LMG5200 Cbst Şarj)                                             ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝
// RATIONALE: HO (high-side) switch sürekli ON olduğunda Cbst boşalır.
//            1ms'de bir kısa pulse ile HO kapatılır → Cbst şarj edilir.
// 
// Refresh GEREKLİ:
//   - Buck duty > 0.97 → Buck HO sürekli ON → Her 1ms refresh
//   - Boost duty < 0.03 → Boost HO sürekli ON → Her 1ms refresh
//
// Refresh GEREKSİZ:
//   - Buck duty < 0.03 → Buck LO ON → Cbst zaten dolu
//   - Boost duty > 0.97 → Boost LO ON → Cbst zaten dolu
//   - Normal PWM çalışma → Doğal şarj döngüsü mevcut

#if BOOTSTRAP_REFRESH_ENABLE
{
    // Sayaçları artır
    bootstrap_buck_counter++;
    bootstrap_boost_counter++;
    
    // ─────────────────────────────────────────────────────────────────────────────────────
    // BUCK HO REFRESH: Sadece duty > 0.97 iken gerekli (HO sürekli ON)
    // V8.9 FIX: Cross-leg compensation kaldırıldı - endüstriyel standart değil
    // PID zaten küçük düşüşü telafi edecek
    // ─────────────────────────────────────────────────────────────────────────────────────
    if (timer_a_duty > 0.97f) {
        // Buck HO sürekli ON - Cbst boşalıyor, refresh gerekli
        if (bootstrap_buck_counter >= BOOTSTRAP_REFRESH_PERIOD) {
            bootstrap_buck_counter = 0;
            // 1 cycle için duty'yi 0.80'e düşür → HO kısa süre OFF → Cbst dolar
            timer_a_duty = 0.80f;
            bootstrap_refresh_active = 1;
            // NOT: Karşı taraf kompanzasyonu KALDIRILDI - %100 üzerine çıkma riski
        }
    } else {
        // Normal PWM veya LO ON - doğal şarj döngüsü mevcut
        bootstrap_buck_counter = 0;  // Sayacı sıfırla
    }
    
    // ─────────────────────────────────────────────────────────────────────────────────────
    // BOOST HO REFRESH: Sadece duty < 0.03 iken gerekli (HO sürekli ON via complementary)
    // V8.9 FIX: Cross-leg compensation kaldırıldı - PID telafi edecek
    // ─────────────────────────────────────────────────────────────────────────────────────
    if (timer_b_duty < 0.03f) {
        // Boost HO sürekli ON (LO=0 → HO=1) - Cbst boşalıyor, refresh gerekli
        if (bootstrap_boost_counter >= BOOTSTRAP_REFRESH_PERIOD) {
            bootstrap_boost_counter = 0;
            // 1 cycle için duty'yi 0.20'ye yükselt → HO kısa süre OFF → Cbst dolar
            timer_b_duty = 0.20f;
            // NOT: Karşı taraf kompanzasyonu KALDIRILDI - %100 üzerine çıkma riski
        }
    } else {
        // Normal PWM veya LO ON - doğal şarj döngüsü mevcut
        bootstrap_boost_counter = 0;  // Sayacı sıfırla
    }
    
    // Refresh flag - bir sonraki cycle'da sıfırla (debug için)
    if (bootstrap_refresh_active) {
        bootstrap_refresh_active = 0;
    }
}
#endif


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                    MODULE 10.6: HARDWARE FAULT OVERRIDE (PLECS Analog Koruma)                                    ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝
// RATIONALE: PLECS'teki SR Flip-flop analog koruma devresinden gelen sinyal.
//            hw_fault = 1 olduğunda tüm duty'ler freewheeling moduna zorlanır.
//            Bu, C-Script korumalarına ek olarak <50ns tepki süresi sağlar.

if (hw_fault > 0.5f) {
    // HARDWARE FAULT AKTIF - FREEWHEELING MODU
    // Buck HO (Q1) = OFF, Buck LO (Q2) = ON → timer_a = 0
    // Boost LO (Q3) = ON, Boost HO (Q4) = OFF → timer_b = 1
    timer_a_duty = 0.0f;   // Buck HO OFF
    timer_b_duty = 1.0f;   // Boost LO ON (HO OFF via complementary)
    
    // Slew rate bypass - anında uygula
    prev_timer_a = 0.0f;
    prev_timer_b = 1.0f;
    
    // Freewheeling flag'i de set et (diğer modüllerle senkronizasyon)
    freewheeling_active = 1;
}


// ╔══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╗
// ║                              MODULE 11: OUTPUT SIGNALS                                                           ║
// ╚══════════════════════════════════════════════════════════════════════════════════════════════════════════════════╝


// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V8.0: FREEWHEELING MODE - LMG5200 KORUMA
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// emergency_duty_override = -2.0f → FREEWHEELING MODE
// LMG5200 için kritik: HO anahtarları TAMAMEN OFF, LO anahtarları ON
// Bu sayede:
//   1. Vin artışında akım darbesi önlenir
//   2. Çıkış gerilimi yükselse de induktör akımı kontrollü kalır
//   3. Body diyotları/senkron rektifikasyon aktif

if (emergency_duty_override < -1.5f) {
    // FREEWHEELING MODE - KRİTİK GÜVENLİK
    // Buck tarafı: Q1 (HO) = OFF, Q2 (LO) = ON → duty = 0
    // Boost tarafı: Q4 (HO) = OFF, Q3 (LO) = ON → duty = 0
    // NOT: duty = 0 demek HO OFF, LO ON (PLECS modeline göre)
    timer_a_duty = 0.0f;   // Buck HO (Master) OFF
    timer_b_duty = 0.0f;   // Boost LO (Master) OFF
    
    // SLEW RATE BYPASS - ANINDA uygula (bekleme yok!)
    // Bir sonraki cycle'da slew rate limiting'in etkisini önle
    prev_timer_a = 0.0f;
    prev_timer_b = 0.0f;
}

OutputSignal(0, 0) = timer_a_duty;      // Buck HO (Master) PWM
OutputSignal(0, 1) = timer_b_duty;      // Boost LO (Master) PWM

// V7.15: ShutDown çıkışı (3. çıkış)
// POLARİTE: 0 = MOSFET OFF = Yük ayrık, 1 = MOSFET ON = Yük bağlı
// shutdown_active=1 → yük ayrık istiyoruz → Output=0
// shutdown_active=0 → yük bağlı istiyoruz → Output=1
#if SHUTDOWN_ENABLE
    OutputSignal(0, 2) = shutdown_active ? 0.0f : 1.0f;
#else
    OutputSignal(0, 2) = 1.0f;  // Her zaman bağlı
#endif

// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V8.3: DUAL-MODE DISCHARGE CIRCUIT (4. Çıkış) - LOAD-STATE AWARE
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// POLARİTE: 0 = MOSFET OFF = Direnç devre dışı, 1 = MOSFET ON = 5Ω direnç aktif (Vout → GND)
// 
// V8.3 DUAL-MODE:
//   YUKLU_DESARJ (load_state == YUKE_BAGLI):
//     - Aktivasyon: Vout > 13.0V
//     - Deaktivasyon: Vout <= 12.25V
//   
//   YUKSUZ_DESARJ (load_state != YUKE_BAGLI):
//     - Aktivasyon: Vout > 13.5V (V8.10: 14.5→13.5)
//     - Deaktivasyon: Vout <= 13.0V (V8.10: 14.0→13.0)
//
// NOT: discharge_active değişkeni ana state machine tarafından güncelleniyor (satır 815-834)
//      Bu bölüm sadece OutputSignal ataması yapıyor.
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
#if DISCHARGE_ENABLE
{
    // Discharge output signal - değer ana state machine'de hesaplandı
    // 1 = MOSFET ON = Discharge aktif (Vout → 5Ω → GND)
    // 0 = MOSFET OFF = Discharge kapalı
    OutputSignal(0, 3) = discharge_active ? 1.0f : 0.0f;
}
#else
    // Discharge devre dışı
    OutputSignal(0, 3) = 0.0f;
#endif

// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V8.2: FREEWHEEL/BRAKE CIRCUIT (5th Output) - INDUCTOR ENERGY DISSIPATION
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// POLARİTE: 0 = MOSFET OFF = Resistor disabled, 1 = MOSFET ON = 5Ω resistor active (Inductor → GND)
// 
// ACTIVATION CONDITIONS (ALL must be true):
//   1. Overvoltage/Load dump detected (freewheeling_active = 1 OR Vout > 13.5V)
//   2. All PWMs are OFF (timer_a_duty ≈ 0 AND timer_b_duty ≈ 0)
//   3. Inductor current above threshold (raw_iind > 15A)
//   4. Load is disconnected (shutdown_active = 1)
//
// DEACTIVATION CONDITIONS:
//   1. Inductor current drops to safe level (< 5A)
//   2. Minimum activation time elapsed (50µs)
//
// SAFETY INTERLOCKS (CRITICAL):
//   - Brake NEVER active when any PWM is ON (shoot-through prevention)
//   - Brake NEVER active when load is connected (safety)
//   - Brake current limited by 5Ω resistor (Imax ≈ 2.7A @ 13.5V)
//   - Complements existing freewheeling mode (works together)
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
#if FREEWHEEL_BRAKE_ENABLE
{
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // SECTION 1: SAFETY INTERLOCKS
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    // Interlock 1: ALL PWMs must be OFF (< 1% duty)
    int pwms_off = (timer_a_duty < 0.01f) && (timer_b_duty < 0.01f);
    
    // Interlock 2: Overvoltage event active
    int overvoltage_event = (freewheeling_active || raw_vout > 13.5f);
    
    // Interlock 3: High inductor current requires dissipation
    int high_inductor_current = (raw_iind > FREEWHEEL_BRAKE_IIND_THRESHOLD);
    
    // Interlock 4: Load must be disconnected
    int load_disconnected = shutdown_active;
    
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // SECTION 2: ACTIVATION LOGIC
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    
    if (load_disconnected && pwms_off && overvoltage_event && high_inductor_current) {
        // All conditions met - activate brake
        if (!freewheel_brake_active) {
            freewheel_brake_active = 1;
            freewheel_brake_timer = 0.0f;
        }
    }
    // ═══════════════════════════════════════════════════════════════════════════════════════
    // V8.4 FIX: freewheeling_active flag'i de freewheel brake'i aktive etmeli!
    // Vin slew protection, bobin akım limiti vs. bu flag'i set ediyor
    // ═══════════════════════════════════════════════════════════════════════════════════════
    else if (freewheeling_active && !freewheel_brake_active) {
        // Vin slew veya diğer korumalar aktif - freewheel brake'i de aktifle!
        freewheel_brake_active = 1;
        freewheel_brake_timer = 0.0f;
    }
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // SECTION 3: DEACTIVATION LOGIC
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    else if (freewheel_brake_active && Abs_Float(raw_iind) < 5.0f) {
        // Inductor current dropped to safe level - deactivate after minimum time
        if (freewheel_brake_timer >= FREEWHEEL_BRAKE_MIN_TIME) {
            freewheel_brake_active = 0;
            freewheel_brake_timer = 0.0f;
        }
    }
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // SECTION 4: CRITICAL SAFETY OVERRIDE
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // FORCE OFF if safety interlocks violated (PWMs turn ON or load connects)
    // V8.4: freewheeling_active varken kapatma! Vin slew koruması öncelikli.
    else if ((!pwms_off || !load_disconnected) && !freewheeling_active) {
        freewheel_brake_active = 0;
        freewheel_brake_timer = 0.0f;
    }
    
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    // SECTION 5: TIMER UPDATE
    // ─────────────────────────────────────────────────────────────────────────────────────────────────────────────
    if (freewheel_brake_active) {
        freewheel_brake_timer += TS;
    }
    
    // ═══════════════════════════════════════════════════════════════════════════
    // V8.9: 5. çıkış = İŞLETİM MODU (0=Buck, 1=Buck-Boost, 2=Boost)
    // RATIONALE: Analog sinyal olarak mod bilgisi - freewheel artık hardware'de
    // ═══════════════════════════════════════════════════════════════════════════
    // Değerler: 0.0 = Buck, 1.0 = Buck-Boost (geçiş/güvenli), 2.0 = Boost
    float mode_output = 1.0f;  // Default: Buck-Boost (en güvenli başlangıç)
    if (ss.operating_mode == MODE_BUCK) {
        mode_output = 0.0f;
    } else if (ss.operating_mode == MODE_BOOST) {
        mode_output = 2.0f;
    } else {
        mode_output = 1.0f;  // Buck-Boost veya bilinmeyen
    }
    OutputSignal(0, 4) = mode_output;
}
#else
    // Freewheel brake disabled - yine de mod çıkışı ver
    float mode_output = 1.0f;
    if (ss.operating_mode == MODE_BUCK) {
        mode_output = 0.0f;
    } else if (ss.operating_mode == MODE_BOOST) {
        mode_output = 2.0f;
    }
    OutputSignal(0, 4) = mode_output;
#endif

// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// V8.3: DEBUG OUTPUTS - STATE MACHINE MONITORING
// ═══════════════════════════════════════════════════════════════════════════════════════════════════════════════════
// DIJITAL ÇIKIŞLAR (1.0 = aktif, 0.0 = pasif):
//   OutputSignal(0, 5) = Yüksüz (LOAD_STATE_YUKSUZ aktif mi?)
//   OutputSignal(0, 6) = Yüke Hazır (LOAD_STATE_YUKE_HAZIR aktif mi?)
//   OutputSignal(0, 7) = Yüke Bağlı (LOAD_STATE_YUKE_BAGLI aktif mi?)
// ANALOG ÇIKIŞ:
//   OutputSignal(0, 8) = Vout Hedef (pid_target_dynamic değeri)

// Yüksüz state (dijital)
OutputSignal(0, 5) = (load_state == LOAD_STATE_YUKSUZ) ? 1.0f : 0.0f;

// Yüke Hazır state (dijital)
OutputSignal(0, 6) = (load_state == LOAD_STATE_YUKE_HAZIR) ? 1.0f : 0.0f;

// Yüke Bağlı state (dijital)
OutputSignal(0, 7) = (load_state == LOAD_STATE_YUKE_BAGLI) ? 1.0f : 0.0f;

// Vout Hedef (analog - DAC ile okunacak)
OutputSignal(0, 8) = pid_target_dynamic;

// V8.6: GERÇEK PWM DUTY OUTPUTS (analog - DAC ile okunacak)
// V8.7: Anahtar Topolojisi:
//   Buck: HO (Q1) = Master, LO (Q2) = (1 - HO_duty) - dead_time
//   Boost: LO (Q3) = Master, HO (Q4) = (1 - LO_duty) - dead_time
// V8.15: VIN OUT-OF-RANGE SAFETY OVERRIDE
// 9V altında veya 40V üstünde freewheeling moduna zorla
if (vin_out_of_range) {
    timer_a_duty = DUTY_MIN;   // Buck HO OFF → Buck LO ON
    timer_b_duty = DUTY_MAX;   // Boost LO ON → Boost HO OFF
    // Bu konfigürasyon induktörü kısa devre yapar (freewheeling)
}


OutputSignal(0, 9) = (float)sys_state * 0.3f;   // 0=0V, 1=0.3V, 2=0.6V...
OutputSignal(0, 10) = (float)ss.phase * 0.3f;    // 0=0V, 1=0.3V, 2=0.6V...
OutputSignal(0, 11) = filt_vin * 0.02475f;       // 10V=0.2475V (Scope w/Gain: 0.825V)


OutputSignal(0, 12) = debug_p_term;      // i_term: ham değer
OutputSignal(0, 13) = debug_i_term;       // error: ham değer [V]
OutputSignal(0, 14) = debug_d_term + 1.65f; // d_term: 1.65V offset (0 = 1.65V)
OutputSignal(0, 15) = timer_a_duty;    // Buck HO (Master) Duty (0.0-1.0)
OutputSignal(0, 16) = timer_b_duty;   // Boost LO (Master) Duty (0.0-1.0)
